\section{Top-K từ xuất hiện nhiều nhất [Cơ bản]}
\subsection{Phân tích \& Thiết kế (M0)}

\begin{itemize}

    \item \textbf{Input:}

    \begin{itemize}
        \item File \texttt{text.txt} chứa nhiều dòng văn bản ASCII
        \item File \texttt{config.txt} chứa số nguyên $K$
    \end{itemize}

    \item \textbf{Xử lý:}

    \begin{itemize}

        \item \textbf{Case biên:}

        \begin{itemize}
            \item File rỗng $\rightarrow$ không có token
            \item $K >$ số token duy nhất $\rightarrow$ in tất cả
            \item Dòng chỉ punctuation $\rightarrow$ bỏ qua
        \end{itemize}

        \item \textbf{Module 1 (Đọc file text):}

        \begin{itemize}
            \item Hàm \texttt{readLines(path, vector<string>\&)}
            \item Kiểm tra mở file
        \end{itemize}

        \item \textbf{Module 2 (Đọc config):}

        \begin{itemize}
            \item Hàm \texttt{readConfig(path, k)}
        \end{itemize}

        \item \textbf{Module 3 (Normalize chuỗi):}

        \begin{itemize}
            \item Hàm \texttt{normalizeLine(const string\& s)}
            \item Chuyển lowercase
            \item Ký tự không phải chữ/số $\rightarrow$ space
            \item Gom nhiều space thành 1
        \end{itemize}

        \item \textbf{Module 4 (Tokenize):}

        \begin{itemize}
            \item Hàm \texttt{tokenize(const string\& normalized)}
            \item Tách token theo khoảng trắng
        \end{itemize}

        \item \textbf{Module 5 (Đếm tần suất):}

        \begin{itemize}
            \item Dùng:
\begin{lstlisting}
unordered_map<string,int> freq
\end{lstlisting}
        \end{itemize}

        \item \textbf{Module 6 (Top-K):}

        \begin{itemize}
            \item Chuyển map sang:
\begin{lstlisting}
vector<pair<string,int>>
\end{lstlisting}

            \item Sort:
            \begin{itemize}
                \item Giảm dần theo count
                \item Nếu bằng count $\rightarrow$ tăng dần theo từ
            \end{itemize}

            \item Hàm:
\begin{lstlisting}
vector<pair<string,int>> topK(...)
\end{lstlisting}
        \end{itemize}

        \item \textbf{Module 7 (Ghi output):}

        \begin{itemize}
            \item Ghi file \texttt{freq.csv}
            \item Format:
\begin{lstlisting}
word,count
\end{lstlisting}
        \end{itemize}

    \end{itemize}

    \item \textbf{Output:}

\begin{lstlisting}
word,count
\end{lstlisting}

\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}

\begin{lstlisting}
#include <iostream>
#include <vector>
#include <fstream>
#include <string>
#include <unordered_map>
#include <sstream>
#include <algorithm>
#include <cctype>

using namespace std;

bool readLines(const string& path,
               vector<string>& lines) {

    ifstream fin(path);

    if (!fin) return false;

    string s;

    while (getline(fin, s)) {

        lines.push_back(s);
    }

    return true;
}

bool readConfig(const string& path,
                int& k) {

    ifstream fin(path);

    if (!fin) return false;

    fin >> k;

    return true;
}

string normalizeLine(const string& s) {

    string result;

    bool lastSpace = true;

    for (char c : s) {

        if (isalnum((unsigned char)c)) {

            result +=
                (char)tolower((unsigned char)c);

            lastSpace = false;
        }
        else {

            if (!lastSpace) {

                result += ' ';
                lastSpace = true;
            }
        }
    }

    if (!result.empty() &&
        result.back() == ' ') {

        result.pop_back();
    }

    return result;
}

vector<string> tokenize(
    const string& normalized) {

    vector<string> tokens;

    stringstream ss(normalized);

    string token;

    while (ss >> token) {

        tokens.push_back(token);
    }

    return tokens;
}

vector<pair<string,int>> topK(
    unordered_map<string,int> freq,
    int k) {

    vector<pair<string,int>> v(
        freq.begin(),
        freq.end()
    );

    sort(v.begin(),
         v.end(),
         [](const pair<string,int>& a,
            const pair<string,int>& b) {

        if (a.second != b.second)
            return a.second > b.second;

        return a.first < b.first;
    });

    if ((int)v.size() > k) {
        v.resize(k);
    }

    return v;
}

bool writeFreq(
    const string& path,
    const vector<pair<string,int>>& v) {

    ofstream fout(path);

    if (!fout) return false;

    for (const auto& p : v) {

        fout << p.first
             << ","
             << p.second
             << "\n";
    }

    return true;
}

int main() {

    vector<string> lines;

    if (!readLines("text.txt", lines)) {

        cerr << "Loi mo file input!\n";
        return 1;
    }

    int k;

    if (!readConfig("config.txt", k)) {

        cerr << "Loi mo file config!\n";
        return 1;
    }

    unordered_map<string,int> freq;

    for (const string& s : lines) {

        string normalized =
            normalizeLine(s);

        vector<string> tokens =
            tokenize(normalized);

        for (const string& token : tokens) {

            freq[token]++;
        }
    }

    vector<pair<string,int>> result =
        topK(freq, k);

    if (!writeFreq("freq.csv", result)) {

        cerr << "Loi mo file output!\n";
        return 1;
    }

    return 0;
}
\end{lstlisting}
