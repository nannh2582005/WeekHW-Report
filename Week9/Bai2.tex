\section{Tokenize và đếm từ [Cơ bản]}
\subsection{Phân tích \& Thiết kế (M0)}

\begin{itemize}

    \item \textbf{Input:}

    File \texttt{text.txt} chứa nhiều dòng văn bản ASCII.

    \item \textbf{Xử lý:}

    \begin{itemize}

        \item \textbf{Case biên:}

        \begin{itemize}
            \item File rỗng $\rightarrow$ total=0, unique=0
            \item Dòng rỗng $\rightarrow$ không sinh token
            \item Dòng chỉ punctuation $\rightarrow$ không sinh token
        \end{itemize}

        \item \textbf{Module 1 (Đọc file):}

        \begin{itemize}
            \item Hàm \texttt{readLines(path, vector<string>\&)}
            \item Kiểm tra mở file
        \end{itemize}

        \item \textbf{Module 2 (Normalize chuỗi):}

        \begin{itemize}
            \item Hàm \texttt{normalizeLine(const string\& s)}
            \item Chuyển lowercase
            \item Ký tự không phải chữ/số $\rightarrow$ space
            \item Gom nhiều space thành 1
        \end{itemize}

        \item \textbf{Module 3 (Tokenize):}

        \begin{itemize}
            \item Hàm \texttt{tokenize(const string\& normalized)}
            \item Tách token theo khoảng trắng
        \end{itemize}

        \item \textbf{Module 4 (Đếm tần suất):}

        \begin{itemize}
            \item Dùng:
\begin{lstlisting}
unordered_map<string,int> freq
\end{lstlisting}

            \item Đếm:
            \begin{itemize}
                \item Tổng số token
                \item Số token duy nhất
            \end{itemize}
        \end{itemize}

    \end{itemize}

    \item \textbf{Output:}

\begin{lstlisting}
total_tokens=<n> / unique_tokens=<m>
\end{lstlisting}

\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}

\begin{lstlisting}
#include <iostream>
#include <vector>
#include <fstream>
#include <string>
#include <unordered_map>
#include <sstream>
#include <cctype>

using namespace std;

bool readLines(const string& path,
               vector<string>& lines) {

    ifstream fin(path);

    if (!fin) return false;

    string s;

    while (getline(fin, s)) {
        lines.push_back(s);
    }

    return true;
}

string normalizeLine(const string& s) {

    string result;

    bool lastSpace = true;

    for (char c : s) {

        if (isalnum((unsigned char)c)) {

            result +=
                (char)tolower((unsigned char)c);

            lastSpace = false;
        }
        else {

            if (!lastSpace) {

                result += ' ';
                lastSpace = true;
            }
        }
    }

    if (!result.empty() &&
        result.back() == ' ') {

        result.pop_back();
    }

    return result;
}

vector<string> tokenize(
    const string& normalized) {

    vector<string> tokens;

    stringstream ss(normalized);

    string token;

    while (ss >> token) {

        tokens.push_back(token);
    }

    return tokens;
}

int main() {

    vector<string> lines;

    if (!readLines("text.txt", lines)) {

        cerr << "Loi mo file input!\n";
        return 1;
    }

    unordered_map<string, int> freq;

    int totalTokens = 0;

    for (const string& s : lines) {

        string normalized =
            normalizeLine(s);

        vector<string> tokens =
            tokenize(normalized);

        for (const string& token : tokens) {

            freq[token]++;
            totalTokens++;
        }
    }

    cout << "total_tokens="
         << totalTokens
         << " / unique_tokens="
         << freq.size();

    return 0;
}
\end{lstlisting}
