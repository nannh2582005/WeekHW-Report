\section{Nhận diện ngôn ngữ đơn giản [Trung bình]}
\subsection{Phân tích \& Thiết kế (M0)}

\begin{itemize}

    \item \textbf{Input:}

    File \texttt{text.txt} chứa văn bản.

    \item \textbf{Xử lý:}

    \begin{itemize}

        \item \textbf{Case biên:}

        \begin{itemize}
            \item File rỗng
            $\rightarrow$
            \texttt{UNKNOWN}
            \item Chỉ có số/punctuation
            $\rightarrow$
            \texttt{UNKNOWN}
            \item Có nhiều ký tự tiếng Việt
            $\rightarrow$
            \texttt{VI\_SIMPLE}
        \end{itemize}

        \item \textbf{Module 1 (Đọc file):}

        \begin{itemize}
            \item Hàm:
\begin{lstlisting}
readLines(path, vector<string>&)
\end{lstlisting}

            \item Kiểm tra mở file
        \end{itemize}

        \item \textbf{Module 2 (Đếm ký tự):}

        \begin{itemize}
            \item Đếm:
            \begin{itemize}
                \item Tổng ký tự chữ
                \item Số ký tự ngoài ASCII
            \end{itemize}

            \item ASCII chuẩn:
\begin{lstlisting}
32 -> 126
\end{lstlisting}
        \end{itemize}

        \item \textbf{Module 3 (Phân loại):}

        \begin{itemize}

            \item Nếu:
\begin{lstlisting}
outside_ascii > 5% total_chars
\end{lstlisting}

            $\rightarrow$
            \texttt{VI\_SIMPLE}

            \item Nếu chỉ có:
\begin{lstlisting}
a-z A-Z
\end{lstlisting}

            $\rightarrow$
            \texttt{EN}

            \item Ngược lại:
\begin{lstlisting}
UNKNOWN
\end{lstlisting}

        \end{itemize}

        \item \textbf{Module 4 (Ghi output):}

        \begin{itemize}
            \item File:
\begin{lstlisting}
lang.txt
\end{lstlisting}
        \end{itemize}

    \end{itemize}

    \item \textbf{Output:}

\begin{lstlisting}
EN
\end{lstlisting}

hoặc

\begin{lstlisting}
VI_SIMPLE
\end{lstlisting}

hoặc

\begin{lstlisting}
UNKNOWN
\end{lstlisting}

\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}

\begin{lstlisting}
#include <iostream>
#include <vector>
#include <fstream>
#include <string>
#include <cctype>

using namespace std;

bool readLines(const string& path,
               vector<string>& lines) {

    ifstream fin(path);

    if (!fin) return false;

    string s;

    while (getline(fin, s)) {

        lines.push_back(s);
    }

    return true;
}

bool writeLang(const string& path,
               const string& lang) {

    ofstream fout(path);

    if (!fout) return false;

    fout << lang;

    return true;
}

int main() {

    vector<string> lines;

    if (!readLines("text.txt", lines)) {

        cerr << "Loi mo file input!\n";
        return 1;
    }

    int totalChars = 0;
    int outsideASCII = 0;
    int englishLetters = 0;

    for (const string& s : lines) {

        for (unsigned char c : s) {

            if (isalpha(c)) {

                totalChars++;

                if (c >= 32 && c <= 126) {

                    if ((c >= 'a' && c <= 'z') ||
                        (c >= 'A' && c <= 'Z')) {

                        englishLetters++;
                    }
                }
                else {

                    outsideASCII++;
                }
            }
        }
    }

    string lang = "UNKNOWN";

    if (totalChars == 0) {

        lang = "UNKNOWN";
    }
    else {

        double ratio =
            (double)outsideASCII
            / totalChars;

        if (ratio > 0.05) {

            lang = "VI_SIMPLE";
        }
        else if (englishLetters ==
                 totalChars) {

            lang = "EN";
        }
    }

    if (!writeLang("lang.txt", lang)) {

        cerr << "Loi mo file output!\n";
        return 1;
    }

    return 0;
}
\end{lstlisting}
