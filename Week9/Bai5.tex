\subsection{Phân tích \& Thiết kế (M0)}

\begin{itemize}

    \item \textbf{Input:}

    \begin{itemize}
        \item File \texttt{file1.txt}
        \item File \texttt{file2.txt}
    \end{itemize}

    \item \textbf{Xử lý:}

    \begin{itemize}

        \item \textbf{Case biên:}

        \begin{itemize}
            \item Một file rỗng $\rightarrow$ toàn bộ từ nằm ở file còn lại
            \item Hai file giống hệt nhau
            $\rightarrow$
            only\_in\_1=0, only\_in\_2=0
            \item Dòng chỉ punctuation $\rightarrow$ bỏ qua
        \end{itemize}

        \item \textbf{Module 1 (Đọc file):}

        \begin{itemize}
            \item Hàm:
\begin{lstlisting}
readLines(path, vector<string>&)
\end{lstlisting}

            \item Kiểm tra mở file
        \end{itemize}

        \item \textbf{Module 2 (Normalize chuỗi):}

        \begin{itemize}
            \item Hàm:
\begin{lstlisting}
normalizeLine(const string& s)
\end{lstlisting}

            \item Chuyển lowercase
            \item Ký tự không phải chữ/số $\rightarrow$ space
            \item Gom nhiều space thành 1
        \end{itemize}

        \item \textbf{Module 3 (Tokenize):}

        \begin{itemize}
            \item Hàm:
\begin{lstlisting}
vector<string> tokenize(...)
\end{lstlisting}
        \end{itemize}

        \item \textbf{Module 4 (Đếm tần suất):}

        \begin{itemize}
            \item Dùng:
\begin{lstlisting}
unordered_map<string,int>
\end{lstlisting}

            cho mỗi file
        \end{itemize}

        \item \textbf{Module 5 (Merge và so sánh):}

        \begin{itemize}
            \item Hàm:
\begin{lstlisting}
mergeFreq(map1, map2, output)
\end{lstlisting}

            \item Duyệt tất cả key của 2 map
            \item Xác định:
            \begin{itemize}
                \item Chỉ có ở file1
                \item Chỉ có ở file2
                \item Có ở cả 2
            \end{itemize}
        \end{itemize}

        \item \textbf{Module 6 (Ghi output):}

        \begin{itemize}
            \item Ghi file:
\begin{lstlisting}
comparison.csv
\end{lstlisting}

            \item Format:
\begin{lstlisting}
word,count_file1,count_file2
\end{lstlisting}

            \item In summary:
\begin{lstlisting}
only_in_1=<n>
only_in_2=<m>
in_both=<k>
\end{lstlisting}
        \end{itemize}

    \end{itemize}

    \item \textbf{Output:}

\begin{lstlisting}
only_in_1=<n>
only_in_2=<m>
in_both=<k>
\end{lstlisting}

\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}

\begin{lstlisting}
#include <iostream>
#include <vector>
#include <fstream>
#include <string>
#include <unordered_map>
#include <unordered_set>
#include <sstream>
#include <cctype>

using namespace std;

struct Row {
    string word;
    int count1;
    int count2;
};

bool readLines(const string& path,
               vector<string>& lines) {

    ifstream fin(path);

    if (!fin) return false;

    string s;

    while (getline(fin, s)) {

        lines.push_back(s);
    }

    return true;
}

string normalizeLine(const string& s) {

    string result;

    bool lastSpace = true;

    for (char c : s) {

        if (isalnum((unsigned char)c)) {

            result +=
                (char)tolower((unsigned char)c);

            lastSpace = false;
        }
        else {

            if (!lastSpace) {

                result += ' ';
                lastSpace = true;
            }
        }
    }

    if (!result.empty() &&
        result.back() == ' ') {

        result.pop_back();
    }

    return result;
}

vector<string> tokenize(
    const string& normalized) {

    vector<string> tokens;

    stringstream ss(normalized);

    string token;

    while (ss >> token) {

        tokens.push_back(token);
    }

    return tokens;
}

unordered_map<string,int>
buildFreq(const vector<string>& lines) {

    unordered_map<string,int> freq;

    for (const string& s : lines) {

        string normalized =
            normalizeLine(s);

        vector<string> tokens =
            tokenize(normalized);

        for (const string& token : tokens) {

            freq[token]++;
        }
    }

    return freq;
}

void mergeFreq(
    const unordered_map<string,int>& map1,
    const unordered_map<string,int>& map2,
    vector<Row>& output,
    int& only1,
    int& only2,
    int& both) {

    unordered_set<string> allWords;

    for (const auto& p : map1) {

        allWords.insert(p.first);
    }

    for (const auto& p : map2) {

        allWords.insert(p.first);
    }

    only1 = 0;
    only2 = 0;
    both = 0;

    for (const string& word : allWords) {

        int c1 = 0;
        int c2 = 0;

        auto it1 = map1.find(word);

        if (it1 != map1.end()) {
            c1 = it1->second;
        }

        auto it2 = map2.find(word);

        if (it2 != map2.end()) {
            c2 = it2->second;
        }

        if (c1 > 0 && c2 == 0)
            only1++;

        else if (c1 == 0 && c2 > 0)
            only2++;

        else
            both++;

        output.push_back({word, c1, c2});
    }
}

bool writeComparison(
    const string& path,
    const vector<Row>& rows) {

    ofstream fout(path);

    if (!fout) return false;

    for (const Row& r : rows) {

        fout << r.word
             << ","
             << r.count1
             << ","
             << r.count2
             << "\n";
    }

    return true;
}

int main() {

    vector<string> lines1;
    vector<string> lines2;

    if (!readLines("file1.txt", lines1)) {

        cerr << "Loi mo file1!\n";
        return 1;
    }

    if (!readLines("file2.txt", lines2)) {

        cerr << "Loi mo file2!\n";
        return 1;
    }

    unordered_map<string,int> freq1 =
        buildFreq(lines1);

    unordered_map<string,int> freq2 =
        buildFreq(lines2);

    vector<Row> rows;

    int only1, only2, both;

    mergeFreq(freq1,
              freq2,
              rows,
              only1,
              only2,
              both);

    if (!writeComparison(
            "comparison.csv",
            rows)) {

        cerr << "Loi mo file output!\n";
        return 1;
    }

    cout << "only_in_1="
         << only1
         << "\n";

    cout << "only_in_2="
         << only2
         << "\n";

    cout << "in_both="
         << both;

    return 0;
}
