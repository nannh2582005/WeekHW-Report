\section{Lọc stop-words [Trung bình]}
\subsection{Phân tích \& Thiết kế (M0)}

\begin{itemize}

    \item \textbf{Input:}

    \begin{itemize}
        \item File \texttt{text.txt} chứa nhiều dòng văn bản ASCII
        \item File \texttt{stopwords.txt} chứa stop-word
        \item Mỗi dòng trong \texttt{stopwords.txt} chứa 1 từ
    \end{itemize}

    \item \textbf{Xử lý:}

    \begin{itemize}

        \item \textbf{Case biên:}

        \begin{itemize}
            \item File text rỗng $\rightarrow$ tất cả thống kê bằng 0
            \item File stopwords rỗng $\rightarrow$ giữ toàn bộ token
            \item Dòng chỉ punctuation $\rightarrow$ bỏ qua
        \end{itemize}

        \item \textbf{Module 1 (Đọc text):}

        \begin{itemize}
            \item Hàm \texttt{readLines(path, vector<string>\&)}
            \item Kiểm tra mở file
        \end{itemize}

        \item \textbf{Module 2 (Đọc stopwords):}

        \begin{itemize}
            \item Hàm:
\begin{lstlisting}
readStopwords(path, unordered_set<string>&)
\end{lstlisting}

            \item Lưu stop-word vào:
\begin{lstlisting}
unordered_set<string> stops
\end{lstlisting}
        \end{itemize}

        \item \textbf{Module 3 (Normalize chuỗi):}

        \begin{itemize}
            \item Hàm \texttt{normalizeLine(const string\& s)}
            \item Chuyển lowercase
            \item Ký tự không phải chữ/số $\rightarrow$ space
            \item Gom nhiều space thành 1
        \end{itemize}

        \item \textbf{Module 4 (Tokenize):}

        \begin{itemize}
            \item Hàm:
\begin{lstlisting}
vector<string> tokenize(...)
\end{lstlisting}
        \end{itemize}

        \item \textbf{Module 5 (Lọc stop-word và đếm):}

        \begin{itemize}
            \item Nếu:
\begin{lstlisting}
stops.count(token) == 0
\end{lstlisting}

            $\rightarrow$ đưa vào thống kê

            \item Dùng:
\begin{lstlisting}
unordered_map<string,int> freq
\end{lstlisting}
        \end{itemize}

        \item \textbf{Module 6 (Ghi output):}

        \begin{itemize}
            \item File \texttt{freq\_filtered.csv}
            \item Format:
\begin{lstlisting}
word,count
\end{lstlisting}

            \item In summary:
\begin{lstlisting}
total_before=<n>
total_after=<m>
stopwords_removed=<k>
\end{lstlisting}
        \end{itemize}

    \end{itemize}

    \item \textbf{Output:}

\begin{lstlisting}
total_before=<n>
total_after=<m>
stopwords_removed=<k>
\end{lstlisting}

\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}

\begin{lstlisting}
#include <iostream>
#include <vector>
#include <fstream>
#include <string>
#include <unordered_map>
#include <unordered_set>
#include <sstream>
#include <cctype>

using namespace std;

bool readLines(const string& path,
               vector<string>& lines) {

    ifstream fin(path);

    if (!fin) return false;

    string s;

    while (getline(fin, s)) {

        lines.push_back(s);
    }

    return true;
}

bool readStopwords(
    const string& path,
    unordered_set<string>& stops) {

    ifstream fin(path);

    if (!fin) return false;

    string word;

    while (getline(fin, word)) {

        if (!word.empty()) {

            stops.insert(word);
        }
    }

    return true;
}

string normalizeLine(const string& s) {

    string result;

    bool lastSpace = true;

    for (char c : s) {

        if (isalnum((unsigned char)c)) {

            result +=
                (char)tolower((unsigned char)c);

            lastSpace = false;
        }
        else {

            if (!lastSpace) {

                result += ' ';
                lastSpace = true;
            }
        }
    }

    if (!result.empty() &&
        result.back() == ' ') {

        result.pop_back();
    }

    return result;
}

vector<string> tokenize(
    const string& normalized) {

    vector<string> tokens;

    stringstream ss(normalized);

    string token;

    while (ss >> token) {

        tokens.push_back(token);
    }

    return tokens;
}

bool writeFreq(
    const string& path,
    const unordered_map<string,int>& freq) {

    ofstream fout(path);

    if (!fout) return false;

    for (const auto& p : freq) {

        fout << p.first
             << ","
             << p.second
             << "\n";
    }

    return true;
}

int main() {

    vector<string> lines;

    if (!readLines("text.txt", lines)) {

        cerr << "Loi mo file text!\n";
        return 1;
    }

    unordered_set<string> stops;

    if (!readStopwords("stopwords.txt",
                       stops)) {

        cerr << "Loi mo file stopwords!\n";
        return 1;
    }

    unordered_map<string,int> freq;

    int totalBefore = 0;
    int totalAfter = 0;

    for (const string& s : lines) {

        string normalized =
            normalizeLine(s);

        vector<string> tokens =
            tokenize(normalized);

        for (const string& token : tokens) {

            totalBefore++;

            if (stops.count(token) == 0) {

                freq[token]++;
                totalAfter++;
            }
        }
    }

    int removed =
        totalBefore - totalAfter;

    if (!writeFreq("freq_filtered.csv",
                   freq)) {

        cerr << "Loi mo file output!\n";
        return 1;
    }

    cout << "total_before="
         << totalBefore
         << "\n";

    cout << "total_after="
         << totalAfter
         << "\n";

    cout << "stopwords_removed="
         << removed;

    return 0;
}
\end{lstlisting}
