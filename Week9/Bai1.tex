\section{Normalize chuỗi ASCII [Cơ bản]}
\subsection{Phân tích \& Thiết kế (M0)}

\begin{itemize}

    \item \textbf{Input:}

    File \texttt{text.txt} chứa nhiều dòng văn bản ASCII.

    \item \textbf{Xử lý:}

    \begin{itemize}

        \item \textbf{Case biên:}

        \begin{itemize}
            \item Dòng rỗng $\rightarrow$ ghi dòng rỗng
            \item Dòng chỉ chứa punctuation $\rightarrow$ ghi dòng rỗng
            \item Nhiều dấu cách liên tiếp $\rightarrow$ gom thành 1 space
        \end{itemize}

        \item \textbf{Module 1 (Đọc file):}

        \begin{itemize}
            \item Hàm \texttt{readLines(path, vector<string>\&)}
            \item Kiểm tra mở file
        \end{itemize}

        \item \textbf{Module 2 (Normalize chuỗi):}

        \begin{itemize}
            \item Hàm \texttt{normalizeLine(const string\& s)}
            \item Duyệt từng ký tự:
            \begin{itemize}
                \item \texttt{isalnum(c)} $\rightarrow$ giữ lại và chuyển lowercase
                \item Ký tự khác $\rightarrow$ đổi thành space
                \item Không ghi nhiều space liên tiếp
            \end{itemize}
        \end{itemize}

        \item \textbf{Module 3 (Xoá space đầu/cuối):}

        \begin{itemize}
            \item Nếu cuối chuỗi là space $\rightarrow$ xoá
        \end{itemize}

        \item \textbf{Module 4 (Ghi file output):}

        \begin{itemize}
            \item Ghi từng dòng vào \texttt{normalized.txt}
        \end{itemize}

    \end{itemize}

    \item \textbf{Output:}

\begin{lstlisting}
hello world 123
cpp basic
\end{lstlisting}

\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}

\begin{lstlisting}
#include <iostream>
#include <vector>
#include <fstream>
#include <string>
#include <cctype>

using namespace std;

bool readLines(const string& path,
               vector<string>& lines) {

    ifstream fin(path);

    if (!fin) return false;

    string s;

    while (getline(fin, s)) {
        lines.push_back(s);
    }

    return true;
}

string normalizeLine(const string& s) {

    string result;

    bool lastSpace = true;

    for (char c : s) {

        if (isalnum((unsigned char)c)) {

            result +=
                (char)tolower((unsigned char)c);

            lastSpace = false;
        }
        else {

            if (!lastSpace) {

                result += ' ';
                lastSpace = true;
            }
        }
    }

    if (!result.empty() &&
        result.back() == ' ') {

        result.pop_back();
    }

    return result;
}

bool writeLines(const string& path,
                const vector<string>& lines) {

    ofstream fout(path);

    if (!fout) return false;

    for (const string& s : lines) {

        fout << s << "\n";
    }

    return true;
}

int main() {

    vector<string> lines;

    if (!readLines("text.txt", lines)) {

        cerr << "Loi mo file input!\n";
        return 1;
    }

    vector<string> normalized;

    for (const string& s : lines) {

        normalized.push_back(
            normalizeLine(s)
        );
    }

    if (!writeLines("normalized.txt",
                    normalized)) {

        cerr << "Loi mo file output!\n";
        return 1;
    }

    return 0;
}
\end{lstlisting}
