\subsection{Kế hoạch kiểm thử (Test Cases)}

\textbf{Test Case 1 (Thông thường)}

\textbf{Input (index.txt):}
\begin{lstlisting}
doc1.txt
doc2.txt
\end{lstlisting}

\textbf{Input (doc1.txt):}
\begin{lstlisting}
apple banana apple
\end{lstlisting}

\textbf{Input (doc2.txt):}
\begin{lstlisting}
banana orange
\end{lstlisting}

\textbf{Giải thích tính toán:}
\begin{itemize}
    \item $N = 2$.
    \item \texttt{apple}: $df=1 \Rightarrow IDF = \ln(2/1) \approx 0.693147$. Trong doc1: $TF = 2/3 \Rightarrow TFIDF \approx 0.462098$.
    \item \texttt{banana}: $df=2 \Rightarrow IDF = \ln(2/2) = 0$.
    \item \texttt{orange}: $df=1 \Rightarrow IDF = \ln(2/1) \approx 0.693147$. Trong doc2: $TF = 1/2 \Rightarrow TFIDF \approx 0.346574$.
\end{itemize}

\textbf{Expected Output (tfidf.csv):}
\begin{lstlisting}
doc1.txt,apple,0.462098
doc1.txt,banana,0.000000
doc2.txt,orange,0.346574
doc2.txt,banana,0.000000
\end{lstlisting}
