\section{Ghi report CSV nhiều cột [Trung bình]}

\subsection{Phân tích \& Thiết kế (M0)}
\begin{itemize}
    \item \textbf{Input:}
    \begin{itemize}
        \item File \texttt{scores.csv}, mỗi dòng gồm 4 cột:
        \[
            \texttt{id, math, lit, sci}
        \]
        \item Các cột điểm có thể missing
        \item Missing được hiểu là ô trống hoặc \texttt{NA}/\texttt{NaN} (không phân biệt hoa/thường)
    \end{itemize}

    \item \textbf{Xử lý:}
    \begin{itemize}
        \item \textbf{Case biên:}
        \begin{itemize}
            \item Missing môn $\rightarrow$ coi là \texttt{0} khi tính tổng, nhưng vẫn đánh dấu trong record
            \item Tất cả môn đều missing $\rightarrow$ \texttt{total = 0} $\rightarrow$ xếp loại \texttt{Yeu}
            \item \texttt{id} rỗng $\rightarrow$ ghi \texttt{UNKNOWN}
            \item Dòng không đủ 4 cột $\rightarrow$ bỏ qua để chương trình chạy an toàn
            \item File rỗng $\rightarrow$ \texttt{result.csv} chỉ có header và dòng summary với tất cả count bằng 0
        \end{itemize}

        \item \textbf{Module 1 (Chuẩn hóa token):}
        \begin{itemize}
            \item Hàm \texttt{trim(string)}: bỏ khoảng trắng đầu/cuối
            \item Hàm \texttt{toLowerStr(string)}: chuyển về chữ thường
            \item Hàm \texttt{isMissingToken(string)}: kiểm tra missing token
        \end{itemize}

        \item \textbf{Module 2 (Tách \& parse 1 dòng CSV):}
        \begin{itemize}
            \item Hàm \texttt{splitCSV(line, vector<string>\&)}: tách dòng theo dấu phẩy
            \item Hàm \texttt{parseScore(cell, value, isMissing)}:
            \begin{itemize}
                \item nếu missing $\rightarrow$ gán \texttt{value = 0}, \texttt{isMissing = true}
                \item nếu parse được số $\rightarrow$ giữ giá trị thật
                \item nếu token không hợp lệ $\rightarrow$ xem như dòng lỗi
            \end{itemize}
            \item Hàm \texttt{parseRow(cells, Record\& rec)} tạo ra một record hoàn chỉnh
        \end{itemize}

        \item \textbf{Module 3 (Tính tổng \& xếp loại):}
        \begin{itemize}
            \item Tổng:
            \[
                \texttt{total = math + lit + sci}
            \]
            \item Hàm \texttt{computeRank(total)}:
            \begin{itemize}
                \item \texttt{total $\geq$ 24} $\rightarrow$ \texttt{Gioi}
                \item \texttt{total $\geq$ 18} $\rightarrow$ \texttt{Kha}
                \item \texttt{total $\geq$ 12} $\rightarrow$ \texttt{Trung binh}
                \item \texttt{total < 12} $\rightarrow$ \texttt{Yeu}
            \end{itemize}
        \end{itemize}

        \item \textbf{Module 4 (Đọc toàn bộ file):}
        \begin{itemize}
            \item Hàm \texttt{readScores(path, vector<Record>\&)}
            \item Đọc từng dòng bằng \texttt{getline}
            \item Với mỗi dòng hợp lệ:
            \begin{itemize}
                \item parse record
                \item tính tổng
                \item gán rank
                \item đưa vào danh sách kết quả
            \end{itemize}
        \end{itemize}

        \item \textbf{Module 5 (Ghi report \& summary):}
        \begin{itemize}
            \item Hàm \texttt{writeResult(path, records)}
            \item Ghi file \texttt{result.csv} gồm các cột:
            \[
                \texttt{id, total, rank}
            \]
            \item Cuối file ghi thêm 1 dòng summary đếm số record theo từng xếp loại
            \item Số thực dùng \texttt{fixed << setprecision(2)}
        \end{itemize}
    \end{itemize}

    \item \textbf{Output:}
    \begin{itemize}
        \item File \texttt{result.csv}
        \item Mỗi dòng dữ liệu có dạng:
        \begin{center}
            \texttt{id,total,rank}
        \end{center}
        \item Dòng cuối là summary:
        \begin{center}
            \texttt{SUMMARY,count\_gioi,count\_kha,count\_trung\_binh,count\_yeu}
        \end{center}
    \end{itemize}
\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}
\begin{lstlisting}
#include <iostream>
#include <fstream>
#include <sstream>
#include <vector>
#include <string>
#include <iomanip>
#include <cctype>

using namespace std;

struct Record {
    string id;
    double math;
    double lit;
    double sci;
    bool missingMath;
    bool missingLit;
    bool missingSci;
    double total;
    string rank;
};

// MODULE: Trim
string trim(const string& s) {
    size_t start = s.find_first_not_of(" \t\r\n");
    size_t end = s.find_last_not_of(" \t\r\n");

    if (start == string::npos) return "";
    return s.substr(start, end - start + 1);
}

// MODULE: To lower
string toLowerStr(string s) {
    for (char& c : s) {
        c = (char)tolower((unsigned char)c);
    }
    return s;
}

// MODULE: Check missing token
bool isMissingToken(string t) {
    t = toLowerStr(trim(t));
    return t.empty() || t == "na" || t == "nan";
}

// MODULE: Parse double safely
bool parseDouble(const string& s, double& value) {
    string t = trim(s);
    if (t.empty()) return false;

    try {
        size_t idx;
        value = stod(t, &idx);
        return idx == t.size();
    } catch (...) {
        return false;
    }
}

// MODULE 1: Split CSV
void splitCSV(const string& line, vector<string>& cells) {
    cells.clear();
    stringstream ss(line);
    string cell;

    while (getline(ss, cell, ',')) {
        cells.push_back(cell);
    }
}

// MODULE 2: Parse one score cell
bool parseScore(const string& cell, double& value, bool& isMissing) {
    if (isMissingToken(cell)) {
        value = 0.0;
        isMissing = true;
        return true;
    }

    isMissing = false;
    return parseDouble(cell, value);
}

// MODULE 3: Compute rank
string computeRank(double total) {
    if (total >= 24.0) return "Gioi";
    if (total >= 18.0) return "Kha";
    if (total >= 12.0) return "Trung binh";
    return "Yeu";
}

// MODULE 4: Parse one row -> Record
bool parseRow(const vector<string>& cells, Record& rec) {
    if (cells.size() != 4) return false;

    rec.id = trim(cells[0]);
    if (rec.id.empty()) rec.id = "UNKNOWN";

    if (!parseScore(cells[1], rec.math, rec.missingMath)) return false;
    if (!parseScore(cells[2], rec.lit, rec.missingLit)) return false;
    if (!parseScore(cells[3], rec.sci, rec.missingSci)) return false;

    rec.total = rec.math + rec.lit + rec.sci;
    rec.rank = computeRank(rec.total);

    return true;
}

// MODULE 5: Read whole file
bool readScores(string path, vector<Record>& records) {
    ifstream file(path);
    if (!file.is_open()) return false;

    string line;
    while (getline(file, line)) {
        if (trim(line).empty()) continue;

        vector<string> cells;
        splitCSV(line, cells);

        Record rec;
        if (parseRow(cells, rec)) {
            records.push_back(rec);
        }
    }

    file.close();
    return true;
}

// MODULE 6: Write result.csv
bool writeResult(string path, const vector<Record>& records) {
    ofstream file(path);
    if (!file.is_open()) return false;

    int countGioi = 0;
    int countKha = 0;
    int countTrungBinh = 0;
    int countYeu = 0;

    file << "id,total,rank\n";
    file << fixed << setprecision(2);

    for (size_t i = 0; i < records.size(); i++) {
        file << records[i].id << ","
             << records[i].total << ","
             << records[i].rank << "\n";

        if (records[i].rank == "Gioi") countGioi++;
        else if (records[i].rank == "Kha") countKha++;
        else if (records[i].rank == "Trung binh") countTrungBinh++;
        else countYeu++;
    }

    file << "SUMMARY,"
         << countGioi << ","
         << countKha << ","
         << countTrungBinh << ","
         << countYeu;

    file.close();
    return true;
}

// MAIN
int main() {
    vector<Record> records;

    if (!readScores("scores.csv", records)) {
        cerr << "Khong mo duoc file scores.csv\n";
        return 1;
    }

    if (!writeResult("result.csv", records)) {
        cerr << "Khong ghi duoc file result.csv\n";
        return 1;
    }

    return 0;
}
\end{lstlisting}

\begin{lstlisting}
SV01,8,9,8
SV02,6,6,7
SV03,4,5,4
\end{lstlisting}

\textbf{Phân tích:}
\begin{itemize}
    \item \texttt{SV01}: tổng \texttt{= 25} $\rightarrow$ \texttt{Gioi}
    \item \texttt{SV02}: tổng \texttt{= 19} $\rightarrow$ \texttt{Kha}
    \item \texttt{SV03}: tổng \texttt{= 13} $\rightarrow$ \texttt{Trung binh}
\end{itemize}

\begin{lstlisting}
id,total,rank
SV01,25.00,Gioi
SV02,19.00,Kha

SUMMARY,1,1,1,0
\end{lstlisting}
