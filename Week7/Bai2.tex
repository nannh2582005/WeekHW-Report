\section{Cộng và trừ 2 phân số [Cơ bản]}
\subsection{Phân tích \& Thiết kế (M0)}
\begin{itemize}
    \item \textbf{Input:}
    File \texttt{pairs.csv} chứa nhiều dòng:
    \begin{itemize}
        \item Mỗi dòng gồm 4 số: \texttt{n1,d1,n2,d2}
        \item Tương ứng hai phân số $\frac{n1}{d1}$ và $\frac{n2}{d2}$
    \end{itemize}

    \item \textbf{Xử lý:}
    \begin{itemize}
        \item \textbf{Case biên:}
        \begin{itemize}
            \item Hai phân số nghịch nhau $\rightarrow$ tổng bằng \texttt{0,1}
            \item Nếu tử số bằng 0 $\rightarrow$ chuẩn hoá thành \texttt{0,1}
            \item Nếu mẫu âm $\rightarrow$ đổi dấu cả tử và mẫu
        \end{itemize}

        \item \textbf{Module 1 (Đọc file):}
        \begin{itemize}
            \item Hàm \texttt{readPairs(path, vector<PairFrac>\&)}
            \item Đọc từng dòng CSV
            \item Kiểm tra mở file (\texttt{fail} $\rightarrow$ \texttt{return false})
        \end{itemize}

        \item \textbf{Module 2 (Ước chung lớn nhất):}
        \begin{itemize}
            \item Thuật toán Euclid để rút gọn phân số
        \end{itemize}

        \item \textbf{Module 3 (Chuẩn hoá phân số):}
        \begin{itemize}
            \item Đưa mẫu về dương
            \item Rút gọn gcd
            \item $0/x \rightarrow 0/1$
        \end{itemize}

        \item \textbf{Module 4 (Cộng và trừ):}
        \begin{itemize}
            \item
            \begin{lstlisting}
a/b + c/d = (a*d + c*b)/(b*d)
a/b - c/d = (a*d - c*b)/(b*d)
            \end{lstlisting}
            \item Sau đó chuẩn hoá kết quả
        \end{itemize}

        \item \textbf{Module 5 (Ghi file):}
        \begin{itemize}
            \item Ghi \texttt{results.csv}
            \item Định dạng:
            \texttt{sum\_num,sum\_den,diff\_num,diff\_den}
        \end{itemize}
    \end{itemize}

    \item \textbf{Output:}
    \begin{lstlisting}
Ghi file results.csv
    \end{lstlisting}
\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}
\begin{lstlisting}
#include <iostream>
#include <vector>
#include <fstream>
#include <sstream>
#include <cstdlib>

using namespace std;

struct Fraction {
    long long num, den;
};

struct PairFrac {
    Fraction a, b;
};

long long gcd(long long a, long long b) {
    a = llabs(a);
    b = llabs(b);

    while (b != 0) {
        long long t = a % b;
        a = b;
        b = t;
    }

    return a;
}

void normalize(Fraction& f) {
    if (f.num == 0) {
        f.num = 0;
        f.den = 1;
        return;
    }

    if (f.den < 0) {
        f.num = -f.num;
        f.den = -f.den;
    }

    long long g = gcd(f.num, f.den);
    f.num /= g;
    f.den /= g;
}

bool readPairs(const string& path,
               vector<PairFrac>& arr) {
    ifstream fin(path);
    if (!fin) return false;

    string line;
    while (getline(fin, line)) {
        if (line.empty()) continue;

        stringstream ss(line);
        string s[4];

        for (int i = 0; i < 4; ++i)
            getline(ss, s[i], i == 3 ? '\n' : ',');

        PairFrac p;
        p.a.num = atoll(s[0].c_str());
        p.a.den = atoll(s[1].c_str());
        p.b.num = atoll(s[2].c_str());
        p.b.den = atoll(s[3].c_str());

        arr.push_back(p);
    }

    return true;
}

Fraction addFrac(Fraction x, Fraction y) {
    Fraction r;
    r.num = x.num * y.den + y.num * x.den;
    r.den = x.den * y.den;
    normalize(r);
    return r;
}

Fraction subFrac(Fraction x, Fraction y) {
    Fraction r;
    r.num = x.num * y.den - y.num * x.den;
    r.den = x.den * y.den;
    normalize(r);
    return r;
}

bool writeResults(const string& path,
                  const vector<PairFrac>& arr) {
    ofstream fout(path);
    if (!fout) return false;

    for (size_t i = 0; i < arr.size(); ++i) {
        Fraction s = addFrac(arr[i].a, arr[i].b);
        Fraction d = subFrac(arr[i].a, arr[i].b);

        fout << s.num << "," << s.den << ","
             << d.num << "," << d.den << "\n";
    }

    return true;
}

int main() {
    vector<PairFrac> arr;

    if (!readPairs("pairs.csv", arr)) {
        cerr << "Loi mo file input!\n";
        return 1;
    }

    if (!writeResults("results.csv", arr)) {
        cerr << "Loi mo file output!\n";
        return 1;
    }

    return 0;
}
\end{lstlisting}
