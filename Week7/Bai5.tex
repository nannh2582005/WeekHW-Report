\section{Lọc phân số trong khoảng [a, b][Trung bình]}
\subsection{Phân tích \& Thiết kế (M0)}
\begin{itemize}
    \item \textbf{Input:}
    \begin{itemize}
        \item File \texttt{fractions.csv}: mỗi dòng dạng \texttt{num,den}
        \item File \texttt{config.txt}: \texttt{a\_num,a\_den,b\_num,b\_den}
    \end{itemize}

    \item \textbf{Xử lý:}
    \begin{itemize}
        \item \textbf{Case biên:}
        \begin{itemize}
            \item \texttt{den = 0} $\rightarrow$ phân số không hợp lệ, bỏ qua
            \item \texttt{num = 0} $\rightarrow$ chuẩn hoá thành \texttt{0/1}
            \item \texttt{den < 0} $\rightarrow$ đổi dấu cả tử và mẫu
            \item Nếu \texttt{lo > hi} $\rightarrow$ khoảng rỗng, \texttt{in\_range = 0}
        \end{itemize}

        \item \textbf{Module 1 (Đọc file):}
        \begin{itemize}
            \item Hàm \texttt{readFractions(path, vector<Fraction>\&)}
            \item Hàm \texttt{readConfig(path, lo, hi)}
            \item Kiểm tra mở file (\texttt{fail} $\rightarrow$ \texttt{return false})
        \end{itemize}

        \item \textbf{Module 2 (Chuẩn hoá phân số):}
        \begin{itemize}
            \item Lọc invalid
            \item Đưa mẫu về dương
            \item Rút gọn gcd
        \end{itemize}

        \item \textbf{Module 3 (So sánh phân số):}
        \begin{itemize}
            \item
            \begin{lstlisting}
a/b < c/d <=> a*d < c*b
            \end{lstlisting}
            \item Dùng \texttt{\_\_int128} tránh tràn số
        \end{itemize}

        \item \textbf{Module 4 (Lọc trong khoảng):}
        \begin{itemize}
            \item
            \begin{lstlisting}
f ∈ [lo,hi] <=> cmpFrac(lo,f)<=0
             && cmpFrac(f,hi)<=0
            \end{lstlisting}
        \end{itemize}

        \item \textbf{Module 5 (Sắp xếp và ghi file):}
        \begin{itemize}
            \item Sắp xếp tăng dần
            \item Ghi \texttt{filtered.csv}: \texttt{num,den}
        \end{itemize}

        \item \textbf{Module 6 (Thống kê):}
        \begin{itemize}
            \item \texttt{total\_valid}
            \item \texttt{in\_range}
            \item \texttt{out\_of\_range}
        \end{itemize}
    \end{itemize}

    \item \textbf{Output:}
    \begin{lstlisting}
Ghi file filtered.csv
total_valid=<n> / in_range=<m> / out_of_range=<k>
    \end{lstlisting}
\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}
\begin{lstlisting}
#include <iostream>
#include <vector>
#include <fstream>
#include <sstream>
#include <algorithm>
#include <cstdlib>

using namespace std;

struct Fraction {
    long long num, den;
};

long long gcd(long long a, long long b) {
    a = llabs(a);
    b = llabs(b);

    while (b != 0) {
        long long t = a % b;
        a = b;
        b = t;
    }
    return a;
}

bool normalize(Fraction& f) {
    if (f.den == 0) return false;

    if (f.num == 0) {
        f.num = 0;
        f.den = 1;
        return true;
    }

    if (f.den < 0) {
        f.num = -f.num;
        f.den = -f.den;
    }

    long long g = gcd(f.num, f.den);
    f.num /= g;
    f.den /= g;
    return true;
}

bool readFractions(const string& path,
                   vector<Fraction>& arr) {
    ifstream fin(path);
    if (!fin) return false;

    string line;
    while (getline(fin, line)) {
        if (line.empty()) continue;

        stringstream ss(line);
        string s1, s2;

        if (getline(ss, s1, ',') &&
            getline(ss, s2)) {
            Fraction f;
            f.num = atoll(s1.c_str());
            f.den = atoll(s2.c_str());
            arr.push_back(f);
        }
    }
    return true;
}

bool readConfig(const string& path,
                Fraction& lo,
                Fraction& hi) {
    ifstream fin(path);
    if (!fin) return false;

    string line;
    getline(fin, line);

    stringstream ss(line);
    string s[4];

    for (int i = 0; i < 4; ++i)
        getline(ss, s[i], i == 3 ? '\n' : ',');

    lo.num = atoll(s[0].c_str());
    lo.den = atoll(s[1].c_str());
    hi.num = atoll(s[2].c_str());
    hi.den = atoll(s[3].c_str());

    return normalize(lo) && normalize(hi);
}

int cmpFrac(Fraction a, Fraction b) {
    __int128 left = (__int128)a.num * b.den;
    __int128 right = (__int128)b.num * a.den;

    if (left < right) return -1;
    if (left > right) return 1;
    return 0;
}

bool inRange(Fraction f,
             Fraction lo,
             Fraction hi) {
    return cmpFrac(lo, f) <= 0 &&
           cmpFrac(f, hi) <= 0;
}

bool writeFractions(const string& path,
                    const vector<Fraction>& arr) {
    ofstream fout(path);
    if (!fout) return false;

    for (size_t i = 0; i < arr.size(); ++i)
        fout << arr[i].num
             << "," << arr[i].den << "\n";

    return true;
}

int main() {
    vector<Fraction> input, valid, filtered;
    Fraction lo, hi;

    if (!readFractions("fractions.csv", input)) {
        cerr << "Loi mo file fractions!\n";
        return 1;
    }

    if (!readConfig("config.txt", lo, hi)) {
        cerr << "Loi mo file config!\n";
        return 1;
    }

    for (size_t i = 0; i < input.size(); ++i) {
        Fraction f = input[i];

        if (normalize(f))
            valid.push_back(f);
    }

    int total_valid = valid.size();
    int in_range = 0, out_of_range = 0;

    if (cmpFrac(lo, hi) <= 0) {
        for (size_t i = 0; i < valid.size(); ++i) {
            if (inRange(valid[i], lo, hi)) {
                filtered.push_back(valid[i]);
                in_range++;
            } else {
                out_of_range++;
            }
        }
    } else {
        out_of_range = total_valid;
    }

    sort(filtered.begin(), filtered.end(),
        [](Fraction a, Fraction b) {
            return cmpFrac(a, b) < 0;
        });

    if (!writeFractions("filtered.csv", filtered)) {
        cerr << "Loi mo file output!\n";
        return 1;
    }

    cout << "total_valid=" << total_valid
         << " / in_range=" << in_range
         << " / out_of_range=" << out_of_range;

    return 0;
}
\end{lstlisting}
