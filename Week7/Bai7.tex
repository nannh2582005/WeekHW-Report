\section{Tổng đoạn lớn nhất (Prefix) [Trung bình]}
\subsection{Phân tích \& Thiết kế (M0)}
    \begin{itemize}
    \item \textbf{Input:}
    File \texttt{array.txt} chứa:
    \begin{itemize}
        \item Số nguyên $n$
        \item $n$ số thực
    \end{itemize}

    \item \textbf{Xử lý:}
        \begin{itemize}
            \item \textbf{Case biên:}
            \begin{itemize}
                \item $n = 0 \rightarrow$ in \texttt{EMPTY}
                \item $n = 1 \rightarrow \min = \max = \text{mean} = a[0]$
            \end{itemize}
            
            \item \textbf{Module 1 (Đọc file):}
            \begin{itemize}
                \item Hàm \texttt{readArray(path, vector<double>\&)}
                \item Kiểm tra mở file (\text{fail} $\rightarrow$ \texttt{cerr} + \texttt{return false})
            \end{itemize}
            
            \item \textbf{Module 2 (Tính mean):}
            \begin{itemize}
                \item Dùng \texttt{long double} để tránh sai số
            \end{itemize}
            
            \item \textbf{Module 3 (Tìm min, max):}
            \begin{itemize}
                \item Duyệt mảng 1 lần
            \end{itemize}
        \end{itemize}

        \item \textbf{Output:}
        \begin{lstlisting}
            min=<f> / max=<f> / mean=<f>\end{lstlisting} 
        (với \texttt{setprecision(4)})
    \end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}
    \begin{lstlisting}
    #include <iostream>
    #include <vector>
    #include <fstream>

    using namespace std;

    int main() {
        ifstream fin("input.txt");
        ofstream fout("output.txt");
        if (!fin || !fout) { cerr << "Loi file!\n"; return 1; }

        int n; fin >> n;
        if (n == 0) return 0;
        
        vector<long long> pre(n);
        long long val; fin >> val;
        pre[0] = val;
        for (int i = 1; i < n; ++i) {
            fin >> val;
            pre[i] = pre[i - 1] + val;
        }

        long long max_sum = pre[0];
        int bestL = 0, bestR = 0;
        
        long long min_pre = 0;
        int min_pre_idx = -1;

        for (int r = 0; r < n; ++r) {
            long long current_sum = pre[r] - min_pre;
            if (current_sum > max_sum) {
                max_sum = current_sum;
                bestL = min_pre_idx + 1;
                bestR = r;
            }
            if (pre[r] < min_pre) {
                min_pre = pre[r];
                min_pre_idx = r;
            }
        }

        fout << "max_sum=" << max_sum << " l=" << bestL << " r=" << bestR << "\n";
        return 0;
    }
\end{lstlisting}
