\section{Dãy Farey bậc N [Nâng cao]}
\subsection{Phân tích \& Thiết kế (M0)}
\begin{itemize}
    \item \textbf{Input:}
    File \texttt{config.txt} chứa:
    \begin{itemize}
        \item Một số nguyên \texttt{N} $(N \le 50)$
    \end{itemize}

    \item \textbf{Xử lý:}
    \begin{itemize}
        \item \textbf{Case biên:}
        \begin{itemize}
            \item \texttt{N = 1} $\rightarrow$ dãy gồm \texttt{0/1, 1/1}
            \item Nếu \texttt{N <= 0} $\rightarrow$ dãy rỗng
        \end{itemize}

        \item \textbf{Module 1 (Đọc file):}
        \begin{itemize}
            \item Hàm \texttt{readN(path, int\& N)}
            \item Kiểm tra mở file (\texttt{fail} $\rightarrow$ \texttt{return false})
        \end{itemize}

        \item \textbf{Module 2 (Ước chung lớn nhất):}
        \begin{itemize}
            \item Thuật toán Euclid:
            \begin{lstlisting}
gcd(a,b)
            \end{lstlisting}
        \end{itemize}

        \item \textbf{Module 3 (Sinh phân số):}
        \begin{itemize}
            \item Duyệt:
            \begin{lstlisting}
0 <= p <= q <= N
            \end{lstlisting}
            \item Chỉ nhận khi:
            \begin{lstlisting}
gcd(p,q) = 1
            \end{lstlisting}
            \item Khi đó phân số đã tối giản
        \end{itemize}

        \item \textbf{Module 4 (Sắp xếp):}
        \begin{itemize}
            \item So sánh:
            \begin{lstlisting}
a/b < c/d <=> a*d < c*b
            \end{lstlisting}
            \item Dùng \texttt{\_\_int128} tránh tràn số
        \end{itemize}

        \item \textbf{Module 5 (Ghi file):}
        \begin{itemize}
            \item Ghi \texttt{farey.csv}: \texttt{num,den}
        \end{itemize}

        \item \textbf{Module 6 (Thống kê):}
        \begin{itemize}
            \item In số lượng phần tử \texttt{count}
        \end{itemize}
    \end{itemize}

    \item \textbf{Output:}
    \begin{lstlisting}
Ghi file farey.csv
count=<n>
    \end{lstlisting}
\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}
\begin{lstlisting}
#include <iostream>
#include <vector>
#include <fstream>
#include <algorithm>

using namespace std;

struct Fraction {
    int num, den;
};

int gcd(int a, int b) {
    if (a < 0) a = -a;
    if (b < 0) b = -b;

    while (b != 0) {
        int t = a % b;
        a = b;
        b = t;
    }

    return a;
}

bool readN(const string& path, int& N) {
    ifstream fin(path);
    if (!fin) return false;
    fin >> N;
    return true;
}

int cmpFrac(Fraction a, Fraction b) {
    __int128 left = (__int128)a.num * b.den;
    __int128 right = (__int128)b.num * a.den;

    if (left < right) return -1;
    if (left > right) return 1;
    return 0;
}

bool writeFractions(const string& path,
                    const vector<Fraction>& arr) {
    ofstream fout(path);
    if (!fout) return false;

    for (size_t i = 0; i < arr.size(); ++i)
        fout << arr[i].num
             << "," << arr[i].den << "\n";

    return true;
}

int main() {
    int N;

    if (!readN("config.txt", N)) {
        cerr << "Loi mo file input!\n";
        return 1;
    }

    vector<Fraction> farey;

    if (N > 0) {
        for (int q = 1; q <= N; ++q) {
            for (int p = 0; p <= q; ++p) {
                if (gcd(p, q) == 1) {
                    Fraction f;
                    f.num = p;
                    f.den = q;
                    farey.push_back(f);
                }
            }
        }

        sort(farey.begin(), farey.end(),
            [](Fraction a, Fraction b) {
                return cmpFrac(a, b) < 0;
            });
    }

    if (!writeFractions("farey.csv", farey)) {
        cerr << "Loi mo file output!\n";
        return 1;
    }

    cout << "count=" << farey.size();

    return 0;
}
\end{lstlisting}
