\section{Trả lời truy vấn tổng đoạn [Cơ bản]}
\subsection{Phân tích \& Thiết kế (M0)}
    \begin{itemize}
    \item \textbf{Input:}
    File \texttt{array.txt} chứa:
    \begin{itemize}
        \item Số nguyên $n$
        \item $n$ số thực
    \end{itemize}

    \item \textbf{Xử lý:}
        \begin{itemize}
            \item \textbf{Case biên:}
            \begin{itemize}
                \item $n = 0 \rightarrow$ in \texttt{EMPTY}
                \item $n = 1 \rightarrow \min = \max = \text{mean} = a[0]$
            \end{itemize}
            
            \item \textbf{Module 1 (Đọc file):}
            \begin{itemize}
                \item Hàm \texttt{readArray(path, vector<double>\&)}
                \item Kiểm tra mở file (\text{fail} $\rightarrow$ \texttt{cerr} + \texttt{return false})
            \end{itemize}
            
            \item \textbf{Module 2 (Tính mean):}
            \begin{itemize}
                \item Dùng \texttt{long double} để tránh sai số
            \end{itemize}
            
            \item \textbf{Module 3 (Tìm min, max):}
            \begin{itemize}
                \item Duyệt mảng 1 lần
            \end{itemize}
        \end{itemize}

        \item \textbf{Output:}
        \begin{lstlisting}
            min=<f> / max=<f> / mean=<f>\end{lstlisting} 
        (với \texttt{setprecision(4)})
    \end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}
    \begin{lstlisting}
    #include <iostream>
    #include <vector>
    #include <fstream>

    using namespace std;

    bool sumRange(const vector<long long>& pre, int l, int r, long long& ans) {
        if (l < 0 || r >= pre.size() || l > r) return false;
        ans = pre[r] - (l > 0 ? pre[l - 1] : 0);
        return true;
    }

    int main() {
        ifstream fin_arr("array.txt"), fin_q("queries.txt");
        ofstream fout("output.txt");
        if (!fin_arr || !fin_q || !fout) {
            cerr << "Loi mo file!\n";
            return 1;
        }

        int n; fin_arr >> n;
        vector<long long> a(n), pre(n);
        for (int i = 0; i < n; ++i) {
            fin_arr >> a[i];
            pre[i] = a[i] + (i > 0 ? pre[i - 1] : 0);
        }

        int q; fin_q >> q;
        while (q--) {
            int l, r; fin_q >> l >> r;
            long long ans;
            if (sumRange(pre, l, r, ans)) fout << ans << "\n";
            else fout << "INVALID\n";
        }
        return 0;
    }
\end{lstlisting}
