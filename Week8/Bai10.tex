\section{Phân tích bình phương tổng [Nâng cao]}
\subsection{Phân tích \& Thiết kế (M0)}

\begin{itemize}

    \item \textbf{Input:}

    File \texttt{fractions.csv} chứa:
    \begin{itemize}
        \item Số nguyên $n$
        \item $n$ phân số dạng \texttt{num/den}
    \end{itemize}

    \item \textbf{Xử lý:}

    \begin{itemize}

        \item \textbf{Case biên:}

        \begin{itemize}
            \item $n = 0 \rightarrow$ tổng và variance bằng $0/1$
            \item Phân số có mẫu âm $\rightarrow$ chuyển mẫu dương
            \item Phân số chưa tối giản $\rightarrow$ rút gọn bằng gcd
        \end{itemize}

        \item \textbf{Module 1 (Đọc file):}

        \begin{itemize}
            \item Hàm \texttt{readFractions(path, vector<Fraction>\&)}
            \item Kiểm tra mở file
        \end{itemize}

        \item \textbf{Module 2 (Chuẩn hoá phân số):}

        \begin{itemize}
            \item Rút gọn bằng gcd
            \item Đưa mẫu về dương
        \end{itemize}

        \item \textbf{Module 3 (Các phép toán phân số):}

        \begin{itemize}
            \item Cộng phân số
            \item Nhân phân số
            \item Bình phương phân số
        \end{itemize}

        \item \textbf{Module 4 (Tính tổng bình phương):}

        \begin{itemize}
            \item 
            \[
            S=\sum_{i=1}^{n}\left(\frac{num_i}{den_i}\right)^2
            \]
        \end{itemize}

        \item \textbf{Module 5 (Tính variance):}

        \begin{itemize}
            \item 
            \[
            E[X]=\frac{1}{n}\sum x_i
            \]

            \item 
            \[
            E[X^2]=\frac{1}{n}\sum x_i^2
            \]

            \item 
            \[
            variance = E[X^2] - (E[X])^2
            \]
        \end{itemize}

        \item \textbf{Module 6 (Ghi file output):}

        \begin{itemize}
            \item Ghi file \texttt{stats.csv}
        \end{itemize}

    \end{itemize}

    \item \textbf{Output:}

\begin{lstlisting}
sum_squares=<num>/<den> / variance=<num>/<den>
\end{lstlisting}

\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}

\begin{lstlisting}
#include <iostream>
#include <vector>
#include <fstream>

using namespace std;

struct Fraction {
    long long num, den;
};

long long gcdll(long long a, long long b) {

    if (b == 0)
        return a >= 0 ? a : -a;

    return gcdll(b, a % b);
}

void normalize(Fraction& f) {

    if (f.den < 0) {
        f.num = -f.num;
        f.den = -f.den;
    }

    long long g = gcdll(f.num, f.den);

    if (g != 0) {
        f.num /= g;
        f.den /= g;
    }
}

Fraction addFraction(const Fraction& a,
                     const Fraction& b) {

    Fraction res;

    res.num =
        a.num * b.den +
        b.num * a.den;

    res.den =
        a.den * b.den;

    normalize(res);

    return res;
}

Fraction subtractFraction(const Fraction& a,
                          const Fraction& b) {

    Fraction res;

    res.num =
        a.num * b.den -
        b.num * a.den;

    res.den =
        a.den * b.den;

    normalize(res);

    return res;
}

Fraction multiplyFraction(const Fraction& a,
                          const Fraction& b) {

    Fraction res;

    res.num = a.num * b.num;
    res.den = a.den * b.den;

    normalize(res);

    return res;
}

Fraction divideFraction(const Fraction& a,
                        const Fraction& b) {

    Fraction res;

    res.num = a.num * b.den;
    res.den = a.den * b.num;

    normalize(res);

    return res;
}

Fraction squareFraction(const Fraction& a) {

    Fraction res;

    res.num = a.num * a.num;
    res.den = a.den * a.den;

    normalize(res);

    return res;
}

bool readFractions(const string& path,
                   vector<Fraction>& a) {

    ifstream fin(path);

    if (!fin) return false;

    int n;

    fin >> n;

    a.resize(n);

    for (int i = 0; i < n; ++i) {

        char slash;

        fin >> a[i].num
            >> slash
            >> a[i].den;

        normalize(a[i]);
    }

    return true;
}

bool writeStats(const string& path,
                const Fraction& sumSquares,
                const Fraction& variance) {

    ofstream fout(path);

    if (!fout) return false;

    fout << "sum_squares="
         << sumSquares.num
         << "/"
         << sumSquares.den
         << " / variance="
         << variance.num
         << "/"
         << variance.den;

    return true;
}

int main() {

    vector<Fraction> a;

    if (!readFractions("fractions.csv", a)) {

        cerr << "Loi mo file input!\n";
        return 1;
    }

    int n = a.size();

    Fraction sum = {0, 1};
    Fraction sumSquares = {0, 1};

    for (int i = 0; i < n; ++i) {

        sum = addFraction(sum, a[i]);

        Fraction sq = squareFraction(a[i]);

        sumSquares =
            addFraction(sumSquares, sq);
    }

    Fraction variance = {0, 1};

    if (n > 0) {

        Fraction nFrac = {(long long)n, 1};

        Fraction mean =
            divideFraction(sum, nFrac);

        Fraction ex2 =
            divideFraction(sumSquares, nFrac);

        Fraction mean2 =
            squareFraction(mean);

        variance =
            subtractFraction(ex2, mean2);
    }

    if (!writeStats("stats.csv",
                    sumSquares,
                    variance)) {

        cerr << "Loi mo file output!\n";
        return 1;
    }

    return 0;
}
