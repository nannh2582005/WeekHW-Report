\section{Đoạn đẹp nhất theo nhiều tiêu chí[Nâng cao]}
\subsection{Phân tích \& Thiết kế (M0)}
    \begin{itemize}
    \item \textbf{Input:}
    File \texttt{array.txt} chứa:
    \begin{itemize}
        \item Số nguyên $n$
        \item $n$ số thực
    \end{itemize}

    \item \textbf{Xử lý:}
        \begin{itemize}
            \item \textbf{Case biên:}
            \begin{itemize}
                \item $n = 0 \rightarrow$ in \texttt{EMPTY}
                \item $n = 1 \rightarrow \min = \max = \text{mean} = a[0]$
            \end{itemize}
            
            \item \textbf{Module 1 (Đọc file):}
            \begin{itemize}
                \item Hàm \texttt{readArray(path, vector<double>\&)}
                \item Kiểm tra mở file (\text{fail} $\rightarrow$ \texttt{cerr} + \texttt{return false})
            \end{itemize}
            
            \item \textbf{Module 2 (Tính mean):}
            \begin{itemize}
                \item Dùng \texttt{long double} để tránh sai số
            \end{itemize}
            
            \item \textbf{Module 3 (Tìm min, max):}
            \begin{itemize}
                \item Duyệt mảng 1 lần
            \end{itemize}
        \end{itemize}

        \item \textbf{Output:}
        \begin{lstlisting}
            min=<f> / max=<f> / mean=<f>\end{lstlisting} 
        (với \texttt{setprecision(4)})
    \end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}
    \begin{lstlisting}
    #include <iostream>
    #include <vector>
    #include <fstream>
    #include <iomanip>

    using namespace std;

    int main() {
        ifstream fin_arr("array.txt"), fin_cfg("config.txt");
        ofstream fout("output.txt");
        if (!fin_arr || !fin_cfg || !fout) { cerr << "Loi file!\n"; return 1; }

        int n; fin_arr >> n;
        vector<double> pre(n + 1, 0.0);
        for (int i = 1; i <= n; ++i) {
            double val; fin_arr >> val;
            pre[i] = pre[i - 1] + val;
        }

        int minLen; fin_cfg >> minLen;

        double max_mean = -1e18;
        int bestL = -1, bestR = -1, bestLen = -1;

        for (int l = 0; l < n; ++l) {
            for (int r = l + minLen - 1; r < n; ++r) {
                int len = r - l + 1;
                double mean = (pre[r + 1] - pre[l]) / len;
                
                bool better = false;
                // Độ chính xác float epsilon
                if (mean > max_mean + 1e-9) better = true;
                else if (abs(mean - max_mean) <= 1e-9) {
                    if (len > bestLen) better = true;
                    else if (len == bestLen && l < bestL) better = true;
                }

                if (better) {
                    max_mean = mean;
                    bestLen = len;
                    bestL = l;
                    bestR = r;
                }
            }
        }

        if (bestL != -1) {
            fout << fixed << setprecision(6);
            fout << max_mean << " " << bestL << " " << bestR << " " << bestLen << "\n";
        }
        return 0;
    }
\end{lstlisting}
