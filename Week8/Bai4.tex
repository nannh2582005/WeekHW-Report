\section{Cộng hai đa thức [Trung bình]}
\subsection{Phân tích \& Thiết kế (M0)}
\begin{itemize}
    \item \textbf{Input:}
    File \texttt{poly\_pairs.csv} chứa nhiều dòng:
    \begin{itemize}
        \item Mỗi dòng gồm 2 đa thức:
        \texttt{id1,deg1,a0..an,id2,deg2,b0..bm}
    \end{itemize}

    \item \textbf{Xử lý:}
    \begin{itemize}
        \item \textbf{Case biên:}
        \begin{itemize}
            \item Hai đa thức đều bậc 0
            \item Một đa thức rỗng (deg < 0 hoặc thiếu dữ liệu)
            \item Thiếu hệ số $\rightarrow$ gán bằng 0
        \end{itemize}

        \item \textbf{Module 1 (Struct dữ liệu):}
\begin{lstlisting}
struct Poly {
    string id;
    int deg;
    vector<double> a;
};
\end{lstlisting}

        \item \textbf{Module 2 (Đọc từng cặp đa thức):}
        \begin{itemize}
            \item Tách dữ liệu CSV
            \item Đọc đa thức thứ nhất
            \item Đọc đa thức thứ hai
        \end{itemize}

        \item \textbf{Module 3 (Cộng đa thức):}
        \begin{itemize}
            \item Hàm:
            \texttt{Poly addPoly(Poly p, Poly q)}
            \item Bậc mới:
\begin{lstlisting}
max(p.deg, q.deg)
\end{lstlisting}
            \item Cộng hệ số cùng bậc
            \item Phần dư giữ nguyên
        \end{itemize}

        \item \textbf{Module 4 (Ghi file):}
        \begin{itemize}
            \item Ghi ra \texttt{sum.csv}
            \item Định dạng:
            \texttt{id,degree\_sum,c0..cm}
        \end{itemize}
    \end{itemize}

    \item \textbf{Output:}
\begin{lstlisting}
sum.csv
id,degree_sum,c0,c1,...
\end{lstlisting}
\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}
\begin{lstlisting}
#include <iostream>
#include <fstream>
#include <sstream>
#include <vector>
#include <string>
#include <cstdlib>
#include <iomanip>
#include <algorithm>

using namespace std;

struct Poly {
    string id;
    int deg;
    vector<double> a;
};

Poly addPoly(Poly p, Poly q) {
    Poly r;
    r.id = p.id + "_" + q.id;
    r.deg = max(p.deg, q.deg);

    r.a.assign(r.deg + 1, 0);

    for (int i = 0; i <= r.deg; ++i) {
        if (i <= p.deg)
            r.a[i] += p.a[i];

        if (i <= q.deg)
            r.a[i] += q.a[i];
    }

    return r;
}

bool readOnePoly(vector<string>& token,
                 int& pos,
                 Poly& p) {
    if (pos + 1 >= (int)token.size())
        return false;

    p.id = token[pos++];
    p.deg = atoi(token[pos++].c_str());

    if (p.deg < 0) return false;

    p.a.assign(p.deg + 1, 0);

    for (int i = 0; i <= p.deg; ++i) {
        if (pos < (int)token.size())
            p.a[i] = atof(token[pos++].c_str());
        else
            p.a[i] = 0;
    }

    return true;
}

int main() {
    ifstream fin("poly_pairs.csv");

    if (!fin) {
        cerr << "Loi mo file input!\n";
        return 1;
    }

    ofstream fout("sum.csv");

    if (!fout) {
        cerr << "Loi mo file output!\n";
        return 1;
    }

    fout << fixed << setprecision(6);

    string line;

    while (getline(fin, line)) {
        if (line.empty()) continue;

        stringstream ss(line);
        vector<string> token;
        string x;

        while (getline(ss, x, ',')) {
            token.push_back(x);
        }

        int pos = 0;
        Poly p, q;

        if (!readOnePoly(token, pos, p))
            continue;

        if (!readOnePoly(token, pos, q))
            continue;

        Poly r = addPoly(p, q);

        fout << r.id << ","
             << r.deg;

        for (int i = 0; i <= r.deg; ++i)
            fout << "," << r.a[i];

        fout << "\n";
    }

    cout << "Done";

    return 0;
}
\end{lstlisting}
