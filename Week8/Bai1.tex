\section{Đọc danh sách đa thức từ CSV [Cơ bản]}
\subsection{Phân tích \& Thiết kế (M0)}
\begin{itemize}
    \item \textbf{Input:}
    File \texttt{polynomials.csv} chứa nhiều dòng:
    \begin{itemize}
        \item Mỗi dòng có dạng:
        \texttt{id,degree,a0,a1,...,an}
        \item Trong đó:
        \begin{itemize}
            \item \texttt{id}: mã đa thức
            \item \texttt{degree}: bậc đa thức
            \item \texttt{a[i]}: hệ số của \(x^i\)
        \end{itemize}
    \end{itemize}

    \item \textbf{Xử lý:}
    \begin{itemize}
        \item \textbf{Case biên:}
        \begin{itemize}
            \item \texttt{degree < 0} $\rightarrow$ invalid
            \item \texttt{degree = 0} $\rightarrow$ đa thức hằng
            \item Thiếu hệ số $\rightarrow$ gán bằng 0
            \item Nếu số hệ số ít hơn bậc yêu cầu vẫn hợp lệ, bổ sung 0
        \end{itemize}

        \item \textbf{Module 1 (Cấu trúc dữ liệu):}
\begin{lstlisting}
struct Poly {
    string id;
    int deg;
    vector<double> a;
};
\end{lstlisting}

        \item \textbf{Module 2 (Đọc file CSV):}
        \begin{itemize}
            \item Hàm:
            \texttt{readPolysCSV(path, vector<Poly>\&, int\& invalid)}
            \item Đọc từng dòng
            \item Tách dữ liệu bằng dấu phẩy
        \end{itemize}

        \item \textbf{Module 3 (Kiểm tra hợp lệ):}
        \begin{itemize}
            \item Nếu lỗi mở file $\rightarrow$ false
            \item Nếu \texttt{degree < 0} $\rightarrow$ tăng invalid
        \end{itemize}

        \item \textbf{Module 4 (Thống kê):}
        \begin{itemize}
            \item Đếm số đa thức hợp lệ
            \item Đếm số đa thức invalid
            \item Tìm bậc lớn nhất
        \end{itemize}

        \item \textbf{Module 5 (Ghi file):}
        \begin{itemize}
            \item Ghi ra \texttt{summary.txt}
        \end{itemize}
    \end{itemize}

    \item \textbf{Output:}
\begin{lstlisting}
valid=<n>
invalid=<m>
max_degree=<d>
\end{lstlisting}
\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}
\begin{lstlisting}
#include <iostream>
#include <fstream>
#include <sstream>
#include <vector>
#include <string>
#include <cstdlib>

using namespace std;

struct Poly {
    string id;
    int deg;
    vector<double> a;
};

bool readPolysCSV(const string& path,
                  vector<Poly>& arr,
                  int& invalid) {
    ifstream fin(path);
    if (!fin) return false;

    string line;
    invalid = 0;

    while (getline(fin, line)) {
        if (line.empty()) continue;

        stringstream ss(line);
        vector<string> token;
        string x;

        while (getline(ss, x, ',')) {
            token.push_back(x);
        }

        if (token.size() < 2) {
            invalid++;
            continue;
        }

        Poly p;
        p.id = token[0];
        p.deg = atoi(token[1].c_str());

        if (p.deg < 0) {
            invalid++;
            continue;
        }

        p.a.assign(p.deg + 1, 0);

        for (int i = 0; i <= p.deg; ++i) {
            int pos = i + 2;

            if (pos < (int)token.size())
                p.a[i] = atof(token[pos].c_str());
            else
                p.a[i] = 0;
        }

        arr.push_back(p);
    }

    return true;
}

int main() {
    vector<Poly> arr;
    int invalid = 0;

    if (!readPolysCSV("polynomials.csv",
                      arr, invalid)) {
        cerr << "Loi mo file input!\n";
        return 1;
    }

    int valid = arr.size();
    int max_degree = 0;

    if (valid > 0) {
        max_degree = arr[0].deg;

        for (size_t i = 1; i < arr.size(); ++i)
            if (arr[i].deg > max_degree)
                max_degree = arr[i].deg;
    }

    ofstream fout("summary.txt");

    if (!fout) {
        cerr << "Loi mo file output!\n";
        return 1;
    }

    fout << "valid=" << valid << "\n";
    fout << "invalid=" << invalid << "\n";
    fout << "max_degree=" << max_degree;

    cout << "Done";

    return 0;
}
\end{lstlisting}
