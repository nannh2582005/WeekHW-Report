\section{Prefix product modulo M [Trung bình]}
\subsection{Phân tích \& Thiết kế (M0)}
    \begin{itemize}
    \item \textbf{Input:}
    File \texttt{array.txt} chứa:
    \begin{itemize}
        \item Số nguyên $n$
        \item $n$ số thực
    \end{itemize}

    \item \textbf{Xử lý:}
        \begin{itemize}
            \item \textbf{Case biên:}
            \begin{itemize}
                \item $n = 0 \rightarrow$ in \texttt{EMPTY}
                \item $n = 1 \rightarrow \min = \max = \text{mean} = a[0]$
            \end{itemize}
            
            \item \textbf{Module 1 (Đọc file):}
            \begin{itemize}
                \item Hàm \texttt{readArray(path, vector<double>\&)}
                \item Kiểm tra mở file (\text{fail} $\rightarrow$ \texttt{cerr} + \texttt{return false})
            \end{itemize}
            
            \item \textbf{Module 2 (Tính mean):}
            \begin{itemize}
                \item Dùng \texttt{long double} để tránh sai số
            \end{itemize}
            
            \item \textbf{Module 3 (Tìm min, max):}
            \begin{itemize}
                \item Duyệt mảng 1 lần
            \end{itemize}
        \end{itemize}

        \item \textbf{Output:}
        \begin{lstlisting}
            min=<f> / max=<f> / mean=<f>\end{lstlisting} 
        (với \texttt{setprecision(4)})
    \end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}
    \begin{lstlisting}
    #include <iostream>
    #include <vector>
    #include <fstream>

    using namespace std;

    long long power(long long base, long long exp, long long mod) {
        long long res = 1;
        base %= mod;
        while (exp > 0) {
            if (exp % 2 == 1) res = (res * base) % mod;
            base = (base * base) % mod;
            exp /= 2;
        }
        return res;
    }

    long long modInverse(long long n, long long M) {
        return power(n, M - 2, M);
    }

    int main() {
        ifstream fin_arr("array.txt"), fin_q("queries.txt");
        ofstream fout("output.txt");
        if (!fin_arr || !fin_q || !fout) { cerr << "Loi file!\n"; return 1; }

        int n; long long M; 
        fin_arr >> n >> M;
        vector<long long> pre(n + 1, 1);
        vector<int> zeros(n + 1, 0);

        for (int i = 1; i <= n; ++i) {
            long long val; fin_arr >> val;
            zeros[i] = zeros[i - 1] + (val == 0 ? 1 : 0);
            val = (val == 0 ? 1 : val % M); // Giữ pre khác 0 để tránh chia 0
            pre[i] = (pre[i - 1] * val) % M;
        }

        int q; fin_q >> q;
        while (q--) {
            int l, r; fin_q >> l >> r;
            if (l < 0 || r >= n || l > r) { fout << "INVALID\n"; continue; }
            
            int zero_count = zeros[r + 1] - zeros[l];
            if (zero_count > 0) {
                fout << "INVALID\n";
            } else {
                long long ans = (pre[r + 1] * modInverse(pre[l], M)) % M;
                fout << ans << "\n";
            }
        }
        return 0;
    }
\end{lstlisting}
