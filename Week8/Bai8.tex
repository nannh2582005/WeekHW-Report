\section{Ma trận đơn vị phân số [Trung bình]}
\subsection{Phân tích \& Thiết kế (M0)}
\begin{itemize}
    \item \textbf{Input:}
    \begin{itemize}
        \item File \texttt{config.txt} chứa số nguyên $n$
        \item File \texttt{matrix.txt} chứa ma trận phân số $n \times n$
    \end{itemize}

    \item \textbf{Xử lý:}
    \begin{itemize}
        \item \textbf{Case biên:}
        \begin{itemize}
            \item $n = 0 \rightarrow$ ma trận rỗng
            \item $n = 1 \rightarrow$ kiểm tra trực tiếp phần tử duy nhất
        \end{itemize}

        \item \textbf{Module 1 (Đọc config):}
        \begin{itemize}
            \item Hàm \texttt{readConfig(path, n)}
            \item Kiểm tra mở file
        \end{itemize}

        \item \textbf{Module 2 (Đọc ma trận phân số):}
        \begin{itemize}
            \item Hàm \texttt{readMatrix(path, matrix, n)}
            \item Mỗi phần tử có dạng \texttt{a/b}
        \end{itemize}

        \item \textbf{Module 3 (Chuẩn hoá phân số):}
        \begin{itemize}
            \item Rút gọn bằng gcd
            \item Đưa mẫu về dương
        \end{itemize}

        \item \textbf{Module 4 (Tạo ma trận đơn vị):}
        \begin{itemize}
            \item Đường chéo chính = $1/1$
            \item Ngoài đường chéo = $0/1$
        \end{itemize}

        \item \textbf{Module 5 (Nhân ma trận):}
        \begin{itemize}
            \item Tính $A \times I$
            \item Tính $I \times A$
        \end{itemize}

        \item \textbf{Module 6 (So sánh ma trận):}
        \begin{itemize}
            \item So sánh từng ô sau khi chuẩn hoá phân số
        \end{itemize}
    \end{itemize}

    \item \textbf{Output:}
\begin{lstlisting}
is_identity_product=1
\end{lstlisting}
hoặc
\begin{lstlisting}
is_identity_product=0
\end{lstlisting}

\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}

\begin{lstlisting}
#include <iostream>
#include <vector>
#include <fstream>

using namespace std;

struct Fraction {
    long long num, den;
};

long long gcdll(long long a, long long b) {
    if (b == 0) return a >= 0 ? a : -a;
    return gcdll(b, a % b);
}

void normalize(Fraction& f) {
    if (f.den < 0) {
        f.num = -f.num;
        f.den = -f.den;
    }

    long long g = gcdll(f.num, f.den);
    if (g != 0) {
        f.num /= g;
        f.den /= g;
    }
}

Fraction addFraction(const Fraction& a, const Fraction& b) {
    Fraction res;
    res.num = a.num * b.den + b.num * a.den;
    res.den = a.den * b.den;
    normalize(res);
    return res;
}

Fraction multiplyFraction(const Fraction& a, const Fraction& b) {
    Fraction res;
    res.num = a.num * b.num;
    res.den = a.den * b.den;
    normalize(res);
    return res;
}

bool equalFraction(Fraction a, Fraction b) {
    normalize(a);
    normalize(b);
    return a.num == b.num && a.den == b.den;
}

bool readConfig(const string& path, int& n) {
    ifstream fin(path);
    if (!fin) return false;
    fin >> n;
    return true;
}

bool readMatrix(const string& path,
                vector<vector<Fraction>>& a,
                int n) {
    ifstream fin(path);
    if (!fin) return false;

    a.assign(n, vector<Fraction>(n));

    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            char slash;
            fin >> a[i][j].num >> slash >> a[i][j].den;
            normalize(a[i][j]);
        }
    }

    return true;
}

vector<vector<Fraction>> buildIdentity(int n) {
    vector<vector<Fraction>> I(n, vector<Fraction>(n));

    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            if (i == j)
                I[i][j] = {1, 1};
            else
                I[i][j] = {0, 1};
        }
    }

    return I;
}

vector<vector<Fraction>> multiplyMatrix(
    const vector<vector<Fraction>>& A,
    const vector<vector<Fraction>>& B) {

    int n = A.size();

    vector<vector<Fraction>> C(
        n,
        vector<Fraction>(n, {0, 1})
    );

    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {

            Fraction sum = {0, 1};

            for (int k = 0; k < n; ++k) {
                Fraction t = multiplyFraction(A[i][k], B[k][j]);
                sum = addFraction(sum, t);
            }

            C[i][j] = sum;
        }
    }

    return C;
}

bool equalMatrix(const vector<vector<Fraction>>& A,
                 const vector<vector<Fraction>>& B) {

    int n = A.size();

    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {

            if (!equalFraction(A[i][j], B[i][j]))
                return false;
        }
    }

    return true;
}

int main() {

    int n;

    if (!readConfig("config.txt", n)) {
        cerr << "Loi mo file config!\n";
        return 1;
    }

    vector<vector<Fraction>> A;

    if (!readMatrix("matrix.txt", A, n)) {
        cerr << "Loi mo file matrix!\n";
        return 1;
    }

    vector<vector<Fraction>> I = buildIdentity(n);

    vector<vector<Fraction>> AI = multiplyMatrix(A, I);
    vector<vector<Fraction>> IA = multiplyMatrix(I, A);

    bool ok =
        equalMatrix(A, AI) &&
        equalMatrix(A, IA);

    cout << "is_identity_product=" << (ok ? 1 : 0);

    return 0;
}
