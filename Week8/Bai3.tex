\section{Thống kê hệ số đa thức [Cơ bản]}
\subsection{Phân tích \& Thiết kế (M0)}
\begin{itemize}
    \item \textbf{Input:}
    File \texttt{polynomials.csv} chứa nhiều dòng:
    \begin{itemize}
        \item Mỗi dòng có dạng:
        \texttt{id,degree,a0,a1,...,an}
    \end{itemize}

    \item \textbf{Xử lý:}
    \begin{itemize}
        \item \textbf{Case biên:}
        \begin{itemize}
            \item \texttt{degree < 0} $\rightarrow$ invalid
            \item Thiếu hệ số $\rightarrow$ gán bằng 0
            \item \texttt{degree = 0} $\rightarrow$ chỉ có 1 hệ số
            \item Nếu tất cả hệ số = 0 $\rightarrow$ \texttt{min\_abs = 0}
        \end{itemize}

        \item \textbf{Module 1 (Struct dữ liệu):}
\begin{lstlisting}
struct Poly {
    string id;
    int deg;
    vector<double> a;
};
\end{lstlisting}

        \item \textbf{Module 2 (Đọc file CSV):}
        \begin{itemize}
            \item Hàm \texttt{readPolysCSV(...)}
            \item Chỉ lưu đa thức hợp lệ
        \end{itemize}

        \item \textbf{Module 3 (Thống kê hệ số):}
        \begin{itemize}
            \item Với mỗi hệ số:
\begin{lstlisting}
abs_val = fabs(a[i]);
\end{lstlisting}
            \item Cập nhật:
            \begin{itemize}
                \item \texttt{max\_abs}
                \item \texttt{min\_abs}
                \item \texttt{zero\_count}
            \end{itemize}
        \end{itemize}

        \item \textbf{Module 4 (Ghi file):}
        \begin{itemize}
            \item Ghi ra \texttt{stats.csv}
            \item Định dạng:
            \texttt{id,degree,max\_abs,min\_abs,zero\_count}
        \end{itemize}
    \end{itemize}

    \item \textbf{Output:}
\begin{lstlisting}
stats.csv
id,degree,max_abs,min_abs,zero_count
...
\end{lstlisting}
\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}
\begin{lstlisting}
#include <iostream>
#include <fstream>
#include <sstream>
#include <vector>
#include <string>
#include <cstdlib>
#include <iomanip>
#include <cmath>

using namespace std;

struct Poly {
    string id;
    int deg;
    vector<double> a;
};

bool readPolysCSV(const string& path,
                  vector<Poly>& arr) {
    ifstream fin(path);
    if (!fin) return false;

    string line;

    while (getline(fin, line)) {
        if (line.empty()) continue;

        stringstream ss(line);
        vector<string> token;
        string x;

        while (getline(ss, x, ',')) {
            token.push_back(x);
        }

        if (token.size() < 2) continue;

        Poly p;
        p.id = token[0];
        p.deg = atoi(token[1].c_str());

        if (p.deg < 0) continue;

        p.a.assign(p.deg + 1, 0);

        for (int i = 0; i <= p.deg; ++i) {
            int pos = i + 2;

            if (pos < (int)token.size())
                p.a[i] = atof(token[pos].c_str());
            else
                p.a[i] = 0;
        }

        arr.push_back(p);
    }

    return true;
}

int main() {
    vector<Poly> arr;

    if (!readPolysCSV("polynomials.csv", arr)) {
        cerr << "Loi mo file input!\n";
        return 1;
    }

    ofstream fout("stats.csv");

    if (!fout) {
        cerr << "Loi mo file output!\n";
        return 1;
    }

    fout << "id,degree,max_abs,min_abs,zero_count\n";
    fout << fixed << setprecision(6);

    for (size_t i = 0; i < arr.size(); ++i) {
        double max_abs = fabs(arr[i].a[0]);
        double min_abs = fabs(arr[i].a[0]);
        int zero_count = 0;

        for (int j = 0; j <= arr[i].deg; ++j) {
            double val = fabs(arr[i].a[j]);

            if (val > max_abs)
                max_abs = val;

            if (val < min_abs)
                min_abs = val;

            if (arr[i].a[j] == 0)
                zero_count++;
        }

        fout << arr[i].id << ","
             << arr[i].deg << ","
             << max_abs << ","
             << min_abs << ","
             << zero_count << "\n";
    }

    cout << "Done";

    return 0;
}
\end{lstlisting}
