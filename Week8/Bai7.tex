\section{Tìm nghiệm nguyên của đa thức [Trung bình]}
\subsection{Phân tích \& Thiết kế (M0)}
\begin{itemize}
    \item \textbf{Input:}
    File \texttt{polynomials.csv} chứa nhiều dòng:
    \begin{itemize}
        \item Mỗi dòng có dạng:
        \texttt{id,degree,a0,a1,...,an}
    \end{itemize}

    \item \textbf{Xử lý:}
    \begin{itemize}
        \item \textbf{Case biên:}
        \begin{itemize}
            \item \texttt{degree < 0} $\rightarrow$ invalid
            \item Thiếu hệ số $\rightarrow$ gán bằng 0
            \item Chỉ xét đa thức có hệ số nguyên
            \item Tìm nghiệm nguyên trong đoạn \texttt{[-50,50]}
            \item Không có nghiệm $\rightarrow$ ghi \texttt{id,NONE}
            \item Đa thức hằng khác 0 $\rightarrow$ không có nghiệm
            \item Đa thức 0 tuyệt đối $\rightarrow$ mọi giá trị đều là nghiệm, vẫn duyệt \texttt{[-50,50]}
        \end{itemize}

        \item \textbf{Module 1 (Struct dữ liệu):}
\begin{lstlisting}
struct Poly {
    string id;
    int deg;
    vector<long long> a;
};
\end{lstlisting}

        \item \textbf{Module 2 (Đọc file CSV):}
        \begin{itemize}
            \item Chỉ lưu đa thức hợp lệ
            \item Chuyển hệ số về kiểu nguyên
        \end{itemize}

        \item \textbf{Module 3 (Tính P(x)):}
        \begin{itemize}
            \item Dùng Horner
        \end{itemize}

\begin{lstlisting}
res = a[n];
for i = n-1 downto 0:
    res = res*x + a[i];
\end{lstlisting}

        \item \textbf{Module 4 (Tìm nghiệm):}
        \begin{itemize}
            \item Duyệt mọi \texttt{x} từ -50 đến 50
            \item Nếu \texttt{P(x)=0} thì ghi nghiệm
        \end{itemize}

        \item \textbf{Module 5 (Ghi file):}
        \begin{itemize}
            \item File \texttt{roots.csv}
            \item Mỗi nghiệm một dòng:
            \texttt{id,root}
        \end{itemize}
    \end{itemize}

    \item \textbf{Output:}
\begin{lstlisting}
roots.csv
id,root
...
\end{lstlisting}
\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}
\begin{lstlisting}
#include <iostream>
#include <fstream>
#include <sstream>
#include <vector>
#include <string>
#include <cstdlib>
#include <cmath>

using namespace std;

struct Poly {
    string id;
    int deg;
    vector<long long> a;
};

bool isIntegerValue(double x) {
    return fabs(x - round(x)) < 1e-9;
}

bool readPolysCSV(const string& path,
                  vector<Poly>& arr) {
    ifstream fin(path);
    if (!fin) return false;

    string line;

    while (getline(fin, line)) {
        if (line.empty()) continue;

        stringstream ss(line);
        vector<string> token;
        string s;

        while (getline(ss, s, ',')) {
            token.push_back(s);
        }

        if (token.size() < 2) continue;

        string id = token[0];
        int deg = atoi(token[1].c_str());

        if (deg < 0) continue;

        vector<long long> coef(deg + 1, 0);
        bool ok = true;

        for (int i = 0; i <= deg; ++i) {
            int pos = i + 2;

            double val = 0;

            if (pos < (int)token.size())
                val = atof(token[pos].c_str());

            if (!isIntegerValue(val)) {
                ok = false;
                break;
            }

            coef[i] = (long long)llround(val);
        }

        if (!ok) continue;

        Poly p;
        p.id = id;
        p.deg = deg;
        p.a = coef;

        arr.push_back(p);
    }

    return true;
}

long long evalHorner(const vector<long long>& a,
                     long long x) {
    int n = a.size() - 1;

    long long res = a[n];

    for (int i = n - 1; i >= 0; --i)
        res = res * x + a[i];

    return res;
}

int main() {
    vector<Poly> arr;

    if (!readPolysCSV("polynomials.csv", arr)) {
        cerr << "Loi mo file input!\n";
        return 1;
    }

    ofstream fout("roots.csv");

    if (!fout) {
        cerr << "Loi mo file output!\n";
        return 1;
    }

    fout << "id,root\n";

    for (size_t i = 0; i < arr.size(); ++i) {
        bool found = false;

        for (int x = -50; x <= 50; ++x) {
            if (evalHorner(arr[i].a, x) == 0) {
                fout << arr[i].id
                     << "," << x << "\n";
                found = true;
            }
        }

        if (!found) {
            fout << arr[i].id
                 << ",NONE\n";
        }
    }

    cout << "Done";

    return 0;
}
\end{lstlisting}
