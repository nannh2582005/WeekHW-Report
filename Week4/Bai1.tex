\section{Max subarray sum (Kadane cơ bản) [Cơ bản]}
\subsection{Phân tích \& Thiết kế (M0)}
    \begin{itemize}
    \item \textbf{Input:}
    File \texttt{array.txt} chứa:
    \begin{itemize}
        \item Số nguyên $n$
        \item $n$ số thực
    \end{itemize}

    \item \textbf{Xử lý:}
        \begin{itemize}
            \item \textbf{Case biên:}
            \begin{itemize}
                \item $n = 0 \rightarrow$ in \texttt{EMPTY}
                \item $n = 1 \rightarrow \min = \max = \text{mean} = a[0]$
            \end{itemize}
            
            \item \textbf{Module 1 (Đọc file):}
            \begin{itemize}
                \item Hàm \texttt{readArray(path, vector<double>\&)}
                \item Kiểm tra mở file (\text{fail} $\rightarrow$ \texttt{cerr} + \texttt{return false})
            \end{itemize}
            
            \item \textbf{Module 2 (Tính mean):}
            \begin{itemize}
                \item Dùng \texttt{long double} để tránh sai số
            \end{itemize}
            
            \item \textbf{Module 3 (Tìm min, max):}
            \begin{itemize}
                \item Duyệt mảng 1 lần
            \end{itemize}
        \end{itemize}

        \item \textbf{Output:}
        \begin{lstlisting}
            min=<f> / max=<f> / mean=<f>\end{lstlisting} 
        (với \texttt{setprecision(4)})
    \end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}
    \begin{lstlisting}
    #include <iostream>
    #include <vector>
    #include <fstream>
    #include <algorithm>

    using namespace std;

    bool readArray(const string& path, vector<long long>& a) {
        ifstream fin(path);
        if (!fin) return false;
        int n;
        if (fin >> n) {
            a.resize(n);
            for (int i = 0; i < n; ++i) fin >> a[i];
        }
        return true;
    }

    long long kadane(const vector<long long>& a) {
        long long bestEnding = a[0];
        long long bestSoFar = a[0];
        for (size_t i = 1; i < a.size(); ++i) {
            bestEnding = max(a[i], bestEnding + a[i]);
            bestSoFar = max(bestSoFar, bestEnding);
        }
        return bestSoFar;
    }

    int main() {
        vector<long long> a;
        if (!readArray("array.txt", a)) {
            cerr << "Loi mo file input!\n";
            return 1;
        }

        ofstream fout("output.txt");
        if (!fout) { cerr << "Loi mo file output!\n"; return 1; }

        if (a.empty()) {
            fout << "EMPTY\n";
        } else {
            fout << "max_sum=" << kadane(a) << "\n";
        }
        return 0;
    }
\end{lstlisting}
