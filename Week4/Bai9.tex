\section{Maximum circular subarray [Nâng cao]}
\subsection{Phân tích \& Thiết kế (M0)}
    \begin{itemize}
    \item \textbf{Input:}
    File \texttt{array.txt} chứa:
    \begin{itemize}
        \item Số nguyên $n$
        \item $n$ số thực
    \end{itemize}

    \item \textbf{Xử lý:}
        \begin{itemize}
            \item \textbf{Case biên:}
            \begin{itemize}
                \item $n = 0 \rightarrow$ in \texttt{EMPTY}
                \item $n = 1 \rightarrow \min = \max = \text{mean} = a[0]$
            \end{itemize}
            
            \item \textbf{Module 1 (Đọc file):}
            \begin{itemize}
                \item Hàm \texttt{readArray(path, vector<double>\&)}
                \item Kiểm tra mở file (\text{fail} $\rightarrow$ \texttt{cerr} + \texttt{return false})
            \end{itemize}
            
            \item \textbf{Module 2 (Tính mean):}
            \begin{itemize}
                \item Dùng \texttt{long double} để tránh sai số
            \end{itemize}
            
            \item \textbf{Module 3 (Tìm min, max):}
            \begin{itemize}
                \item Duyệt mảng 1 lần
            \end{itemize}
        \end{itemize}

        \item \textbf{Output:}
        \begin{lstlisting}
            min=<f> / max=<f> / mean=<f>\end{lstlisting} 
        (với \texttt{setprecision(4)})
    \end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}
    \begin{lstlisting}
    #include <iostream>
    #include <vector>
    #include <fstream>
    #include <algorithm>

    using namespace std;

    long long kadaneMax(const vector<long long>& a) {
        long long bestEnding = a[0], bestSoFar = a[0];
        for (size_t i = 1; i < a.size(); ++i) {
            bestEnding = max(a[i], bestEnding + a[i]);
            bestSoFar = max(bestSoFar, bestEnding);
        }
        return bestSoFar;
    }

    long long kadaneMin(const vector<long long>& a) {
        long long bestEnding = a[0], bestSoFar = a[0];
        for (size_t i = 1; i < a.size(); ++i) {
            bestEnding = min(a[i], bestEnding + a[i]);
            bestSoFar = min(bestSoFar, bestEnding);
        }
        return bestSoFar;
    }

    int main() {
        ifstream fin("array.txt");
        ofstream fout("output.txt");
        if (!fin || !fout) { cerr << "Loi file!\n"; return 1; }

        int n; fin >> n;
        if (n == 0) { fout << "EMPTY\n"; return 0; }

        vector<long long> a(n);
        long long total_sum = 0;
        for (int i = 0; i < n; ++i) {
            fin >> a[i];
            total_sum += a[i];
        }

        long long max_k = kadaneMax(a);
        long long min_k = kadaneMin(a);

        long long result;
        if (max_k < 0) {
            result = max_k;
        } else {
            result = max(max_k, total_sum - min_k);
        }

        fout << "max_circular=" << result << "\n";
        return 0;
    }
\end{lstlisting}
