\section{Kadane có truy vết l, r [Cơ bản]}
\subsection{Phân tích \& Thiết kế (M0)}
    \begin{itemize}
    \item \textbf{Input:}
    File \texttt{array.txt} chứa:
    \begin{itemize}
        \item Số nguyên $n$
        \item $n$ số thực
    \end{itemize}

    \item \textbf{Xử lý:}
        \begin{itemize}
            \item \textbf{Case biên:}
            \begin{itemize}
                \item $n = 0 \rightarrow$ in \texttt{EMPTY}
                \item $n = 1 \rightarrow \min = \max = \text{mean} = a[0]$
            \end{itemize}
            
            \item \textbf{Module 1 (Đọc file):}
            \begin{itemize}
                \item Hàm \texttt{readArray(path, vector<double>\&)}
                \item Kiểm tra mở file (\text{fail} $\rightarrow$ \texttt{cerr} + \texttt{return false})
            \end{itemize}
            
            \item \textbf{Module 2 (Tính mean):}
            \begin{itemize}
                \item Dùng \texttt{long double} để tránh sai số
            \end{itemize}
            
            \item \textbf{Module 3 (Tìm min, max):}
            \begin{itemize}
                \item Duyệt mảng 1 lần
            \end{itemize}
        \end{itemize}

        \item \textbf{Output:}
        \begin{lstlisting}
            min=<f> / max=<f> / mean=<f>\end{lstlisting} 
        (với \texttt{setprecision(4)})
    \end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}
    \begin{lstlisting}
    #include <iostream>
    #include <vector>
    #include <fstream>

    using namespace std;

    struct SubarrayResult {
        long long sum;
        int l, r;
    };

    SubarrayResult kadaneWithTrace(const vector<long long>& a) {
        long long bestEnding = a[0];
        long long bestSoFar = a[0];
        int start = 0, bestL = 0, bestR = 0;

        for (int i = 1; i < (int)a.size(); ++i) {
            if (a[i] > bestEnding + a[i]) {
                bestEnding = a[i];
                start = i;
            } else {
                bestEnding += a[i];
            }

            if (bestEnding > bestSoFar) {
                bestSoFar = bestEnding;
                bestL = start;
                bestR = i;
            }
        }
        return {bestSoFar, bestL, bestR};
    }

    int main() {
        ifstream fin("array.txt");
        ofstream fout("output.txt");
        if (!fin || !fout) { cerr << "Loi mo file!\n"; return 1; }

        int n; fin >> n;
        if (n == 0) { fout << "EMPTY\n"; return 0; }
        
        vector<long long> a(n);
        for (int i = 0; i < n; ++i) fin >> a[i];

        SubarrayResult res = kadaneWithTrace(a);
        fout << "max_sum=" << res.sum << " / l=" << res.l << " / r=" << res.r << "\n";
        return 0;
    }
\end{lstlisting}
