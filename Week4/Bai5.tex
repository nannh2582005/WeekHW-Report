\section{LIS truy vết dãy con [Trung bình]}
\subsection{Phân tích \& Thiết kế (M0)}
    \begin{itemize}
    \item \textbf{Input:}
    File \texttt{array.txt} chứa:
    \begin{itemize}
        \item Số nguyên $n$
        \item $n$ số thực
    \end{itemize}

    \item \textbf{Xử lý:}
        \begin{itemize}
            \item \textbf{Case biên:}
            \begin{itemize}
                \item $n = 0 \rightarrow$ in \texttt{EMPTY}
                \item $n = 1 \rightarrow \min = \max = \text{mean} = a[0]$
            \end{itemize}
            
            \item \textbf{Module 1 (Đọc file):}
            \begin{itemize}
                \item Hàm \texttt{readArray(path, vector<double>\&)}
                \item Kiểm tra mở file (\text{fail} $\rightarrow$ \texttt{cerr} + \texttt{return false})
            \end{itemize}
            
            \item \textbf{Module 2 (Tính mean):}
            \begin{itemize}
                \item Dùng \texttt{long double} để tránh sai số
            \end{itemize}
            
            \item \textbf{Module 3 (Tìm min, max):}
            \begin{itemize}
                \item Duyệt mảng 1 lần
            \end{itemize}
        \end{itemize}

        \item \textbf{Output:}
        \begin{lstlisting}
            min=<f> / max=<f> / mean=<f>\end{lstlisting} 
        (với \texttt{setprecision(4)})
    \end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}
    \begin{lstlisting}
    #include <iostream>
#include <vector>
#include <fstream>
#include <algorithm>

using namespace std;

void traceLIS(const vector<int>& a, vector<int>& indices) {
    if (a.empty()) return;
    int n = a.size();
    vector<int> dp(n, 1), parent(n, -1);
    
    int max_len = 1, end_idx = 0;
    
    for (int i = 1; i < n; ++i) {
        for (int j = 0; j < i; ++j) {
            if (a[j] < a[i] && dp[j] + 1 > dp[i]) {
                dp[i] = dp[j] + 1;
                parent[i] = j;
            }
        }
        if (dp[i] > max_len) {
            max_len = dp[i];
            end_idx = i;
        }
    }

    int curr = end_idx;
    while (curr != -1) {
        indices.push_back(curr);
        curr = parent[curr];
    }
    reverse(indices.begin(), indices.end());
}

int main() {
    ifstream fin("array.txt");
    ofstream fout("output.txt");
    if (!fin || !fout) { cerr << "Loi file!\n"; return 1; }

    int n; fin >> n;
    if (n == 0) { fout << "lis_len=0 / indices=\n"; return 0; }

    vector<int> a(n);
    for (int i = 0; i < n; ++i) fin >> a[i];

    vector<int> indices;
    traceLIS(a, indices);
    
    fout << "lis_len=" << indices.size() << " / indices=";
    for (size_t i = 0; i < indices.size(); ++i) {
        fout << indices[i] << (i + 1 == indices.size() ? "" : " ");
    }
    fout << "\n";
    return 0;
}
\end{lstlisting}
