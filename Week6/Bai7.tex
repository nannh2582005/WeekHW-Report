\section{Vùng con tổng lớn nhất (brute force) [Trung bình]}
\subsection{Phân tích \& Thiết kế (M0)}
\begin{itemize}
    \item \textbf{Input:}
    File \texttt{input.txt} chứa:
    \begin{itemize}
        \item Hai số nguyên $m, n$
        \item Ma trận $m \times n$ $(m,n \le 50)$
    \end{itemize}

    \item \textbf{Xử lý:}
    \begin{itemize}
        \item \textbf{Case biên:}
        \begin{itemize}
            \item $m = 0$ hoặc $n = 0 \rightarrow$ không xử lý
            \item Ma trận 1 ô $\rightarrow$ đáp án là chính ô đó
            \item Ma trận toàn số âm $\rightarrow$ chọn vùng có giá trị lớn nhất
        \end{itemize}

        \item \textbf{Module 1 (Đọc file):}
        \begin{itemize}
            \item Hàm \texttt{readMatrix(path, vector<vector<long long>>\&)}
            \item Kiểm tra mở file (\texttt{fail} $\rightarrow$ \texttt{return false})
        \end{itemize}

        \item \textbf{Module 2 (Xây prefix 2D):}
        \begin{itemize}
            \item 
            \begin{lstlisting}
pre[i][j] = a[i][j] + pre[i-1][j]
          + pre[i][j-1] - pre[i-1][j-1]
            \end{lstlisting}
        \end{itemize}

        \item \textbf{Module 3 (Duyệt mọi hình chữ nhật):}
        \begin{itemize}
            \item Duyệt mọi bộ $(x1,y1,x2,y2)$
            \item Tính tổng vùng trong $O(1)$ bằng prefix 2D
            \item Cập nhật kết quả lớn nhất
            \item Độ phức tạp: $O(m^2n^2)$
        \end{itemize}
    \end{itemize}

    \item \textbf{Output:}
    \begin{lstlisting}
max_sum x1 y1 x2 y2
    \end{lstlisting}
\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}
\begin{lstlisting}
#include <iostream>
#include <vector>
#include <fstream>
#include <climits>

using namespace std;

bool readMatrix(const string& path, vector<vector<long long>>& a) {
    ifstream fin(path);
    if (!fin) return false;

    int m, n;
    fin >> m >> n;
    a.assign(m, vector<long long>(n));

    for (int i = 0; i < m; ++i)
        for (int j = 0; j < n; ++j)
            fin >> a[i][j];

    return true;
}

vector<vector<long long>> buildPrefix2D(
        const vector<vector<long long>>& a) {
    int m = a.size();
    if (m == 0) return {};

    int n = a[0].size();
    vector<vector<long long>> pre(m, vector<long long>(n, 0));

    for (int i = 0; i < m; ++i)
        for (int j = 0; j < n; ++j) {
            pre[i][j] = a[i][j];
            if (i > 0) pre[i][j] += pre[i - 1][j];
            if (j > 0) pre[i][j] += pre[i][j - 1];
            if (i > 0 && j > 0) pre[i][j] -= pre[i - 1][j - 1];
        }

    return pre;
}

long long sumRect(const vector<vector<long long>>& pre,
                  int x1, int y1, int x2, int y2) {
    long long ans = pre[x2][y2];

    if (x1 > 0) ans -= pre[x1 - 1][y2];
    if (y1 > 0) ans -= pre[x2][y1 - 1];
    if (x1 > 0 && y1 > 0) ans += pre[x1 - 1][y1 - 1];

    return ans;
}

int main() {
    vector<vector<long long>> a;

    if (!readMatrix("input.txt", a)) {
        cerr << "Loi mo file input!\n";
        return 1;
    }

    if (a.empty() || a[0].empty()) return 0;

    int m = a.size();
    int n = a[0].size();

    vector<vector<long long>> pre = buildPrefix2D(a);

    long long best = LLONG_MIN;
    int bx1 = 0, by1 = 0, bx2 = 0, by2 = 0;

    for (int x1 = 0; x1 < m; ++x1)
        for (int y1 = 0; y1 < n; ++y1)
            for (int x2 = x1; x2 < m; ++x2)
                for (int y2 = y1; y2 < n; ++y2) {
                    long long cur =
                        sumRect(pre, x1, y1, x2, y2);

                    if (cur > best) {
                        best = cur;
                        bx1 = x1;
                        by1 = y1;
                        bx2 = x2;
                        by2 = y2;
                    }
                }

    cout << best << " "
         << bx1 << " " << by1 << " "
         << bx2 << " " << by2;

    return 0;
}
\end{lstlisting}
