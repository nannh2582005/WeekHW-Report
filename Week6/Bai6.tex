\section{Nhân hai ma trận [Trung bình]}
\subsection{Phân tích \& Thiết kế (M0)}
\begin{itemize}
    \item \textbf{Input:}
    File \texttt{input.txt} chứa:
    \begin{itemize}
        \item Hai số nguyên $m, k$ và ma trận $A(m \times k)$
        \item Hai số nguyên $r, n$ và ma trận $B(r \times n)$
    \end{itemize}

    \item \textbf{Xử lý:}
    \begin{itemize}
        \item \textbf{Case biên:}
        \begin{itemize}
            \item Nếu $k \ne r \rightarrow$ in \texttt{INCOMPATIBLE}
            \item Nếu ma trận có kích thước $1 \times 1$ vẫn xử lý bình thường
        \end{itemize}

        \item \textbf{Module 1 (Đọc file):}
        \begin{itemize}
            \item Hàm \texttt{readMatrices(path, A, B)}
            \item Kiểm tra mở file (\texttt{fail} $\rightarrow$ \texttt{return false})
        \end{itemize}

        \item \textbf{Module 2 (Kiểm tra kích thước):}
        \begin{itemize}
            \item Điều kiện nhân được: số cột của $A$ bằng số hàng của $B$
        \end{itemize}

        \item \textbf{Module 3 (Nhân ma trận):}
        \begin{itemize}
            \item Công thức:
            \begin{lstlisting}
C[i][j] += A[i][t] * B[t][j]
            \end{lstlisting}
            \item Duyệt 3 vòng lặp
        \end{itemize}
    \end{itemize}

    \item \textbf{Output:}
    \begin{lstlisting}
Mỗi hàng là 1 dòng, các phần tử cách nhau bởi space
    \end{lstlisting}
\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}
\begin{lstlisting}
#include <iostream>
#include <vector>
#include <fstream>

using namespace std;

bool readMatrices(const string& path,
                  vector<vector<long long>>& A,
                  vector<vector<long long>>& B) {
    ifstream fin(path);
    if (!fin) return false;

    int m, k;
    fin >> m >> k;
    A.assign(m, vector<long long>(k));

    for (int i = 0; i < m; ++i)
        for (int j = 0; j < k; ++j)
            fin >> A[i][j];

    int r, n;
    fin >> r >> n;
    B.assign(r, vector<long long>(n));

    for (int i = 0; i < r; ++i)
        for (int j = 0; j < n; ++j)
            fin >> B[i][j];

    return true;
}

vector<vector<long long>> multiplyMatrix(
        const vector<vector<long long>>& A,
        const vector<vector<long long>>& B) {
    int m = A.size();
    int k = A[0].size();
    int n = B[0].size();

    vector<vector<long long>> C(m, vector<long long>(n, 0));

    for (int i = 0; i < m; ++i)
        for (int t = 0; t < k; ++t)
            for (int j = 0; j < n; ++j)
                C[i][j] += A[i][t] * B[t][j];

    return C;
}

void printMatrix(const vector<vector<long long>>& C) {
    for (size_t i = 0; i < C.size(); ++i) {
        for (size_t j = 0; j < C[i].size(); ++j) {
            if (j > 0) cout << " ";
            cout << C[i][j];
        }
        cout << "\n";
    }
}

int main() {
    vector<vector<long long>> A, B;

    if (!readMatrices("input.txt", A, B)) {
        cerr << "Loi mo file input!\n";
        return 1;
    }

    if (A.empty() || B.empty()) return 0;

    if (A[0].size() != B.size()) {
        cout << "INCOMPATIBLE";
        return 0;
    }

    vector<vector<long long>> C = multiplyMatrix(A, B);
    printMatrix(C);

    return 0;
}
\end{lstlisting}
