\section{Đếm vùng con tổng bằng K [Nâng cao]}
\subsection{Phân tích \& Thiết kế (M0)}
\begin{itemize}
    \item \textbf{Input:}
    File \texttt{input.txt} chứa:
    \begin{itemize}
        \item Hai số nguyên $m, n$ $(m,n \le 100)$
        \item Ma trận $m \times n$
        \item Giá trị $K$
    \end{itemize}

    \item \textbf{Xử lý:}
    \begin{itemize}
        \item \textbf{Case biên:}
        \begin{itemize}
            \item $m = 0$ hoặc $n = 0 \rightarrow$ không xử lý
            \item Ma trận 1 ô: kết quả là 1 nếu phần tử bằng $K$, ngược lại là 0
        \end{itemize}

        \item \textbf{Module 1 (Đọc file):}
        \begin{itemize}
            \item Hàm \texttt{readMatrix(path, vector<vector<long long>>\&, long long\& K)}
            \item Kiểm tra mở file (\texttt{fail} $\rightarrow$ \texttt{return false})
        \end{itemize}

        \item \textbf{Module 2 (Duyệt cặp hàng):}
        \begin{itemize}
            \item Chọn hai hàng \texttt{top} và \texttt{bottom}
            \item Gom tổng từng cột giữa hai hàng thành mảng 1 chiều
        \end{itemize}

        \item \textbf{Module 3 (Đếm subarray sum = K):}
        \begin{itemize}
            \item Dùng prefix sum 1D + \texttt{map}
            \item Nếu prefix hiện tại là \texttt{sum}, cần số lần xuất hiện của \texttt{sum - K}
            \item Cộng vào đáp án
        \end{itemize}

        \item \textbf{Độ phức tạp:}
        \begin{itemize}
            \item $O(m^2 \cdot n)$
        \end{itemize}
    \end{itemize}

    \item \textbf{Output:}
    \begin{lstlisting}
count=<n>
    \end{lstlisting}
\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}
\begin{lstlisting}
#include <iostream>
#include <vector>
#include <fstream>
#include <map>

using namespace std;

bool readMatrix(const string& path,
                vector<vector<long long>>& a,
                long long& K) {
    ifstream fin(path);
    if (!fin) return false;

    int m, n;
    fin >> m >> n;
    a.assign(m, vector<long long>(n));

    for (int i = 0; i < m; ++i)
        for (int j = 0; j < n; ++j)
            fin >> a[i][j];

    fin >> K;
    return true;
}

long long countSubarraySumK(
        const vector<long long>& arr,
        long long K) {
    map<long long, long long> freq;
    freq[0] = 1;

    long long sum = 0;
    long long cnt = 0;

    for (size_t i = 0; i < arr.size(); ++i) {
        sum += arr[i];

        if (freq.count(sum - K))
            cnt += freq[sum - K];

        freq[sum]++;
    }

    return cnt;
}

long long countRectangles(
        const vector<vector<long long>>& a,
        long long K) {
    int m = a.size();
    int n = a[0].size();

    long long ans = 0;

    for (int top = 0; top < m; ++top) {
        vector<long long> col(n, 0);

        for (int bottom = top; bottom < m; ++bottom) {
            for (int j = 0; j < n; ++j)
                col[j] += a[bottom][j];

            ans += countSubarraySumK(col, K);
        }
    }

    return ans;
}

int main() {
    vector<vector<long long>> a;
    long long K;

    if (!readMatrix("input.txt", a, K)) {
        cerr << "Loi mo file input!\n";
        return 1;
    }

    if (a.empty() || a[0].empty()) return 0;

    long long result = countRectangles(a, K);

    cout << "count=" << result;

    return 0;
}
\end{lstlisting}

\subsection{Kế hoạch kiểm thử (Test Cases)}

\textbf{Test Case 1 (Thông thường)}

\textbf{Input (input.txt):}
\begin{lstlisting}
2 2
1 2
3 4
3
\end{lstlisting}

\textbf{Expected Output:}
\begin{lstlisting}
count=2
\end{lstlisting}

\vspace{0.5cm}

\textbf{Giải thích:}
Hai vùng có tổng bằng 3:
\begin{itemize}
    \item Ô $(1,0)$ có giá trị 3
    \item Hàng đầu: $1 + 2 = 3$
\end{itemize}

\vspace{0.5cm}
