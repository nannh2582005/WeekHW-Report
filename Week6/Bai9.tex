\section{Xoay ma trận 90 độ theo chiều kim đồng hồ [Trung bình]}
\subsection{Phân tích \& Thiết kế (M0)}
\begin{itemize}
    \item \textbf{Input:}
    File \texttt{input.txt} chứa:
    \begin{itemize}
        \item Hai số nguyên $m, n$
        \item Ma trận $m \times n$
    \end{itemize}

    \item \textbf{Xử lý:}
    \begin{itemize}
        \item \textbf{Case biên:}
        \begin{itemize}
            \item $m = 0$ hoặc $n = 0 \rightarrow$ không xử lý
            \item Ma trận $1 \times 1 \rightarrow$ giữ nguyên
            \item Ma trận kết quả có kích thước $n \times m$
        \end{itemize}

        \item \textbf{Module 1 (Đọc file):}
        \begin{itemize}
            \item Hàm \texttt{readMatrix(path, vector<vector<long long>>\&)}
            \item Kiểm tra mở file (\texttt{fail} $\rightarrow$ \texttt{return false})
        \end{itemize}

        \item \textbf{Module 2 (Xoay ma trận):}
        \begin{itemize}
            \item Với mỗi phần tử $(i,j)$ của ma trận cũ:
            \begin{lstlisting}
new[j][m - 1 - i] = a[i][j]
            \end{lstlisting}
        \end{itemize}

        \item \textbf{Module 3 (Ghi file):}
        \begin{itemize}
            \item Hàm \texttt{writeMatrix(path, rotated)}
            \item Mỗi hàng là 1 dòng, các phần tử cách nhau bởi space
        \end{itemize}
    \end{itemize}

    \item \textbf{Output:}
    \begin{lstlisting}
Ghi ma trận kết quả ra file rotated.txt
    \end{lstlisting}
\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}
\begin{lstlisting}
#include <iostream>
#include <vector>
#include <fstream>

using namespace std;

bool readMatrix(const string& path, vector<vector<long long>>& a) {
    ifstream fin(path);
    if (!fin) return false;

    int m, n;
    fin >> m >> n;
    a.assign(m, vector<long long>(n));

    for (int i = 0; i < m; ++i)
        for (int j = 0; j < n; ++j)
            fin >> a[i][j];

    return true;
}

vector<vector<long long>> rotateClockwise(
        const vector<vector<long long>>& a) {
    int m = a.size();
    if (m == 0) return {};

    int n = a[0].size();
    vector<vector<long long>> b(n, vector<long long>(m));

    for (int i = 0; i < m; ++i)
        for (int j = 0; j < n; ++j)
            b[j][m - 1 - i] = a[i][j];

    return b;
}

bool writeMatrix(const string& path,
                 const vector<vector<long long>>& a) {
    ofstream fout(path);
    if (!fout) return false;

    for (size_t i = 0; i < a.size(); ++i) {
        for (size_t j = 0; j < a[i].size(); ++j) {
            if (j > 0) fout << " ";
            fout << a[i][j];
        }
        fout << "\n";
    }

    return true;
}

int main() {
    vector<vector<long long>> a;

    if (!readMatrix("input.txt", a)) {
        cerr << "Loi mo file input!\n";
        return 1;
    }

    if (a.empty() || a[0].empty()) return 0;

    vector<vector<long long>> rotated =
        rotateClockwise(a);

    if (!writeMatrix("rotated.txt", rotated)) {
        cerr << "Loi mo file output!\n";
        return 1;
    }

    return 0;
}
\end{lstlisting}
