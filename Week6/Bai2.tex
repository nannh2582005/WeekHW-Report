\section{Kiểm tra ma trận đối xứng [Cơ bản]}
\subsection{Phân tích \& Thiết kế (M0)}

\begin{itemize}
    \item \textbf{Input:}  
    File \texttt{matrix.txt} chứa:
    \begin{itemize}
        \item Số nguyên $n$
        \item Ma trận $n \times n$ số nguyên
    \end{itemize}

    \item \textbf{Xử lý:}
    \begin{itemize}
        \item \textbf{Case biên:}
        \begin{itemize}
            \item $n=1$ $\rightarrow$ luôn đối xứng
            \item $n=0$ $\rightarrow$ ma trận rỗng, đối xứng
            \item Không vuông $\rightarrow$ \texttt{is\_symmetric=0}
        \end{itemize}

        \item \textbf{Module 1 (Đọc file):}
\begin{lstlisting}
readMatrix(path, vector<vector<long long>>&)
\end{lstlisting}

        \item \textbf{Module 2 (Kiểm tra đối xứng):}
        \begin{itemize}
            \item Chỉ duyệt nửa trên:
\begin{lstlisting}
i < j
\end{lstlisting}
            \item Nếu tồn tại:
\begin{lstlisting}
a[i][j] != a[j][i]
\end{lstlisting}
            thì không đối xứng
            \item Độ phức tạp $O(n^2/2)$
        \end{itemize}

        \item \textbf{Module 3 (Ghi file):}
\begin{lstlisting}
is_symmetric=1
\end{lstlisting}
hoặc
\begin{lstlisting}
is_symmetric=0
\end{lstlisting}
    \end{itemize}

    \item \textbf{Output:}
\begin{lstlisting}
is_symmetric=1
\end{lstlisting}
hoặc
\begin{lstlisting}
is_symmetric=0
\end{lstlisting}
\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}

\begin{lstlisting}
#include <iostream>
#include <vector>
#include <fstream>

using namespace std;

bool readMatrix(const string& path, vector<vector<long long>>& a) {
    ifstream fin(path);
    if (!fin) return false;

    int n;
    fin >> n;

    a.assign(n, vector<long long>(n));

    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            fin >> a[i][j];

    return true;
}

bool isSymmetric(const vector<vector<long long>>& a) {
    int n = a.size();

    if (n == 0) return true;
    if (n == 1) return true;

    for (int i = 0; i < n; i++) {
        if ((int)a[i].size() != n) return false;
    }

    for (int i = 0; i < n; i++) {
        for (int j = i + 1; j < n; j++) {
            if (a[i][j] != a[j][i])
                return false;
        }
    }

    return true;
}

bool writeOutput(const string& path, bool ok) {
    ofstream fout(path);
    if (!fout) return false;

    fout << "is_symmetric=" << (ok ? 1 : 0);
    return true;
}

int main() {
    vector<vector<long long>> a;

    if (!readMatrix("matrix.txt", a)) {
        cerr << "Loi mo file input!\n";
        return 1;
    }

    bool ans = isSymmetric(a);

    if (!writeOutput("output.txt", ans)) {
        cerr << "Loi mo file output!\n";
        return 1;
    }

    return 0;
}
\end{lstlisting}
