\section{Tổng hàng và cột ma trận [Cơ bản]}
\subsection{Phân tích \& Thiết kế (M0)}

\begin{itemize}
    \item \textbf{Input:}  
    File \texttt{matrix.txt} chứa:
    \begin{itemize}
        \item Số nguyên $m$
        \item Số nguyên $n$
        \item Ma trận $m \times n$ số nguyên
    \end{itemize}

    \item \textbf{Xử lý:}
    \begin{itemize}
        \item \textbf{Case biên:}
        \begin{itemize}
            \item Ma trận $1 \times 1$
            \item $m=0$ hoặc $n=0$ $\rightarrow$ file output rỗng
        \end{itemize}

        \item \textbf{Module 1 (Đọc file):}
\begin{lstlisting}
readMatrix(path, vector<vector<long long>>&)
\end{lstlisting}

        \item \textbf{Module 2 (Tính tổng hàng):}
\begin{lstlisting}
vector<long long> rowSums(matrix)
\end{lstlisting}

        \item \textbf{Module 3 (Tính tổng cột):}
\begin{lstlisting}
vector<long long> colSums(matrix)
\end{lstlisting}

        \item \textbf{Module 4 (Ghi file):}
        \begin{itemize}
            \item File \texttt{row\_sums.txt}
            \item File \texttt{col\_sums.txt}
            \item Mỗi dòng:
\begin{lstlisting}
i, tong
\end{lstlisting}
        \end{itemize}
    \end{itemize}

    \item \textbf{Output:}
\begin{lstlisting}
0, ...
1, ...
\end{lstlisting}
\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}

\begin{lstlisting}
#include <iostream>
#include <vector>
#include <fstream>

using namespace std;

bool readMatrix(const string& path, vector<vector<long long>>& a) {
    ifstream fin(path);
    if (!fin) return false;

    int m, n;
    fin >> m >> n;

    a.assign(m, vector<long long>(n));

    for (int i = 0; i < m; i++)
        for (int j = 0; j < n; j++)
            fin >> a[i][j];

    return true;
}

vector<long long> rowSums(const vector<vector<long long>>& a) {
    vector<long long> ans;

    for (int i = 0; i < (int)a.size(); i++) {
        long long sum = 0;
        for (int j = 0; j < (int)a[i].size(); j++)
            sum += a[i][j];
        ans.push_back(sum);
    }

    return ans;
}

vector<long long> colSums(const vector<vector<long long>>& a) {
    vector<long long> ans;

    if (a.empty()) return ans;

    int m = a.size();
    int n = a[0].size();

    ans.assign(n, 0);

    for (int j = 0; j < n; j++)
        for (int i = 0; i < m; i++)
            ans[j] += a[i][j];

    return ans;
}

bool writeFile(const string& path, const vector<long long>& v) {
    ofstream fout(path);
    if (!fout) return false;

    for (int i = 0; i < (int)v.size(); i++)
        fout << i << ", " << v[i] << "\n";

    return true;
}

int main() {
    vector<vector<long long>> a;

    if (!readMatrix("matrix.txt", a)) {
        cerr << "Loi mo file input!\n";
        return 1;
    }

    vector<long long> rs = rowSums(a);
    vector<long long> cs = colSums(a);

    if (!writeFile("row_sums.txt", rs)) {
        cerr << "Loi ghi row_sums!\n";
        return 1;
    }

    if (!writeFile("col_sums.txt", cs)) {
        cerr << "Loi ghi col_sums!\n";
        return 1;
    }

    return 0;
}
\end{lstlisting}
