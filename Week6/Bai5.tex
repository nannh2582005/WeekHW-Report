\section{Ma trận đường chéo [Trung bình]}
\subsection{Phân tích \& Thiết kế (M0)}
\begin{itemize}
    \item \textbf{Input:}
    File \texttt{input.txt} chứa:
    \begin{itemize}
        \item Số nguyên $n$
        \item Ma trận vuông $n \times n$
    \end{itemize}

    \item \textbf{Xử lý:}
    \begin{itemize}
        \item \textbf{Case biên:}
        \begin{itemize}
            \item $n = 0 \rightarrow$ không xử lý
            \item $n = 1 \rightarrow$ đường chéo chính = đường chéo phụ = $a[0][0]$
            \item \texttt{is\_diagonal = 1}
        \end{itemize}

        \item \textbf{Module 1 (Đọc file):}
        \begin{itemize}
            \item Hàm \texttt{readMatrix(path, vector<vector<long long>>\&)}
            \item Kiểm tra mở file (\texttt{fail} $\rightarrow$ \texttt{return false})
        \end{itemize}

        \item \textbf{Module 2 (Tính tổng đường chéo):}
        \begin{itemize}
            \item Với mọi $i = 0..n-1$
            \begin{lstlisting}
mainSum += a[i][i]
antiSum += a[i][n - 1 - i]
            \end{lstlisting}
        \end{itemize}

        \item \textbf{Module 3 (Kiểm tra ma trận đường chéo):}
        \begin{itemize}
            \item Với mọi $i \ne j$ thì $a[i][j] = 0$
            \item Nếu tồn tại phần tử khác 0 $\rightarrow$ \texttt{is\_diagonal = 0}
        \end{itemize}
    \end{itemize}

    \item \textbf{Output:}
    \begin{lstlisting}
main_diag_sum=<v> / anti_diag_sum=<v> / is_diagonal=<0/1>
    \end{lstlisting}
\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}
\begin{lstlisting}
#include <iostream>
#include <vector>
#include <fstream>

using namespace std;

bool readMatrix(const string& path, vector<vector<long long>>& a) {
    ifstream fin(path);
    if (!fin) return false;

    int n;
    fin >> n;
    a.assign(n, vector<long long>(n));

    for (int i = 0; i < n; ++i)
        for (int j = 0; j < n; ++j)
            fin >> a[i][j];

    return true;
}

long long mainDiagSum(const vector<vector<long long>>& a) {
    int n = a.size();
    long long sum = 0;

    for (int i = 0; i < n; ++i)
        sum += a[i][i];

    return sum;
}

long long antiDiagSum(const vector<vector<long long>>& a) {
    int n = a.size();
    long long sum = 0;

    for (int i = 0; i < n; ++i)
        sum += a[i][n - 1 - i];

    return sum;
}

bool isDiagonalMatrix(const vector<vector<long long>>& a) {
    int n = a.size();

    for (int i = 0; i < n; ++i)
        for (int j = 0; j < n; ++j)
            if (i != j && a[i][j] != 0)
                return false;

    return true;
}

int main() {
    vector<vector<long long>> a;

    if (!readMatrix("input.txt", a)) {
        cerr << "Loi mo file input!\n";
        return 1;
    }

    if (a.empty()) return 0;

    long long mainSum = mainDiagSum(a);
    long long antiSum = antiDiagSum(a);
    bool diagonal = isDiagonalMatrix(a);

    cout << "main_diag_sum=" << mainSum
         << " / anti_diag_sum=" << antiSum
         << " / is_diagonal=" << diagonal;

    return 0;
}
\end{lstlisting}
