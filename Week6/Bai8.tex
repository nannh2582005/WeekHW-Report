\section{Kiểm tra ma trận tam giác [Trung bình]}
\subsection{Phân tích \& Thiết kế (M0)}
\begin{itemize}
    \item \textbf{Input:}
    File \texttt{input.txt} chứa:
    \begin{itemize}
        \item Số nguyên $n$
        \item Ma trận vuông $n \times n$
    \end{itemize}

    \item \textbf{Xử lý:}
    \begin{itemize}
        \item \textbf{Case biên:}
        \begin{itemize}
            \item $n = 0 \rightarrow$ không xử lý
            \item $n = 1 \rightarrow$ vừa là ma trận tam giác trên, vừa là tam giác dưới
        \end{itemize}

        \item \textbf{Module 1 (Đọc file):}
        \begin{itemize}
            \item Hàm \texttt{readMatrix(path, vector<vector<long long>>\&)}
            \item Kiểm tra mở file (\texttt{fail} $\rightarrow$ \texttt{return false})
        \end{itemize}

        \item \textbf{Module 2 (Kiểm tra tam giác trên):}
        \begin{itemize}
            \item Với mọi $i > j$ thì $a[i][j] = 0$
            \item Nếu tồn tại phần tử khác 0 $\rightarrow$ \texttt{is\_upper = 0}
        \end{itemize}

        \item \textbf{Module 3 (Kiểm tra tam giác dưới):}
        \begin{itemize}
            \item Với mọi $i < j$ thì $a[i][j] = 0$
            \item Nếu tồn tại phần tử khác 0 $\rightarrow$ \texttt{is\_lower = 0}
        \end{itemize}
    \end{itemize}

    \item \textbf{Output:}
    \begin{lstlisting}
is_upper=<0/1> / is_lower=<0/1>
    \end{lstlisting}
\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}
\begin{lstlisting}
#include <iostream>
#include <vector>
#include <fstream>

using namespace std;

bool readMatrix(const string& path, vector<vector<long long>>& a) {
    ifstream fin(path);
    if (!fin) return false;

    int n;
    fin >> n;
    a.assign(n, vector<long long>(n));

    for (int i = 0; i < n; ++i)
        for (int j = 0; j < n; ++j)
            fin >> a[i][j];

    return true;
}

bool isUpperTriangular(const vector<vector<long long>>& a) {
    int n = a.size();

    for (int i = 0; i < n; ++i)
        for (int j = 0; j < i; ++j)
            if (a[i][j] != 0)
                return false;

    return true;
}

bool isLowerTriangular(const vector<vector<long long>>& a) {
    int n = a.size();

    for (int i = 0; i < n; ++i)
        for (int j = i + 1; j < n; ++j)
            if (a[i][j] != 0)
                return false;

    return true;
}

int main() {
    vector<vector<long long>> a;

    if (!readMatrix("input.txt", a)) {
        cerr << "Loi mo file input!\n";
        return 1;
    }

    if (a.empty()) return 0;

    bool upper = isUpperTriangular(a);
    bool lower = isLowerTriangular(a);

    cout << "is_upper=" << upper
         << " / is_lower=" << lower;

    return 0;
}
\end{lstlisting}
