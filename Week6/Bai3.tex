\section{Xây dựng prefix sum 2D [Cơ bản]}
\subsection{Phân tích \& Thiết kế (M0)}

\begin{itemize}
    \item \textbf{Input:}  
    File \texttt{matrix.txt} chứa:
    \begin{itemize}
        \item Số nguyên $m$
        \item Số nguyên $n$
        \item Ma trận $m \times n$ số nguyên
    \end{itemize}

    \item \textbf{Xử lý:}
    \begin{itemize}
        \item \textbf{Case biên:}
        \begin{itemize}
            \item $1 \times 1 \rightarrow pre[0][0] = a[0][0]$
            \item $m=0$ hoặc $n=0 \rightarrow$ file output rỗng
        \end{itemize}

        \item \textbf{Module 1 (Đọc file):}
\begin{lstlisting}
readMatrix(path, vector<vector<long long>>&)
\end{lstlisting}

        \item \textbf{Module 2 (Xây dựng prefix sum 2D):}
        \begin{itemize}
            \item Công thức:
\begin{lstlisting}
pre[i][j] = a[i][j]
          + pre[i-1][j]
          + pre[i][j-1]
          - pre[i-1][j-1]
\end{lstlisting}
            \item Nếu $i=0$ thì phần trên bằng $0$
            \item Nếu $j=0$ thì phần trái bằng $0$
        \end{itemize}

        \item \textbf{Module 3 (Ghi file):}
        \begin{itemize}
            \item File \texttt{pre2D.txt}
            \item Mỗi dòng:
\begin{lstlisting}
i j pre[i][j]
\end{lstlisting}
        \end{itemize}
    \end{itemize}

    \item \textbf{Output:}
\begin{lstlisting}
0 0 ...
0 1 ...
...
\end{lstlisting}
\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}

\begin{lstlisting}
#include <iostream>
#include <vector>
#include <fstream>

using namespace std;

typedef vector<vector<long long>> Matrix;

bool readMatrix(const string& path, Matrix& a) {
    ifstream fin(path);
    if (!fin) return false;

    int m, n;
    fin >> m >> n;

    a.assign(m, vector<long long>(n));

    for (int i = 0; i < m; i++)
        for (int j = 0; j < n; j++)
            fin >> a[i][j];

    return true;
}

Matrix buildPre2D(const Matrix& a) {
    int m = a.size();
    Matrix pre;

    if (m == 0) return pre;

    int n = a[0].size();
    pre.assign(m, vector<long long>(n, 0));

    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            long long up = (i > 0) ? pre[i - 1][j] : 0;
            long long left = (j > 0) ? pre[i][j - 1] : 0;
            long long diag = (i > 0 && j > 0) ? pre[i - 1][j - 1] : 0;

            pre[i][j] = a[i][j] + up + left - diag;
        }
    }

    return pre;
}

bool writeFile(const string& path, const Matrix& pre) {
    ofstream fout(path);
    if (!fout) return false;

    for (int i = 0; i < (int)pre.size(); i++) {
        for (int j = 0; j < (int)pre[i].size(); j++) {
            fout << i << " " << j << " " << pre[i][j] << "\n";
        }
    }

    return true;
}

int main() {
    Matrix a;

    if (!readMatrix("matrix.txt", a)) {
        cerr << "Loi mo file input!\n";
        return 1;
    }

    Matrix pre = buildPre2D(a);

    if (!writeFile("pre2D.txt", pre)) {
        cerr << "Loi mo file output!\n";
        return 1;
    }

    return 0;
}
\end{lstlisting}
