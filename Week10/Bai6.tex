\section{Dashboard tổng hợp 1 file [Trung bình]}
\subsection{Phân tích \& Thiết kế (M0)}

\begin{itemize}

    \item \textbf{Input:}

    File \texttt{data.csv} chứa:

    \begin{itemize}
        \item Dòng đầu là header:
\begin{lstlisting}
id,value1,value2,...,valueN
\end{lstlisting}

        \item Các dòng tiếp theo:
\begin{lstlisting}
id,number,number,...
\end{lstlisting}

        \item Missing có thể là:
        \begin{itemize}
            \item ô rỗng
            \item \texttt{NA}
        \end{itemize}
    \end{itemize}

    \item \textbf{Xử lý:}

    \begin{itemize}

        \item \textbf{Case biên:}

        \begin{itemize}
            \item Một cột toàn missing
            \[
                \Rightarrow
                min=max=mean=std=0
            \]

            \item File rỗng
            \[
                \Rightarrow \text{không có dữ liệu}
            \]

            \item Thiếu cột ở một dòng
            \[
                \Rightarrow \text{coi là missing}
            \]
        \end{itemize}

        \item \textbf{Module 1 (Đọc file):}

        \begin{itemize}
            \item Hàm \texttt{readCSV()}
            \item Đọc:
            \begin{itemize}
                \item header
                \item dữ liệu từng cột
            \end{itemize}
        \end{itemize}

        \item \textbf{Module 2 (Parse dữ liệu):}

        \begin{itemize}
            \item Hàm \texttt{parseRow()}
            \item Xử lý missing
        \end{itemize}

        \item \textbf{Module 3 (Tính thống kê):}

        \begin{itemize}
            \item Hàm \texttt{computeStats()}
            \item Với mỗi cột tính:
            \begin{itemize}
                \item count
                \item missing
                \item min
                \item max
                \item mean
                \item std
            \end{itemize}

            \item Dùng \texttt{long double}
            để giảm sai số
        \end{itemize}

        \item \textbf{Module 4 (Ghi dashboard):}

        \begin{itemize}
            \item Hàm \texttt{writeDashboard()}
            \item Mỗi hàng ứng với một cột dữ liệu
        \end{itemize}

    \end{itemize}

    \item \textbf{Output:}

    File \texttt{dashboard.csv}

\begin{lstlisting}
column,count,missing,min,max,mean,std
...
\end{lstlisting}

    Dùng:
\begin{lstlisting}
setprecision(6)
\end{lstlisting}

\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}

\begin{lstlisting}
#include <iostream>
#include <fstream>
#include <sstream>
#include <vector>
#include <string>
#include <iomanip>
#include <cmath>

using namespace std;

struct ColumnStats {

    string name;

    int count;
    int missing;

    double minVal;
    double maxVal;

    double mean;
    double stdev;
};

vector<string> splitCSV(
    const string& line
) {
    vector<string> parts;

    stringstream ss(line);

    string token;

    while (getline(ss, token, ',')) {
        parts.push_back(token);
    }

    return parts;
}

bool readCSV(
    const string& path,
    vector<string>& headers,
    vector<vector<string>>& rows
) {
    ifstream fin(path);

    if (!fin) return false;

    string line;

    if (!getline(fin, line)) {
        return true;
    }

    headers = splitCSV(line);

    while (getline(fin, line)) {

        rows.push_back(
            splitCSV(line)
        );
    }

    return true;
}

void parseRow(
    const vector<string>& row,
    vector<vector<double>>& cols,
    vector<vector<bool>>& missing
) {
    int totalCols = cols.size();

    for (int i = 0; i < totalCols; ++i) {

        int idx = i + 1;

        if (idx >= (int)row.size()
            || row[idx].empty()
            || row[idx] == "NA") {

            cols[i].push_back(0);

            missing[i].push_back(true);
        }
        else {

            cols[i].push_back(
                stod(row[idx])
            );

            missing[i].push_back(false);
        }
    }
}

ColumnStats computeStats(
    const string& name,
    const vector<double>& values,
    const vector<bool>& missing
) {
    ColumnStats s;

    s.name = name;

    s.count = 0;
    s.missing = 0;

    long double sum = 0;

    bool first = true;

    for (size_t i = 0; i < values.size(); ++i) {

        if (missing[i]) {

            ++s.missing;
        }
        else {

            ++s.count;

            sum += values[i];

            if (first) {

                s.minVal = values[i];
                s.maxVal = values[i];

                first = false;
            }
            else {

                if (values[i] < s.minVal) {
                    s.minVal = values[i];
                }

                if (values[i] > s.maxVal) {
                    s.maxVal = values[i];
                }
            }
        }
    }

    if (s.count == 0) {

        s.minVal = 0;
        s.maxVal = 0;

        s.mean = 0;
        s.stdev = 0;

        return s;
    }

    s.mean = sum / s.count;

    long double variance = 0;

    for (size_t i = 0; i < values.size(); ++i) {

        if (!missing[i]) {

            variance +=
                (values[i] - s.mean)
                *
                (values[i] - s.mean);
        }
    }

    variance /= s.count;

    s.stdev = sqrt(variance);

    return s;
}

bool writeDashboard(
    const string& path,
    const vector<ColumnStats>& stats
) {
    ofstream fout(path);

    if (!fout) return false;

    fout << fixed << setprecision(6);

    fout << "column,count,missing,min,max,mean,std\n";

    for (size_t i = 0; i < stats.size(); ++i) {

        fout << stats[i].name << ","
             << stats[i].count << ","
             << stats[i].missing << ","
             << stats[i].minVal << ","
             << stats[i].maxVal << ","
             << stats[i].mean << ","
             << stats[i].stdev << "\n";
    }

    return true;
}

int main() {

    vector<string> headers;

    vector<vector<string>> rows;

    if (!readCSV(
        "data.csv",
        headers,
        rows
    )) {
        cerr << "Loi mo file input!\n";
        return 1;
    }

    if (headers.size() <= 1) {

        cerr << "Khong co cot du lieu!\n";
        return 1;
    }

    int numCols = headers.size() - 1;

    vector<vector<double>> cols(numCols);

    vector<vector<bool>> missing(numCols);

    for (size_t i = 0; i < rows.size(); ++i) {

        parseRow(
            rows[i],
            cols,
            missing
        );
    }

    vector<ColumnStats> stats;

    for (int i = 0; i < numCols; ++i) {

        stats.push_back(

            computeStats(
                headers[i + 1],
                cols[i],
                missing[i]
            )
        );
    }

    if (!writeDashboard(
        "dashboard.csv",
        stats
    )) {
        cerr << "Loi mo file output!\n";
        return 1;
    }

    return 0;
}
\end{lstlisting}
