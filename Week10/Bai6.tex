\section{Phần tử xuất hiện nhiều nhất trong đoạn [Trung bình]}
\subsection{Phân tích \& Thiết kế (M0)}
    \begin{itemize}
    \item \textbf{Input:}
    File \texttt{array.txt} chứa:
    \begin{itemize}
        \item Số nguyên $n$
        \item $n$ số thực
    \end{itemize}

    \item \textbf{Xử lý:}
        \begin{itemize}
            \item \textbf{Case biên:}
            \begin{itemize}
                \item $n = 0 \rightarrow$ in \texttt{EMPTY}
                \item $n = 1 \rightarrow \min = \max = \text{mean} = a[0]$
            \end{itemize}
            
            \item \textbf{Module 1 (Đọc file):}
            \begin{itemize}
                \item Hàm \texttt{readArray(path, vector<double>\&)}
                \item Kiểm tra mở file (\text{fail} $\rightarrow$ \texttt{cerr} + \texttt{return false})
            \end{itemize}
            
            \item \textbf{Module 2 (Tính mean):}
            \begin{itemize}
                \item Dùng \texttt{long double} để tránh sai số
            \end{itemize}
            
            \item \textbf{Module 3 (Tìm min, max):}
            \begin{itemize}
                \item Duyệt mảng 1 lần
            \end{itemize}
        \end{itemize}

        \item \textbf{Output:}
        \begin{lstlisting}
            min=<f> / max=<f> / mean=<f>\end{lstlisting} 
        (với \texttt{setprecision(4)})
    \end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}
    \begin{lstlisting}
    #include <iostream>
    #include <vector>
    #include <fstream>

    using namespace std;

    int main() {
        ifstream fin_arr("array.txt"), fin_q("queries.txt");
        ofstream fout("output.txt");
        if (!fin_arr || !fin_q || !fout) { cerr << "Loi file!\n"; return 1; }

        int n; fin_arr >> n;
        vector<vector<int>> pre(101, vector<int>(n + 1, 0));
        
        for (int i = 1; i <= n; ++i) {
            int val; fin_arr >> val;
            for (int v = 0; v <= 100; ++v) pre[v][i] = pre[v][i - 1];
            if (val >= 0 && val <= 100) pre[val][i]++;
        }

        int q; fin_q >> q;
        while (q--) {
            int l, r; fin_q >> l >> r;
            if (l < 0 || r >= n || l > r) { fout << "INVALID\n"; continue; }
            
            int max_freq = -1, best_val = -1;
            for (int v = 0; v <= 100; ++v) {
                int freq = pre[v][r + 1] - pre[v][l];
                if (freq > max_freq) {
                    max_freq = freq;
                    best_val = v;
                }
            }
            fout << best_val << "\n";
        }
        return 0;
    }
\end{lstlisting}
