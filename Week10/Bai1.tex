\section{Pipeline M0 hoàn chỉnh: CSV Cleaning [Cơ bản]}
\subsection{Phân tích \& Thiết kế (M0)}

\begin{itemize}

    \item \textbf{Input:}

    File \texttt{data.csv} chứa:
    
    \begin{itemize}
        \item Mỗi dòng gồm: \texttt{id,value}
        \item Giá trị thiếu có thể là \texttt{NA} hoặc rỗng
    \end{itemize}

    \item \textbf{Xử lý:}

    \begin{itemize}

        \item \textbf{Case biên:}

        \begin{itemize}
            \item Tất cả giá trị đều missing
            \[
                \Rightarrow \text{mean} = 0,\quad \text{std} = 0
            \]
            \[
                \Rightarrow \text{mọi } z = 0
            \]

            \item \texttt{std = 0}
            \[
                \Rightarrow z = 0
            \]
        \end{itemize}

        \item \textbf{Module 1 (Đọc CSV):}

        \begin{itemize}
            \item Hàm \texttt{parseCSV(path, ids, values, missing)}
            \item Đọc từng dòng bằng \texttt{getline()}
            \item Kiểm tra missing:
            \begin{itemize}
                \item \texttt{NA}
                \item Chuỗi rỗng
            \end{itemize}
        \end{itemize}

        \item \textbf{Module 2 (Tính thống kê):}

        \begin{itemize}
            \item Hàm \texttt{computeStats()}
            \item Tính:
            \begin{itemize}
                \item count\_valid
                \item count\_missing
                \item mean
                \item std
            \end{itemize}
            \item Bỏ qua các giá trị missing
            \item Dùng \texttt{long double} để giảm sai số
        \end{itemize}

        \item \textbf{Module 3 (Impute + Z-score):}

        \begin{itemize}
            \item Hàm \texttt{imputeAndScore()}
            \item Missing sẽ được thay bằng mean
            \item Công thức:
            \[
                z = \frac{x - mean}{std}
            \]
        \end{itemize}

    \end{itemize}

    \item \textbf{Output:}

    File \texttt{report.csv}

\begin{lstlisting}
id,value_imputed,zscore
\end{lstlisting}

    File \texttt{summary.txt}

\begin{lstlisting}
count_valid=<n>
count_missing=<n>
mean=<f>
std=<f>
\end{lstlisting}

(với \texttt{setprecision(6)})

\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}

\begin{lstlisting}
#include <iostream>
#include <fstream>
#include <sstream>
#include <vector>
#include <iomanip>
#include <cmath>

using namespace std;

bool parseCSV(
    const string& path,
    vector<string>& ids,
    vector<double>& values,
    vector<bool>& missing
) {
    ifstream fin(path);
    if (!fin) return false;

    string line;

    while (getline(fin, line)) {
        if (line.empty()) continue;

        stringstream ss(line);

        string id, valStr;

        getline(ss, id, ',');
        getline(ss, valStr);

        ids.push_back(id);

        if (valStr.empty() || valStr == "NA") {
            values.push_back(0);
            missing.push_back(true);
        } else {
            values.push_back(stod(valStr));
            missing.push_back(false);
        }
    }

    return true;
}

void computeStats(
    const vector<double>& values,
    const vector<bool>& missing,
    int& countValid,
    int& countMissing,
    double& mean,
    double& stdev
) {
    countValid = 0;
    countMissing = 0;

    long double sum = 0;

    for (size_t i = 0; i < values.size(); ++i) {
        if (missing[i]) {
            ++countMissing;
        } else {
            ++countValid;
            sum += values[i];
        }
    }

    if (countValid == 0) {
        mean = 0;
        stdev = 0;
        return;
    }

    mean = sum / countValid;

    long double variance = 0;

    for (size_t i = 0; i < values.size(); ++i) {
        if (!missing[i]) {
            variance += (values[i] - mean) * (values[i] - mean);
        }
    }

    variance /= countValid;

    stdev = sqrt(variance);
}

void imputeAndScore(
    const vector<string>& ids,
    const vector<double>& values,
    const vector<bool>& missing,
    double mean,
    double stdev
) {
    ofstream fout("report.csv");

    fout << fixed << setprecision(6);

    fout << "id,value_imputed,zscore\n";

    for (size_t i = 0; i < ids.size(); ++i) {

        double x = missing[i] ? mean : values[i];

        double z = 0;

        if (stdev != 0) {
            z = (x - mean) / stdev;
        }

        fout << ids[i] << ","
             << x << ","
             << z << "\n";
    }
}

bool writeSummary(
    const string& path,
    int countValid,
    int countMissing,
    double mean,
    double stdev
) {
    ofstream fout(path);

    if (!fout) return false;

    fout << fixed << setprecision(6);

    fout << "count_valid=" << countValid << "\n";
    fout << "count_missing=" << countMissing << "\n";
    fout << "mean=" << mean << "\n";
    fout << "std=" << stdev << "\n";

    return true;
}

int main() {

    vector<string> ids;
    vector<double> values;
    vector<bool> missing;

    if (!parseCSV("data.csv", ids, values, missing)) {
        cerr << "Loi mo file input!\n";
        return 1;
    }

    int countValid, countMissing;

    double mean, stdev;

    computeStats(
        values,
        missing,
        countValid,
        countMissing,
        mean,
        stdev
    );

    imputeAndScore(
        ids,
        values,
        missing,
        mean,
        stdev
    );

    if (!writeSummary(
        "summary.txt",
        countValid,
        countMissing,
        mean,
        stdev
    )) {
        cerr << "Loi mo file output!\n";
        return 1;
    }

    return 0;
}
\end{lstlisting}
