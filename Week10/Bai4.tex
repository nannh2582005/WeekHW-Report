\section{Multi-query prefix sum tổng hợp [Trung bình]}
\subsection{Phân tích \& Thiết kế (M0)}

\begin{itemize}

    \item \textbf{Input:}

    File \texttt{input.txt} chứa:

    \begin{itemize}
        \item Số nguyên \(n\)
        \item \(n\) phần tử của mảng
        \item Số truy vấn \(q\)
        \item Danh sách truy vấn:
\begin{lstlisting}
SUM l r
AVG l r
CNT l r v
\end{lstlisting}
    \end{itemize}

    \item \textbf{Xử lý:}

    \begin{itemize}

        \item \textbf{Case biên:}

        \begin{itemize}
            \item \(l < 0\)
            \item \(r \ge n\)
            \item \(l > r\)

            \[
                \Rightarrow \texttt{INVALID}
            \]
        \end{itemize}

        \item \textbf{Module 1 (Đọc file):}

        \begin{itemize}
            \item Hàm \texttt{readInput()}
            \item Đọc:
            \begin{itemize}
                \item mảng
                \item danh sách truy vấn
            \end{itemize}
        \end{itemize}

        \item \textbf{Module 2 (Xây prefix sum):}

        \begin{itemize}
            \item Hàm \texttt{buildPrefix()}
            \item Prefix kiểu \texttt{long long}
            \[
                pre[i]=a[0]+a[1]+\cdots+a[i]
            \]
        \end{itemize}

        \item \textbf{Module 3 (SUM Query):}

        \begin{itemize}
            \item Hàm \texttt{handleSUM()}
            \item Công thức:
            \[
                sum(l,r)=pre[r]-pre[l-1]
            \]
        \end{itemize}

        \item \textbf{Module 4 (AVG Query):}

        \begin{itemize}
            \item Hàm \texttt{handleAVG()}
            \item Công thức:
            \[
                avg=\frac{sum(l,r)}{r-l+1}
            \]
        \end{itemize}

        \item \textbf{Module 5 (CNT Query):}

        \begin{itemize}
            \item Hàm \texttt{handleCNT()}
            \item Với \(n \le 1000\):
            \begin{itemize}
                \item Duyệt brute force \(O(n)\)
            \end{itemize}
        \end{itemize}

        \item \textbf{Module 6 (Ghi file):}

        \begin{itemize}
            \item Hàm \texttt{writeOutput()}
            \item Ghi kết quả từng truy vấn
        \end{itemize}

    \end{itemize}

    \item \textbf{Output:}

    File \texttt{output.txt}

\begin{lstlisting}
<ket qua query 1>
<ket qua query 2>
...
\end{lstlisting}

    Nếu truy vấn không hợp lệ:
\begin{lstlisting}
INVALID
\end{lstlisting}

\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}

\begin{lstlisting}
#include <iostream>
#include <fstream>
#include <vector>
#include <sstream>
#include <iomanip>

using namespace std;

struct Query {
    string type;
    int l, r;
    long long v;
};

bool readInput(
    const string& path,
    vector<long long>& a,
    vector<Query>& queries
) {
    ifstream fin(path);

    if (!fin) return false;

    int n;

    fin >> n;

    a.resize(n);

    for (int i = 0; i < n; ++i) {
        fin >> a[i];
    }

    int q;

    fin >> q;

    for (int i = 0; i < q; ++i) {

        Query qu;

        fin >> qu.type;

        if (qu.type == "SUM"
            || qu.type == "AVG") {

            fin >> qu.l >> qu.r;

            qu.v = 0;
        }
        else if (qu.type == "CNT") {

            fin >> qu.l >> qu.r >> qu.v;
        }

        queries.push_back(qu);
    }

    return true;
}

vector<long long> buildPrefix(
    const vector<long long>& a
) {
    vector<long long> pre(a.size());

    if (a.empty()) return pre;

    pre[0] = a[0];

    for (size_t i = 1; i < a.size(); ++i) {
        pre[i] = pre[i - 1] + a[i];
    }

    return pre;
}

bool validRange(
    int l,
    int r,
    int n
) {
    return !(l < 0 || r >= n || l > r);
}

string handleSUM(
    const vector<long long>& pre,
    int l,
    int r
) {
    if (!validRange(l, r, pre.size())) {
        return "INVALID";
    }

    long long sum = pre[r];

    if (l > 0) {
        sum -= pre[l - 1];
    }

    return to_string(sum);
}

string handleAVG(
    const vector<long long>& pre,
    int l,
    int r
) {
    if (!validRange(l, r, pre.size())) {
        return "INVALID";
    }

    long long sum = pre[r];

    if (l > 0) {
        sum -= pre[l - 1];
    }

    double avg =
        (double)sum / (r - l + 1);

    stringstream ss;

    ss << fixed << setprecision(6)
       << avg;

    return ss.str();
}

string handleCNT(
    const vector<long long>& a,
    int l,
    int r,
    long long v
) {
    if (!validRange(l, r, a.size())) {
        return "INVALID";
    }

    int cnt = 0;

    for (int i = l; i <= r; ++i) {
        if (a[i] == v) {
            ++cnt;
        }
    }

    return to_string(cnt);
}

bool writeOutput(
    const string& path,
    const vector<string>& results
) {
    ofstream fout(path);

    if (!fout) return false;

    for (size_t i = 0; i < results.size(); ++i) {
        fout << results[i] << "\n";
    }

    return true;
}

int main() {

    vector<long long> a;

    vector<Query> queries;

    if (!readInput(
        "input.txt",
        a,
        queries
    )) {
        cerr << "Loi mo file input!\n";
        return 1;
    }

    vector<long long> pre =
        buildPrefix(a);

    vector<string> results;

    for (size_t i = 0; i < queries.size(); ++i) {

        Query q = queries[i];

        if (q.type == "SUM") {

            results.push_back(
                handleSUM(
                    pre,
                    q.l,
                    q.r
                )
            );
        }
        else if (q.type == "AVG") {

            results.push_back(
                handleAVG(
                    pre,
                    q.l,
                    q.r
                )
            );
        }
        else if (q.type == "CNT") {

            results.push_back(
                handleCNT(
                    a,
                    q.l,
                    q.r,
                    q.v
                )
            );
        }
    }

    if (!writeOutput(
        "output.txt",
        results
    )) {
        cerr << "Loi mo file output!\n";
        return 1;
    }

    return 0;
}
\end{lstlisting}
