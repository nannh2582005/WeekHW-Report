\section{Polynomial records + evaluate [Cơ bản]}
\subsection{Phân tích \& Thiết kế (M0)}

\begin{itemize}

    \item \textbf{Input:}

    File \texttt{polynomials.csv} chứa:

    \begin{itemize}
        \item Mỗi dòng gồm:
\begin{lstlisting}
id,degree,a0,a1,a2,...
\end{lstlisting}
        \item Đa thức:
        \[
            P(x)=a_0+a_1x+a_2x^2+\cdots+a_nx^n
        \]
    \end{itemize}

    File \texttt{x\_values.csv} chứa:

    \begin{itemize}
        \item Mỗi dòng là một giá trị \(x\)
    \end{itemize}

    \item \textbf{Xử lý:}

    \begin{itemize}

        \item \textbf{Case biên:}

        \begin{itemize}
            \item \texttt{degree < 0}
            \[
                \Rightarrow \text{invalid}
            \]

            \item Thiếu hệ số
            \[
                \Rightarrow 0
            \]

            \item File \texttt{x\_values.csv} rỗng
            \[
                \Rightarrow \text{không ghi dữ liệu nhưng vẫn hợp lệ}
            \]
        \end{itemize}

        \item \textbf{Module IO:}

        \begin{itemize}
            \item Hàm \texttt{readPoly()}
            \item Hàm \texttt{readX()}
            \item Hàm \texttt{writeReport()}
        \end{itemize}

        \item \textbf{Module Processing:}

        \begin{itemize}
            \item Hàm \texttt{evalHorner()}
            \item Dùng thuật toán Horner:
            \[
                P(x)=(((a_nx+a_{n-1})x+a_{n-2})\cdots)x+a_0
            \]
        \end{itemize}

    \end{itemize}

    \item \textbf{Output:}

    File \texttt{report.csv}

\begin{lstlisting}
id,x,P(x)
...
---
valid=<n>
invalid=<n>
\end{lstlisting}

    Với mọi số thực:
\begin{lstlisting}
setprecision(6)
\end{lstlisting}

\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}

\begin{lstlisting}
#include <iostream>
#include <fstream>
#include <sstream>
#include <vector>
#include <iomanip>

using namespace std;

struct Polynomial {
    string id;
    int degree;
    vector<double> coef;
    bool valid;
};

bool readPoly(
    const string& path,
    vector<Polynomial>& polys,
    int& validCount,
    int& invalidCount
) {
    ifstream fin(path);

    if (!fin) return false;

    validCount = 0;
    invalidCount = 0;

    string line;

    while (getline(fin, line)) {

        if (line.empty()) continue;

        stringstream ss(line);

        vector<string> parts;

        string token;

        while (getline(ss, token, ',')) {
            parts.push_back(token);
        }

        if (parts.size() < 2) continue;

        Polynomial p;

        p.id = parts[0];

        p.degree = stoi(parts[1]);

        p.valid = true;

        if (p.degree < 0) {
            p.valid = false;
            ++invalidCount;
        } else {

            p.coef.assign(p.degree + 1, 0);

            for (int i = 0; i <= p.degree; ++i) {

                int idx = i + 2;

                if (idx < (int)parts.size()
                    && !parts[idx].empty()) {

                    p.coef[i] = stod(parts[idx]);
                } else {
                    p.coef[i] = 0;
                }
            }

            ++validCount;
        }

        polys.push_back(p);
    }

    return true;
}

bool readX(
    const string& path,
    vector<double>& xs
) {
    ifstream fin(path);

    if (!fin) return false;

    double x;

    while (fin >> x) {
        xs.push_back(x);
    }

    return true;
}

double evalHorner(
    const vector<double>& coef,
    double x
) {
    if (coef.empty()) return 0;

    double result = coef.back();

    for (int i = (int)coef.size() - 2; i >= 0; --i) {
        result = result * x + coef[i];
    }

    return result;
}

bool writeReport(
    const string& path,
    const vector<Polynomial>& polys,
    const vector<double>& xs,
    int validCount,
    int invalidCount
) {
    ofstream fout(path);

    if (!fout) return false;

    fout << fixed << setprecision(6);

    fout << "id,x,P(x)\n";

    for (size_t i = 0; i < polys.size(); ++i) {

        if (!polys[i].valid) continue;

        for (size_t j = 0; j < xs.size(); ++j) {

            double value =
                evalHorner(polys[i].coef, xs[j]);

            fout << polys[i].id << ","
                 << xs[j] << ","
                 << value << "\n";
        }
    }

    fout << "---\n";

    fout << "valid=" << validCount << "\n";
    fout << "invalid=" << invalidCount << "\n";

    return true;
}

int main() {

    vector<Polynomial> polys;

    int validCount, invalidCount;

    if (!readPoly(
        "polynomials.csv",
        polys,
        validCount,
        invalidCount
    )) {
        cerr << "Loi mo file polynomial!\n";
        return 1;
    }

    vector<double> xs;

    if (!readX("x_values.csv", xs)) {
        cerr << "Loi mo file x_values!\n";
        return 1;
    }

    if (!writeReport(
        "report.csv",
        polys,
        xs,
        validCount,
        invalidCount
    )) {
        cerr << "Loi mo file output!\n";
        return 1;
    }

    return 0;
}
\end{lstlisting}
