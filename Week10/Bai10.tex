\subsection{Mục tiêu project}

Xây dựng pipeline xử lý văn bản hoàn chỉnh:

\[
normalize
\rightarrow
tokenize
\rightarrow
TF
\rightarrow
IDF
\rightarrow
TF\text{-}IDF
\rightarrow
similarity
\]

Project đọc 2 file văn bản:

\begin{itemize}
    \item \texttt{text1.txt}
    \item \texttt{text2.txt}
\end{itemize}

Sau đó:

\begin{itemize}
    \item chuẩn hoá dữ liệu
    \item tách token
    \item tính TF-IDF
    \item tìm top-10 từ quan trọng nhất
    \item tính cosine similarity giữa 2 văn bản
\end{itemize}

\subsection{Phân tích \& Thiết kế}

\begin{itemize}

    \item \textbf{Input:}

    \begin{itemize}
        \item Hai file văn bản:
\begin{lstlisting}
text1.txt
text2.txt
\end{lstlisting}
    \end{itemize}

    \item \textbf{Xử lý:}

    \begin{itemize}

        \item Normalize:
        \begin{itemize}
            \item lowercase
            \item bỏ ký tự đặc biệt
        \end{itemize}

        \item Tokenize:
        \begin{itemize}
            \item tách từ bằng stringstream
        \end{itemize}

        \item TF:
        \[
            TF(t)=\frac{\text{count}(t)}{\text{total terms}}
        \]

        \item IDF:
        \[
            IDF(t)=\log \frac{N}{df(t)}
        \]

        \item TF-IDF:
        \[
            TFIDF(t)=TF(t)\times IDF(t)
        \]

        \item Cosine Similarity:
        \[
            sim(A,B)=
            \frac{A\cdot B}
            {|A||B|}
        \]
    \end{itemize}

    \item \textbf{Case biên:}

    \begin{itemize}
        \item File rỗng
        \[
            \Rightarrow similarity=0
        \]

        \item Không có token hợp lệ
        \[
            \Rightarrow top10.csv \text{ chỉ có header}
        \]

        \item Hai file giống hệt nhau
        \[
            \Rightarrow similarity=1
        \]
    \end{itemize}

\end{itemize}

string readText(const string& path) {

    ifstream fin(path);

    string line, text;

    while (getline(fin, line)) {
        text += line + " ";
    }

    return text;
}

#include "text_processing.h"

#include <sstream>
#include <cctype>

string normalize(const string& text) {

    string s = text;

    for (size_t i = 0; i < s.size(); ++i) {

        if (isalnum(s[i])) {
            s[i] = tolower(s[i]);
        }
        else {
            s[i] = ' ';
        }
    }

    return s;
}

vector<string> tokenize(
    const string& text
) {
    vector<string> tokens;

    stringstream ss(text);

    string token;

    while (ss >> token) {
        tokens.push_back(token);
    }

    return tokens;
}
\end{lstlisting}

\subsection{Module 6 -- tfidf.cpp}

\begin{lstlisting}
#include "tfidf.h"

#include <cmath>

map<string,double>
computeTF(
    const vector<string>& tokens
) {
    map<string,double> tf;

    for (size_t i = 0; i < tokens.size(); ++i) {
        tf[tokens[i]]++;
    }

    for (auto& x : tf) {
        x.second /= tokens.size();
    }

    return tf;
}

map<string,double>
computeTFIDF(
    const vector<string>& d1,
    const vector<string>& d2
) {
    map<string,double> tf1 =
        computeTF(d1);

    map<string,double> tfidf;

    for (auto x : tf1) {

        int df = 0;

        for (size_t i = 0; i < d1.size(); ++i) {

            if (d1[i] == x.first) {
                df++;
                break;
            }
        }

        for (size_t i = 0; i < d2.size(); ++i) {

            if (d2[i] == x.first) {
                df++;
                break;
            }
        }

        double idf =
            log(2.0 / df);

        tfidf[x.first] =
            x.second * idf;
    }

    return tfidf;
}
\end{lstlisting}

#include "similarity.h"

#include <map>
#include <cmath>

double cosineSimilarity(
    const vector<string>& d1,
    const vector<string>& d2
) {
    map<string,int> f1, f2;

    for (size_t i = 0; i < d1.size(); ++i) {
        f1[d1[i]]++;
    }

    for (size_t i = 0; i < d2.size(); ++i) {
        f2[d2[i]]++;
    }

    double dot = 0;

    for (auto x : f1) {

        dot +=
            x.second
            *
            f2[x.first];
    }

    double norm1 = 0;
    double norm2 = 0;

    for (auto x : f1) {
        norm1 += x.second * x.second;
    }

    for (auto x : f2) {
        norm2 += x.second * x.second;
    }

    norm1 = sqrt(norm1);
    norm2 = sqrt(norm2);

    if (norm1 == 0 || norm2 == 0) {
        return 0;
    }

    return dot / (norm1 * norm2);
}
\end{lstlisting}

\subsection{Module 9 -- main.cpp}

\begin{lstlisting}
#include <iostream>
#include <fstream>
#include <vector>
#include <algorithm>
#include <iomanip>

#include "io.h"
#include "text_processing.h"
#include "tfidf.h"
#include "similarity.h"

using namespace std;

int main() {

    string text1 =
        readText("text1.txt");

    string text2 =
        readText("text2.txt");

    vector<string> d1 =
        tokenize(
            normalize(text1)
        );

    vector<string> d2 =
        tokenize(
            normalize(text2)
        );

    map<string,double> tfidf =
        computeTFIDF(d1, d2);

    vector<pair<string,double>> arr(
        tfidf.begin(),
        tfidf.end()
    );

    sort(
        arr.begin(),
        arr.end(),

        [](const pair<string,double>& a,
           const pair<string,double>& b) {

            return a.second > b.second;
        }
    );

    ofstream fout("top10.csv");

    fout << fixed << setprecision(6);

    fout << "word,tfidf\n";

    int limit =
        min(10, (int)arr.size());

    for (int i = 0; i < limit; ++i) {

        fout << arr[i].first
             << ","
             << arr[i].second
             << "\n";
    }

    double sim =
        cosineSimilarity(d1, d2);

    ofstream fs("similarity.txt");

    fs << fixed << setprecision(6);

    fs << "similarity="
       << sim
       << "\n";

    return 0;
}
\end{lstlisting}
