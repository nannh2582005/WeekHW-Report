\section{Sliding window multi-stat [Trung bình]}
\subsection{Phân tích \& Thiết kế (M0)}

\begin{itemize}

    \item \textbf{Input:}

    File \texttt{input.txt} chứa:

    \begin{itemize}
        \item Số nguyên \(n\)
        \item \(n\) số thực
        \item Kích thước window \(k\)
    \end{itemize}

    \item \textbf{Xử lý:}

    Với mỗi vị trí \(i\):

    \[
    [i-k+1, i]
    \]

    tính:

    \begin{itemize}
        \item rolling mean
        \item rolling max
        \item rolling min
    \end{itemize}

    \begin{itemize}

        \item \textbf{Case biên:}

        \begin{itemize}
            \item \(n = 0\)
            \[
                \Rightarrow \text{file output chỉ có header}
            \]

            \item \(k \le 0\)
            \[
                \Rightarrow \text{INVALID}
            \]

            \item \(k > n\)
            \[
                \Rightarrow k=n
            \]
        \end{itemize}

        \item \textbf{Module 1 (Đọc file):}

        \begin{itemize}
            \item Hàm \texttt{readInput()}
        \end{itemize}

        \item \textbf{Module 2 (Rolling Mean):}

        \begin{itemize}
            \item Hàm \texttt{rollingMean()}
            \item Dùng sliding window sum:
            \[
                sum = sum - a[i-k] + a[i]
            \]
        \end{itemize}

        \item \textbf{Module 3 (Rolling Max):}

        \begin{itemize}
            \item Hàm \texttt{rollingMax()}
            \item Dùng deque để giữ max
        \end{itemize}

        \item \textbf{Module 4 (Rolling Min):}

        \begin{itemize}
            \item Hàm \texttt{rollingMin()}
            \item Dùng deque để giữ min
        \end{itemize}

        \item \textbf{Module 5 (Ghi file):}

        \begin{itemize}
            \item Hàm \texttt{writeOutput()}
        \end{itemize}

    \end{itemize}

    \item \textbf{Output:}

    File \texttt{multi\_rolling.csv}

\begin{lstlisting}
i,mean_i,max_i,min_i
...
\end{lstlisting}

    Dùng:
\begin{lstlisting}
setprecision(6)
\end{lstlisting}

\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}

\begin{lstlisting}
#include <iostream>
#include <fstream>
#include <vector>
#include <deque>
#include <iomanip>

using namespace std;

bool readInput(
    const string& path,
    vector<double>& a,
    int& k
) {
    ifstream fin(path);

    if (!fin) return false;

    int n;

    fin >> n;

    a.resize(n);

    for (int i = 0; i < n; ++i) {
        fin >> a[i];
    }

    fin >> k;

    return true;
}

vector<double> rollingMean(
    const vector<double>& a,
    int k
) {
    vector<double> result;

    if (a.empty()) return result;

    if (k > (int)a.size()) {
        k = a.size();
    }

    double sum = 0;

    for (int i = 0; i < k; ++i) {
        sum += a[i];
    }

    result.push_back(sum / k);

    for (size_t i = k; i < a.size(); ++i) {

        sum -= a[i - k];

        sum += a[i];

        result.push_back(sum / k);
    }

    return result;
}

vector<double> rollingMax(
    const vector<double>& a,
    int k
) {
    vector<double> result;

    deque<int> dq;

    for (size_t i = 0; i < a.size(); ++i) {

        while (!dq.empty()
            && dq.front() <= (int)i - k) {

            dq.pop_front();
        }

        while (!dq.empty()
            && a[dq.back()] <= a[i]) {

            dq.pop_back();
        }

        dq.push_back(i);

        if (i >= k - 1) {
            result.push_back(a[dq.front()]);
        }
    }

    return result;
}

vector<double> rollingMin(
    const vector<double>& a,
    int k
) {
    vector<double> result;

    deque<int> dq;

    for (size_t i = 0; i < a.size(); ++i) {

        while (!dq.empty()
            && dq.front() <= (int)i - k) {

            dq.pop_front();
        }

        while (!dq.empty()
            && a[dq.back()] >= a[i]) {

            dq.pop_back();
        }

        dq.push_back(i);

        if (i >= k - 1) {
            result.push_back(a[dq.front()]);
        }
    }

    return result;
}

bool writeOutput(
    const string& path,
    const vector<double>& means,
    const vector<double>& maxs,
    const vector<double>& mins
) {
    ofstream fout(path);

    if (!fout) return false;

    fout << fixed << setprecision(6);

    fout << "i,mean_i,max_i,min_i\n";

    for (size_t i = 0; i < means.size(); ++i) {

        fout << i << ","
             << means[i] << ","
             << maxs[i] << ","
             << mins[i] << "\n";
    }

    return true;
}

int main() {

    vector<double> a;

    int k;

    if (!readInput(
        "input.txt",
        a,
        k
    )) {
        cerr << "Loi mo file input!\n";
        return 1;
    }

    if (k <= 0) {

        cerr << "INVALID\n";

        return 1;
    }

    if (k > (int)a.size()) {
        k = a.size();
    }

    vector<double> means =
        rollingMean(a, k);

    vector<double> maxs =
        rollingMax(a, k);

    vector<double> mins =
        rollingMin(a, k);

    if (!writeOutput(
        "multi_rolling.csv",
        means,
        maxs,
        mins
    )) {
        cerr << "Loi mo file output!\n";
        return 1;
    }

    return 0;
}
\end{lstlisting}
