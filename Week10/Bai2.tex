\section{Fraction List tổng hợp [Cơ bản]}
\subsection{Phân tích \& Thiết kế (M0)}

\begin{itemize}

    \item \textbf{Input:}

    File \texttt{fractions.csv} chứa:

    \begin{itemize}
        \item Mỗi dòng gồm:
        \begin{lstlisting}
num,den
\end{lstlisting}
    \end{itemize}

    \item \textbf{Xử lý:}

    \begin{itemize}

        \item \textbf{Case biên:}

        \begin{itemize}
            \item Phân số có mẫu bằng 0
            \[
                \Rightarrow \text{invalid}
            \]

            \item Tử số bằng 0
            \[
                \Rightarrow 0/1
            \]

            \item Không có phân số hợp lệ
            \[
                \Rightarrow \text{sum} = 0/1
            \]
        \end{itemize}

        \item \textbf{Module 1 (Đọc file):}

        \begin{itemize}
            \item Hàm \texttt{readFractions()}
            \item Đọc dữ liệu từ \texttt{fractions.csv}
            \item Lưu vào vector phân số
        \end{itemize}

        \item \textbf{Module 2 (Lọc + Normalize):}

        \begin{itemize}
            \item Hàm \texttt{filterAndNormalize()}
            \item Loại bỏ phân số invalid
            \item Chuẩn hoá:
            \begin{itemize}
                \item Mẫu luôn dương
                \item Rút gọn bằng gcd
                \item \texttt{0/x → 0/1}
            \end{itemize}
        \end{itemize}

        \item \textbf{Module 3 (Sort):}

        \begin{itemize}
            \item Hàm \texttt{sortFractions()}
            \item Sắp xếp tăng dần
            \item So sánh bằng nhân chéo
        \end{itemize}

        \item \textbf{Module 4 (Tính tổng):}

        \begin{itemize}
            \item Hàm \texttt{addFrac()}
            \item Cộng phân số chính xác
            \item Không dùng \texttt{double}
        \end{itemize}

        \item \textbf{Module 5 (Ghi file):}

        \begin{itemize}
            \item Hàm \texttt{writeFractionReport()}
            \item Ghi:
            \begin{itemize}
                \item valid\_count
                \item invalid\_count
                \item sum
                \item danh sách phân số đã sort
            \end{itemize}
        \end{itemize}

    \end{itemize}

    \item \textbf{Output:}

    File \texttt{fraction\_out.csv}

\begin{lstlisting}
valid_count=<n>
invalid_count=<n>
sum=num/den

num,den
...
\end{lstlisting}

\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}

\begin{lstlisting}
#include <iostream>
#include <fstream>
#include <vector>
#include <sstream>
#include <algorithm>

using namespace std;

struct Fraction {
    long long num;
    long long den;
};

long long gcdll(long long a, long long b) {
    a = abs(a);
    b = abs(b);

    while (b != 0) {
        long long r = a % b;
        a = b;
        b = r;
    }

    return a;
}

bool normalize(Fraction& f) {

    if (f.den == 0) return false;

    if (f.num == 0) {
        f.den = 1;
        return true;
    }

    if (f.den < 0) {
        f.num = -f.num;
        f.den = -f.den;
    }

    long long g = gcdll(f.num, f.den);

    f.num /= g;
    f.den /= g;

    return true;
}

bool readFractions(
    const string& path,
    vector<Fraction>& fractions
) {
    ifstream fin(path);

    if (!fin) return false;

    string line;

    while (getline(fin, line)) {

        if (line.empty()) continue;

        stringstream ss(line);

        string sNum, sDen;

        getline(ss, sNum, ',');
        getline(ss, sDen);

        Fraction f;

        f.num = stoll(sNum);
        f.den = stoll(sDen);

        fractions.push_back(f);
    }

    return true;
}

void filterAndNormalize(
    const vector<Fraction>& input,
    vector<Fraction>& validFractions,
    int& invalidCount
) {
    invalidCount = 0;

    for (size_t i = 0; i < input.size(); ++i) {

        Fraction f = input[i];

        if (!normalize(f)) {
            ++invalidCount;
        } else {
            validFractions.push_back(f);
        }
    }
}

bool fracLess(const Fraction& a, const Fraction& b) {
    return a.num * b.den < b.num * a.den;
}

void sortFractions(vector<Fraction>& a) {
    sort(a.begin(), a.end(), fracLess);
}

Fraction addFrac(const Fraction& a, const Fraction& b) {

    Fraction res;

    res.num = a.num * b.den + b.num * a.den;
    res.den = a.den * b.den;

    normalize(res);

    return res;
}

bool writeFractionReport(
    const string& path,
    const vector<Fraction>& fractions,
    int validCount,
    int invalidCount,
    const Fraction& sum
) {
    ofstream fout(path);

    if (!fout) return false;

    fout << "valid_count=" << validCount << "\n";
    fout << "invalid_count=" << invalidCount << "\n";
    fout << "sum=" << sum.num << "/" << sum.den << "\n\n";

    for (size_t i = 0; i < fractions.size(); ++i) {
        fout << fractions[i].num
             << ","
             << fractions[i].den
             << "\n";
    }

    return true;
}

int main() {

    vector<Fraction> inputFractions;

    if (!readFractions("fractions.csv", inputFractions)) {
        cerr << "Loi mo file input!\n";
        return 1;
    }

    vector<Fraction> validFractions;

    int invalidCount;

    filterAndNormalize(
        inputFractions,
        validFractions,
        invalidCount
    );

    sortFractions(validFractions);

    Fraction sum = {0, 1};

    for (size_t i = 0; i < validFractions.size(); ++i) {
        sum = addFrac(sum, validFractions[i]);
    }

    if (!writeFractionReport(
        "fraction_out.csv",
        validFractions,
        validFractions.size(),
        invalidCount,
        sum
    )) {
        cerr << "Loi mo file output!\n";
        return 1;
    }

    return 0;
}
\end{lstlisting}
