\section{Đoạn có tổng bằng K [Trung bình]}
\subsection{Phân tích \& Thiết kế (M0)}
    \begin{itemize}
    \item \textbf{Input:}
    File \texttt{array.txt} chứa:
    \begin{itemize}
        \item Số nguyên $n$
        \item $n$ số thực
    \end{itemize}

    \item \textbf{Xử lý:}
        \begin{itemize}
            \item \textbf{Case biên:}
            \begin{itemize}
                \item $n = 0 \rightarrow$ in \texttt{EMPTY}
                \item $n = 1 \rightarrow \min = \max = \text{mean} = a[0]$
            \end{itemize}
            
            \item \textbf{Module 1 (Đọc file):}
            \begin{itemize}
                \item Hàm \texttt{readArray(path, vector<double>\&)}
                \item Kiểm tra mở file (\text{fail} $\rightarrow$ \texttt{cerr} + \texttt{return false})
            \end{itemize}
            
            \item \textbf{Module 2 (Tính mean):}
            \begin{itemize}
                \item Dùng \texttt{long double} để tránh sai số
            \end{itemize}
            
            \item \textbf{Module 3 (Tìm min, max):}
            \begin{itemize}
                \item Duyệt mảng 1 lần
            \end{itemize}
        \end{itemize}

        \item \textbf{Output:}
        \begin{lstlisting}
            min=<f> / max=<f> / mean=<f>\end{lstlisting} 
        (với \texttt{setprecision(4)})
    \end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}
    \begin{lstlisting}
    #include <iostream>
    #include <vector>
    #include <fstream>
    #include <unordered_map>

    using namespace std;

    int countRangesEqK(const vector<long long>& a, long long K, int& firstL, int& firstR) {
        unordered_map<long long, int> freq;
        unordered_map<long long, int> first_idx;
        long long pre = 0;
        int count = 0;
        firstL = -1; firstR = -1;
        
        freq[0] = 1;
        first_idx[0] = -1;

        for (int i = 0; i < (int)a.size(); ++i) {
            pre += a[i];
            if (freq.count(pre - K)) {
                count += freq[pre - K];
                if (firstL == -1) {
                    firstL = first_idx[pre - K] + 1;
                    firstR = i;
                }
            }
            freq[pre]++;
            if (!first_idx.count(pre)) first_idx[pre] = i;
        }
        return count;
    }

    int main() {
        ifstream fin("input.txt");
        ofstream fout("output.txt");
        if (!fin || !fout) { cerr << "Loi file!\n"; return 1; }

        int n; long long K;
        fin >> n >> K;
        vector<long long> a(n);
        for (int i = 0; i < n; ++i) fin >> a[i];

        int firstL, firstR;
        int count = countRangesEqK(a, K, firstL, firstR);
        fout << "count=" << count << "\n";
        if (count > 0) fout << "l=" << firstL << " r=" << firstR << "\n";

        return 0;
    }
\end{lstlisting}
