\section{Rolling max cửa sổ k [Trung bình]}
\subsection{Phân tích \& Thiết kế (M0)}

\begin{itemize}
    \item \textbf{Input:} File \texttt{input.txt} chứa:
    \begin{itemize}
        \item Số nguyên $n$
        \item $n$ số thực
        \item Số nguyên $k$
    \end{itemize}

    \item \textbf{Xử lý:}
    \begin{itemize}
        \item \textbf{Case biên:}
        \begin{itemize}
            \item $n=0$ $\rightarrow$ ghi chỉ header
            \item $k>n$ $\rightarrow$ không có cửa sổ nào, ghi chỉ header
            \item $k=n$ $\rightarrow$ đúng 1 cửa sổ
            \item $k=1$ $\rightarrow$ max[i] = a[i]
        \end{itemize}

        \item \textbf{Module 1 (Đọc file):}
\begin{lstlisting}
readInput(path, vector<double>&, int&)
\end{lstlisting}

        \item \textbf{Module 2 (Tính rolling max):}
        \begin{itemize}
            \item Dùng \texttt{deque<int>} lưu chỉ số
            \item Bỏ chỉ số ra ngoài cửa sổ
            \item Bỏ các phần tử nhỏ hơn hoặc bằng phần tử mới
            \item Phần tử đầu deque luôn là max
            \item Độ phức tạp $O(n)$
        \end{itemize}

        \item \textbf{Module 3 (Ghi file CSV):}
        \begin{itemize}
            \item File \texttt{rolling\_max.csv}
            \item Header: \texttt{i,max}
        \end{itemize}
    \end{itemize}

    \item \textbf{Output:}
\begin{lstlisting}
i,max
0,...
1,...
\end{lstlisting}
\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}

\begin{lstlisting}
#include <iostream>
#include <vector>
#include <fstream>
#include <iomanip>
#include <deque>
using namespace std;

bool readInput(const string& path, vector<double>& a, int& k) {
    ifstream fin(path);
    if (!fin) return false;

    int n;
    fin >> n;
    a.resize(n);

    for (int i = 0; i < n; i++) fin >> a[i];
    fin >> k;
    return true;
}

vector<double> rollingMax(const vector<double>& a, int k) {
    int n = a.size();
    vector<double> ans;

    if (n == 0 || k <= 0 || k > n) return ans;

    deque<int> dq;

    for (int i = 0; i < n; i++) {
        while (!dq.empty() && dq.front() <= i - k)
            dq.pop_front();

        while (!dq.empty() && a[dq.back()] <= a[i])
            dq.pop_back();

        dq.push_back(i);

        if (i >= k - 1)
            ans.push_back(a[dq.front()]);
    }

    return ans;
}

bool writeCSV(const string& path, const vector<double>& mx) {
    ofstream fout(path);
    if (!fout) return false;

    fout << "i,max\n";
    fout << fixed << setprecision(6);

    for (int i = 0; i < (int)mx.size(); i++) {
        fout << i << "," << mx[i] << "\n";
    }
    return true;
}

int main() {
    vector<double> a;
    int k;

    if (!readInput("input.txt", a, k)) {
        cerr << "Loi mo file input!\n";
        return 1;
    }

    vector<double> mx = rollingMax(a, k);

    if (!writeCSV("rolling_max.csv", mx)) {
        cerr << "Loi mo file output!\n";
        return 1;
    }

    return 0;
}
\end{lstlisting}
