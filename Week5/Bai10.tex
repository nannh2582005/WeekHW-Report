\section{Smallest window chứa tất cả $K$ giá trị phân biệt [Nâng cao]}
\subsection{Phân tích \& Thiết kế (M0)}
\begin{itemize}
    \item \textbf{Input:}
    File \texttt{input.txt} chứa:
    \begin{itemize}
        \item Số nguyên $n$
        \item $n$ số nguyên
        \item Số nguyên $K$
    \end{itemize}

    \item \textbf{Xử lý:}
    \begin{itemize}
        \item \textbf{Case biên:}
        \begin{itemize}
            \item $n = 0 \rightarrow$ \texttt{IMPOSSIBLE}
            \item $K \leq 0 \rightarrow$ \texttt{IMPOSSIBLE}
            \item Nếu $K$ lớn hơn số giá trị phân biệt của mảng $\rightarrow$ \texttt{IMPOSSIBLE}
        \end{itemize}

        \item \textbf{Module 1 (Đọc file):}
        \begin{itemize}
            \item Hàm \texttt{readInput(path, vector<long long>\&, int\&)}
            \item Kiểm tra mở file (\texttt{fail} $\rightarrow$ \texttt{cerr + return false})
        \end{itemize}

        \item \textbf{Module 2 (Tìm cửa sổ ngắn nhất):}
        \begin{itemize}
            \item Dùng kỹ thuật \texttt{Sliding Window}
            \item Dùng \texttt{unordered\_map} lưu tần suất phần tử trong cửa sổ
            \item Mở rộng \texttt{right} cho đến khi đủ $K$ giá trị phân biệt
            \item Thu hẹp từ trái để tối ưu độ dài
            \item Cập nhật đáp án ngắn nhất
        \end{itemize}

        \item \textbf{Module 3 (Ghi file):}
        \begin{itemize}
            \item Nếu tồn tại:
\begin{lstlisting}
len=<v> / l=<i> / r=<j>
\end{lstlisting}
            \item Ngược lại ghi \texttt{IMPOSSIBLE}
        \end{itemize}
    \end{itemize}

    \item \textbf{Output:}
\begin{lstlisting}
len=<v> / l=<i> / r=<j>
\end{lstlisting}
Hoặc:
\begin{lstlisting}
IMPOSSIBLE
\end{lstlisting}
\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}
\begin{lstlisting}
#include <iostream>
#include <vector>
#include <fstream>
#include <unordered_map>
#include <unordered_set>
#include <climits>

using namespace std;

struct WindowAns {
    bool ok;
    int len, l, r;
};

bool readInput(const string& path, vector<long long>& a, int& K) {
    ifstream fin(path);
    if (!fin) return false;

    int n;
    if (fin >> n) {
        a.resize(n);
        for (int i = 0; i < n; ++i) fin >> a[i];
        fin >> K;
    }
    return true;
}

WindowAns smallestWindowKDistinct(const vector<long long>& a, int K) {
    int n = (int)a.size();
    if (n == 0 || K <= 0) return {false, 0, -1, -1};

    unordered_set<long long> st(a.begin(), a.end());
    if (K > (int)st.size()) return {false, 0, -1, -1};

    unordered_map<long long, int> freq;
    int left = 0, distinct = 0;
    int bestLen = INT_MAX, bestL = -1, bestR = -1;

    for (int right = 0; right < n; ++right) {
        if (++freq[a[right]] == 1) ++distinct;

        while (distinct >= K) {
            if (right - left + 1 < bestLen) {
                bestLen = right - left + 1;
                bestL = left;
                bestR = right;
            }

            --freq[a[left]];
            if (freq[a[left]] == 0) {
                freq.erase(a[left]);
                --distinct;
            }
            ++left;
        }
    }

    if (bestLen == INT_MAX) return {false, 0, -1, -1};
    return {true, bestLen, bestL, bestR};
}

bool writeOutput(const string& path, const WindowAns& ans) {
    ofstream fout(path);
    if (!fout) return false;

    if (!ans.ok) {
        fout << "IMPOSSIBLE";
    } else {
        fout << "len=" << ans.len
             << " / l=" << ans.l
             << " / r=" << ans.r;
    }
    return true;
}

int main() {
    vector<long long> a;
    int K;

    if (!readInput("input.txt", a, K)) {
        cerr << "Loi mo file input!\n";
        return 1;
    }

    WindowAns ans = smallestWindowKDistinct(a, K);

    if (!writeOutput("output.txt", ans)) {
        cerr << "Loi mo file output!\n";
        return 1;
    }

    return 0;
}
\end{lstlisting}
