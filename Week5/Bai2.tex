\section{Rolling mean cửa sổ k [Cơ bản]}
\subsection{Phân tích \& Thiết kế (M0)}
\begin{itemize}
    \item \textbf{Input:}
    File \texttt{input.txt} chứa:
    \begin{itemize}
        \item Số nguyên $n$
        \item $n$ số thực
        \item Số nguyên $k$
    \end{itemize}

    \item \textbf{Xử lý:}
    \begin{itemize}
        \item \textbf{Case biên:}
        \begin{itemize}
            \item $n = 0 \rightarrow$ ghi chỉ header
            \item $k > n \rightarrow$ không có cửa sổ nào, ghi chỉ header
            \item $k = n \rightarrow$ đúng 1 cửa sổ
            \item $k = 1 \rightarrow mean[i] = a[i]$
        \end{itemize}

        \item \textbf{Module 1 (Đọc file):}
        \begin{itemize}
            \item Hàm \texttt{readInput(path, vector<double>\&, int\&)}
            \item Kiểm tra mở file (\texttt{fail} $\rightarrow$ \texttt{cerr + return false})
        \end{itemize}

        \item \textbf{Module 2 (Tính rolling mean):}
        \begin{itemize}
            \item Tính tổng cửa sổ đầu tiên
            \item Mỗi bước cập nhật:
\begin{lstlisting}
sum = sum + a[i+k] - a[i]
\end{lstlisting}
            \item Mean = sum / k
        \end{itemize}

        \item \textbf{Module 3 (Ghi file CSV):}
        \begin{itemize}
            \item Ghi file \texttt{rolling.csv}
            \item Header: \texttt{i,mean}
            \item Mỗi dòng: chỉ số và giá trị mean
        \end{itemize}
    \end{itemize}

    \item \textbf{Output:}
\begin{lstlisting}
i,mean
0,...
1,...
\end{lstlisting}
(với \texttt{setprecision(6)})
\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}
\begin{lstlisting}
#include <iostream>
#include <vector>
#include <fstream>
#include <iomanip>

using namespace std;

bool readInput(const string& path, vector<double>& a, int& k) {
    ifstream fin(path);
    if (!fin) return false;

    int n;
    if (fin >> n) {
        a.resize(n);
        for (int i = 0; i < n; ++i) fin >> a[i];
        fin >> k;
    }
    return true;
}

vector<double> rollingMean(const vector<double>& a, int k) {
    int n = (int)a.size();
    vector<double> ans;

    if (n == 0 || k > n || k <= 0) return ans;

    long double sum = 0;
    for (int i = 0; i < k; ++i) sum += a[i];

    ans.push_back((double)(sum / k));

    for (int i = 0; i + k < n; ++i) {
        sum = sum + a[i + k] - a[i];
        ans.push_back((double)(sum / k));
    }

    return ans;
}

bool writeCSV(const string& path, const vector<double>& mean) {
    ofstream fout(path);
    if (!fout) return false;

    fout << "i,mean\n";
    fout << fixed << setprecision(6);

    for (size_t i = 0; i < mean.size(); ++i) {
        fout << i << "," << mean[i] << "\n";
    }
    return true;
}

int main() {
    vector<double> a;
    int k;

    if (!readInput("input.txt", a, k)) {
        cerr << "Loi mo file input!\n";
        return 1;
    }

    vector<double> mean = rollingMean(a, k);

    if (!writeCSV("rolling.csv", mean)) {
        cerr << "Loi mo file output!\n";
        return 1;
    }

    return 0;
}
\end{lstlisting}
