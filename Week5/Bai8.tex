\section{Cặp tổng gần $K$ nhất [Trung bình]}
\subsection{Phân tích \& Thiết kế (M0)}
\begin{itemize}
    \item \textbf{Input:}
    File \texttt{input.txt} chứa:
    \begin{itemize}
        \item Số nguyên $n$
        \item $n$ số nguyên đã sắp xếp tăng
        \item Số nguyên $K$
    \end{itemize}

    \item \textbf{Xử lý:}
    \begin{itemize}
        \item \textbf{Case biên:}
        \begin{itemize}
            \item $n < 2 \rightarrow$ không tồn tại cặp
            \item Nếu nhiều cặp cùng sai khác nhỏ nhất $\rightarrow$ chọn cặp có $i$ nhỏ nhất
        \end{itemize}

        \item \textbf{Module 1 (Đọc file):}
        \begin{itemize}
            \item Hàm \texttt{readInput(path, vector<long long>\&, long long\&)}
            \item Kiểm tra mở file (\texttt{fail} $\rightarrow$ \texttt{cerr + return false})
        \end{itemize}

        \item \textbf{Module 2 (Tìm cặp gần nhất):}
        \begin{itemize}
            \item Dùng kỹ thuật \texttt{Two Pointers}
            \item \texttt{left = 0, right = n-1}
            \item Tính:
\begin{lstlisting}
abs(a[left] + a[right] - K)
\end{lstlisting}
            \item Cập nhật đáp án tốt nhất
            \item Nếu tổng $< K$ tăng \texttt{left}, ngược lại giảm \texttt{right}
        \end{itemize}

        \item \textbf{Module 3 (Ghi file):}
        \begin{itemize}
            \item Ghi:
\begin{lstlisting}
i, j, a[i]+a[j]
\end{lstlisting}
        \end{itemize}
    \end{itemize}

    \item \textbf{Output:}
\begin{lstlisting}
i, j, a[i]+a[j]
\end{lstlisting}
\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}
\begin{lstlisting}
#include <iostream>
#include <vector>
#include <fstream>
#include <cstdlib>
#include <climits>

using namespace std;

struct ClosestAns {
    int i, j;
    long long sum;
};

bool readInput(const string& path, vector<long long>& a, long long& K) {
    ifstream fin(path);
    if (!fin) return false;

    int n;
    if (fin >> n) {
        a.resize(n);
        for (int t = 0; t < n; ++t) fin >> a[t];
        fin >> K;
    }
    return true;
}

ClosestAns closestPair(const vector<long long>& a, long long K) {
    int n = (int)a.size();
    if (n < 2) return {-1, -1, 0};

    int left = 0, right = n - 1;
    long long bestDiff = LLONG_MAX;
    ClosestAns ans = {-1, -1, 0};

    while (left < right) {
        long long sum = a[left] + a[right];
        long long diff = llabs(sum - K);

        if (diff < bestDiff ||
           (diff == bestDiff && left < ans.i)) {
            bestDiff = diff;
            ans = {left, right, sum};
        }

        if (sum < K) ++left;
        else --right;
    }

    return ans;
}

bool writeOutput(const string& path, const ClosestAns& ans) {
    ofstream fout(path);
    if (!fout) return false;

    fout << ans.i << ", "
         << ans.j << ", "
         << ans.sum;
    return true;
}

int main() {
    vector<long long> a;
    long long K;

    if (!readInput("input.txt", a, K)) {
        cerr << "Loi mo file input!\n";
        return 1;
    }

    ClosestAns ans = closestPair(a, K);

    if (!writeOutput("output.txt", ans)) {
        cerr << "Loi mo file output!\n";
        return 1;
    }

    return 0;
}
\end{lstlisting}
