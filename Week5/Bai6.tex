\section{Rolling standard deviation [Trung bình]}
\subsection{Phân tích \& Thiết kế (M0)}

\begin{itemize}
    \item \textbf{Input:} File \texttt{input.txt} chứa:
    \begin{itemize}
        \item Số nguyên $n$
        \item $n$ số thực
        \item Số nguyên $k$
    \end{itemize}

    \item \textbf{Xử lý:}
    \begin{itemize}
        \item \textbf{Case biên:}
        \begin{itemize}
            \item $n=0$ $\rightarrow$ ghi chỉ header
            \item $k>n$ $\rightarrow$ không có cửa sổ nào, ghi chỉ header
            \item $k=n$ $\rightarrow$ đúng 1 cửa sổ
            \item $k=1$ $\rightarrow$ std = 0
        \end{itemize}

        \item \textbf{Module 1 (Đọc file):}
\begin{lstlisting}
readInput(path, vector<double>&, int&)
\end{lstlisting}

        \item \textbf{Module 2 (Tính rolling std):}
        \begin{itemize}
            \item Công thức:
\begin{lstlisting}
Var = E[X^2] - (E[X])^2
\end{lstlisting}
            \item Lưu \texttt{sum} và \texttt{sumSq}
            \item Cập nhật mỗi bước $O(1)$
            \item \texttt{std = sqrt(var)}
        \end{itemize}

        \item \textbf{Module 3 (Ghi file CSV):}
        \begin{itemize}
            \item File \texttt{rolling\_std.csv}
            \item Header: \texttt{i,std}
        \end{itemize}
    \end{itemize}

    \item \textbf{Output:}
\begin{lstlisting}
i,std
0,...
1,...
\end{lstlisting}
\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}

\begin{lstlisting}
#include <iostream>
#include <vector>
#include <fstream>
#include <iomanip>
#include <cmath>
using namespace std;

bool readInput(const string& path, vector<double>& a, int& k) {
    ifstream fin(path);
    if (!fin) return false;

    int n;
    fin >> n;
    a.resize(n);

    for (int i = 0; i < n; i++) fin >> a[i];
    fin >> k;
    return true;
}

vector<double> rollingStd(const vector<double>& a, int k) {
    int n = a.size();
    vector<double> ans;

    if (n == 0 || k <= 0 || k > n) return ans;

    if (k == 1) {
        for (int i = 0; i < n; i++) ans.push_back(0.0);
        return ans;
    }

    long double sum = 0.0, sumSq = 0.0;

    for (int i = 0; i < k; i++) {
        sum += a[i];
        sumSq += a[i] * a[i];
    }

    for (int start = 0; start + k <= n; start++) {
        long double mean = sum / k;
        long double var = sumSq / k - mean * mean;

        if (var < 0) var = 0;
        ans.push_back((double)sqrt((double)var));

        if (start + k < n) {
            double oldVal = a[start];
            double newVal = a[start + k];

            sum += newVal - oldVal;
            sumSq += newVal * newVal - oldVal * oldVal;
        }
    }

    return ans;
}

bool writeCSV(const string& path, const vector<double>& v) {
    ofstream fout(path);
    if (!fout) return false;

    fout << "i,std\n";
    fout << fixed << setprecision(6);

    for (int i = 0; i < (int)v.size(); i++) {
        fout << i << "," << v[i] << "\n";
    }

    return true;
}

int main() {
    vector<double> a;
    int k;

    if (!readInput("input.txt", a, k)) {
        cerr << "Loi mo file input!\n";
        return 1;
    }

    vector<double> ans = rollingStd(a, k);

    if (!writeCSV("rolling_std.csv", ans)) {
        cerr << "Loi mo file output!\n";
        return 1;
    }

    return 0;
}
\end{lstlisting}
