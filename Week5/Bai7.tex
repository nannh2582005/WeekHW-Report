\section{Đoạn không âm dài nhất [Cơ bản]}
\subsection{Phân tích \& Thiết kế (M0)}
\begin{itemize}
    \item \textbf{Input:}
    File \texttt{input.txt} chứa:
    \begin{itemize}
        \item Số nguyên $n$
        \item $n$ số nguyên
    \end{itemize}

    \item \textbf{Xử lý:}
    \begin{itemize}
        \item \textbf{Case biên:}
        \begin{itemize}
            \item $n = 0 \rightarrow max\_len = 0$
            \item Toàn số âm $\rightarrow max\_len = 0$
            \item Nếu tất cả phần tử $\geq 0 \rightarrow$ lấy toàn mảng
        \end{itemize}

        \item \textbf{Module 1 (Đọc file):}
        \begin{itemize}
            \item Hàm \texttt{readInput(path, vector<long long>\&)}
            \item Kiểm tra mở file (\texttt{fail} $\rightarrow$ \texttt{cerr + return false})
        \end{itemize}

        \item \textbf{Module 2 (Tìm đoạn không âm dài nhất):}
        \begin{itemize}
            \item Duyệt mảng 1 lần
            \item Nếu phần tử $\geq 0$ thì mở rộng đoạn hiện tại
            \item Nếu phần tử $< 0$ thì kết thúc đoạn hiện tại
            \item Cập nhật đoạn dài nhất
        \end{itemize}

        \item \textbf{Module 3 (Ghi file):}
        \begin{itemize}
            \item Ghi:
\begin{lstlisting}
max_len=<v> / l=<i> / r=<j>
\end{lstlisting}
        \end{itemize}
    \end{itemize}

    \item \textbf{Output:}
\begin{lstlisting}
max_len=<v> / l=<i> / r=<j>
\end{lstlisting}
\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}
\begin{lstlisting}
#include <iostream>
#include <vector>
#include <fstream>

using namespace std;

struct SegmentAns {
    int len, l, r;
};

bool readInput(const string& path, vector<long long>& a) {
    ifstream fin(path);
    if (!fin) return false;

    int n;
    if (fin >> n) {
        a.resize(n);
        for (int i = 0; i < n; ++i) fin >> a[i];
    }
    return true;
}

SegmentAns longestNonNegative(const vector<long long>& a) {
    int n = (int)a.size();
    if (n == 0) return {0, -1, -1};

    int bestLen = 0, bestL = -1, bestR = -1;
    int curL = -1, curLen = 0;

    for (int i = 0; i < n; ++i) {
        if (a[i] >= 0) {
            if (curLen == 0) curL = i;
            ++curLen;

            if (curLen > bestLen) {
                bestLen = curLen;
                bestL = curL;
                bestR = i;
            }
        } else {
            curLen = 0;
        }
    }

    return {bestLen, bestL, bestR};
}

bool writeOutput(const string& path, const SegmentAns& ans) {
    ofstream fout(path);
    if (!fout) return false;

    fout << "max_len=" << ans.len
         << " / l=" << ans.l
         << " / r=" << ans.r;
    return true;
}

int main() {
    vector<long long> a;

    if (!readInput("input.txt", a)) {
        cerr << "Loi mo file input!\n";
        return 1;
    }

    SegmentAns ans = longestNonNegative(a);

    if (!writeOutput("output.txt", ans)) {
        cerr << "Loi mo file output!\n";
        return 1;
    }

    return 0;
}
\end{lstlisting}
