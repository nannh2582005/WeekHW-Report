\section{Dãy không giảm dài nhất liên tiếp [Cơ bản]}
\subsection{Phân tích \& Thiết kế (M0)}
\begin{itemize}
    \item \textbf{Input:}
    File \texttt{input.txt} chứa:
    \begin{itemize}
        \item Số nguyên $n$
        \item $n$ số nguyên
    \end{itemize}

    \item \textbf{Xử lý:}
    \begin{itemize}
        \item \textbf{Case biên:}
        \begin{itemize}
            \item $n = 0 \rightarrow max\_len = 0$
            \item $n = 1 \rightarrow max\_len = 1$
            \item Nếu toàn mảng không giảm $\rightarrow$ lấy toàn mảng
        \end{itemize}

        \item \textbf{Module 1 (Đọc file):}
        \begin{itemize}
            \item Hàm \texttt{readInput(path, vector<long long>\&)}
            \item Kiểm tra mở file (\texttt{fail} $\rightarrow$ \texttt{cerr + return false})
        \end{itemize}

        \item \textbf{Module 2 (Tìm dãy không giảm dài nhất):}
        \begin{itemize}
            \item Duyệt mảng 1 lần
            \item Nếu:
\begin{lstlisting}
a[i-1] <= a[i]
\end{lstlisting}
            thì tiếp tục mở rộng đoạn hiện tại
            \item Ngược lại bắt đầu đoạn mới
            \item Cập nhật đoạn dài nhất
        \end{itemize}

        \item \textbf{Module 3 (Ghi file):}
        \begin{itemize}
            \item Ghi:
\begin{lstlisting}
max_len=<v> / l=<i> / r=<j>
\end{lstlisting}
        \end{itemize}
    \end{itemize}

    \item \textbf{Output:}
\begin{lstlisting}
max_len=<v> / l=<i> / r=<j>
\end{lstlisting}
\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}
\begin{lstlisting}
#include <iostream>
#include <vector>
#include <fstream>

using namespace std;

struct SegmentAns {
    int len, l, r;
};

bool readInput(const string& path, vector<long long>& a) {
    ifstream fin(path);
    if (!fin) return false;

    int n;
    if (fin >> n) {
        a.resize(n);
        for (int i = 0; i < n; ++i) fin >> a[i];
    }
    return true;
}

SegmentAns longestNonDecreasing(const vector<long long>& a) {
    int n = (int)a.size();
    if (n == 0) return {0, -1, -1};
    if (n == 1) return {1, 0, 0};

    int bestLen = 1, bestL = 0, bestR = 0;
    int curL = 0, curLen = 1;

    for (int i = 1; i < n; ++i) {
        if (a[i - 1] <= a[i]) {
            ++curLen;
        } else {
            curLen = 1;
            curL = i;
        }

        if (curLen > bestLen) {
            bestLen = curLen;
            bestL = curL;
            bestR = i;
        }
    }

    return {bestLen, bestL, bestR};
}

bool writeOutput(const string& path, const SegmentAns& ans) {
    ofstream fout(path);
    if (!fout) return false;

    fout << "max_len=" << ans.len
         << " / l=" << ans.l
         << " / r=" << ans.r;
    return true;
}

int main() {
    vector<long long> a;

    if (!readInput("input.txt", a)) {
        cerr << "Loi mo file input!\n";
        return 1;
    }

    SegmentAns ans = longestNonDecreasing(a);

    if (!writeOutput("output.txt", ans)) {
        cerr << "Loi mo file output!\n";
        return 1;
    }

    return 0;
}
\end{lstlisting}
