\section{Đoạn ngắn nhất tổng $\geq S$ [Cơ bản]}
\subsection{Phân tích \& Thiết kế (M0)}
\begin{itemize}
    \item \textbf{Input:}
    File \texttt{input.txt} chứa:
    \begin{itemize}
        \item Số nguyên $n$
        \item $n$ số nguyên không âm
        \item Số nguyên $S$
    \end{itemize}

    \item \textbf{Xử lý:}
    \begin{itemize}
        \item \textbf{Case biên:}
        \begin{itemize}
            \item $n = 0 \rightarrow$ in \texttt{NONE}
            \item $S = 0 \rightarrow$ in \texttt{len=1 / l=0 / r=0} (nếu tồn tại phần tử)
            \item Không tồn tại đoạn nào có tổng $\geq S \rightarrow$ in \texttt{NONE}
        \end{itemize}

        \item \textbf{Module 1 (Đọc file):}
        \begin{itemize}
            \item Hàm \texttt{readInput(path, vector<long long>\&, long long\&)}
            \item Kiểm tra mở file (\texttt{fail} $\rightarrow$ \texttt{cerr + return false})
        \end{itemize}

        \item \textbf{Module 2 (Tìm đoạn ngắn nhất):}
        \begin{itemize}
            \item Dùng kỹ thuật \texttt{Sliding Window}
            \item Tăng \texttt{right}, cộng dồn tổng
            \item Khi tổng $\geq S$ thì thu hẹp từ trái
        \end{itemize}

        \item \textbf{Module 3 (Ghi file):}
        \begin{itemize}
            \item Nếu tồn tại: ghi \texttt{len=<v> / l=<i> / r=<j>}
            \item Ngược lại ghi \texttt{NONE}
        \end{itemize}
    \end{itemize}

    \item \textbf{Output:}
\begin{lstlisting}
len=<v> / l=<i> / r=<j>
\end{lstlisting}
Hoặc:
\begin{lstlisting}
NONE
\end{lstlisting}
\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}
\begin{lstlisting}
#include <iostream>
#include <vector>
#include <fstream>
#include <climits>

using namespace std;

struct ShortestAns {
    bool ok;
    int len, l, r;
};

bool readInput(const string& path, vector<long long>& a, long long& S) {
    ifstream fin(path);
    if (!fin) return false;
    int n;
    if (fin >> n) {
        a.resize(n);
        for (int i = 0; i < n; ++i) fin >> a[i];
        fin >> S;
    }
    return true;
}

ShortestAns shortestAtLeastS(const vector<long long>& a, long long S) {
    int n = (int)a.size();
    if (n == 0) return {false, 0, -1, -1};
    if (S == 0) return {true, 1, 0, 0};

    long long sum = 0;
    int left = 0;
    int bestLen = INT_MAX, bestL = -1, bestR = -1;

    for (int right = 0; right < n; ++right) {
        sum += a[right];

        while (sum >= S) {
            if (right - left + 1 < bestLen) {
                bestLen = right - left + 1;
                bestL = left;
                bestR = right;
            }
            sum -= a[left];
            ++left;
        }
    }

    if (bestLen == INT_MAX) return {false, 0, -1, -1};
    return {true, bestLen, bestL, bestR};
}

bool writeOutput(const string& path, const ShortestAns& ans) {
    ofstream fout(path);
    if (!fout) return false;

    if (!ans.ok) fout << "NONE";
    else fout << "len=" << ans.len
              << " / l=" << ans.l
              << " / r=" << ans.r;
    return true;
}

int main() {
    vector<long long> a;
    long long S;

    if (!readInput("input.txt", a, S)) {
        cerr << "Loi mo file input!\n";
        return 1;
    }

    ShortestAns ans = shortestAtLeastS(a, S);

    if (!writeOutput("output.txt", ans)) {
        cerr << "Loi mo file output!\n";
        return 1;
    }

    return 0;
}
\end{lstlisting}
