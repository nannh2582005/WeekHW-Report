\section{Đoạn dài nhất không vượt quá $S$ [Cơ bản]}
\subsection{Phân tích \& Thiết kế (M0)}
\begin{itemize}
    \item \textbf{Input:}
    File \texttt{input.txt} chứa:
    \begin{itemize}
        \item Số nguyên $n$
        \item $n$ số nguyên không âm
        \item Số nguyên $S$
    \end{itemize}

    \item \textbf{Xử lý:}
    \begin{itemize}
        \item \textbf{Case biên:}
        \begin{itemize}
            \item $n = 0 \rightarrow max\_len = 0$
            \item Không có đoạn nào hợp lệ $\rightarrow max\_len = 0$
            \item Nếu $a[0] > S$ và mọi phần tử đều lớn hơn $S \rightarrow max\_len = 0$
        \end{itemize}

        \item \textbf{Module 1 (Đọc file):}
        \begin{itemize}
            \item Hàm \texttt{readInput(path, vector<long long>\&, long long\&)}
            \item Kiểm tra mở file (\texttt{fail} $\rightarrow$ \texttt{cerr + return false})
        \end{itemize}

        \item \textbf{Module 2 (Tìm đoạn dài nhất):}
        \begin{itemize}
            \item Dùng kỹ thuật \texttt{Sliding Window}
            \item Tăng \texttt{right}, cộng vào tổng
            \item Khi tổng $> S$ thì tăng \texttt{left} đến khi tổng $\leq S$
            \item Sau mỗi bước cập nhật độ dài lớn nhất
        \end{itemize}

        \item \textbf{Module 3 (Ghi file):}
        \begin{itemize}
            \item Ghi:
\begin{lstlisting}
max_len=<v> / l=<i> / r=<j>
\end{lstlisting}
        \end{itemize}
    \end{itemize}

    \item \textbf{Output:}
\begin{lstlisting}
max_len=<v> / l=<i> / r=<j>
\end{lstlisting}
\end{itemize}

\subsection{Mã nguồn C++11 (Tách module)}
\begin{lstlisting}
#include <iostream>
#include <vector>
#include <fstream>

using namespace std;

struct LongestAns {
    int len, l, r;
};

bool readInput(const string& path, vector<long long>& a, long long& S) {
    ifstream fin(path);
    if (!fin) return false;

    int n;
    if (fin >> n) {
        a.resize(n);
        for (int i = 0; i < n; ++i) fin >> a[i];
        fin >> S;
    }
    return true;
}

LongestAns longestAtMostS(const vector<long long>& a, long long S) {
    int n = (int)a.size();
    if (n == 0) return {0, -1, -1};

    long long sum = 0;
    int left = 0;
    int bestLen = 0, bestL = -1, bestR = -1;

    for (int right = 0; right < n; ++right) {
        sum += a[right];

        while (left <= right && sum > S) {
            sum -= a[left];
            ++left;
        }

        if (sum <= S && right - left + 1 > bestLen) {
            bestLen = right - left + 1;
            bestL = left;
            bestR = right;
        }
    }

    return {bestLen, bestL, bestR};
}

bool writeOutput(const string& path, const LongestAns& ans) {
    ofstream fout(path);
    if (!fout) return false;

    fout << "max_len=" << ans.len
         << " / l=" << ans.l
         << " / r=" << ans.r;
    return true;
}

int main() {
    vector<long long> a;
    long long S;

    if (!readInput("input.txt", a, S)) {
        cerr << "Loi mo file input!\n";
        return 1;
    }

    LongestAns ans = longestAtMostS(a, S);

    if (!writeOutput("output.txt", ans)) {
        cerr << "Loi mo file output!\n";
        return 1;
    }

    return 0;
}
\end{lstlisting}
